23 /**
24  * Depending programming model and compiler, we print compiler hint
25  * for parallelization. For example:
26  *
27  * #pragma ivdep
28  * for(int id=0; id<nodecount; id++) {
29  *
30  * #pragma acc parallel loop
31  * for(int id=0; id<nodecount; id++) {
32  *
33  */
34 void CodegenAccVisitor::print_channel_iteration_block_parallel_hint(BlockType type,
35  const ast::Block* block) {
36  if (info.artificial_cell) {
37  return;
38  }
39 
40  std::ostringstream present_clause;
41  present_clause << "present(inst";
42 
43  if (type == BlockType::NetReceive) {
44  present_clause << ", nrb";
45  } else {
46  present_clause << ", node_index, data, voltage, indexes, thread";
47  if (type == BlockType::Equation) {
48  present_clause << ", vec_rhs, vec_d";
49  }
50  }
51  present_clause << ')';
52  printer->fmt_line("nrn_pragma_acc(parallel loop {} async(nt->stream_id) if(nt->compute_gpu))",
53  present_clause.str());
54  printer->add_line("nrn_pragma_omp(target teams distribute parallel for if(nt->compute_gpu))");
55 }

75 void CodegenAccVisitor::print_memory_allocation_routine() const {
76  // memory for artificial cells should be allocated on CPU
77  if (info.artificial_cell) {
78  CodegenCoreneuronCppVisitor::print_memory_allocation_routine();
79  return;
80  }
81  printer->add_newline(2);
82  printer->push_block(
83  "static inline void* mem_alloc(size_t num, size_t size, size_t alignment = 16)");
84  printer->add_multi_line(R"CODE(
85  void* ptr;
86  cudaMallocManaged(&ptr, num*size);
87  cudaMemset(ptr, 0, num*size);
88  return ptr;
89  )CODE");
90  printer->pop_block();
91 
92  printer->add_newline(2);
93  printer->push_block("static inline void mem_free(void* ptr)");
94  printer->add_line("cudaFree(ptr);");
95  printer->pop_block();
96 }

98 /**
99  * OpenACC kernels running on GPU doesn't support `abort()`. CUDA/OpenACC supports
100  * `assert()` in device kernel that can be used for similar purpose. Also, `printf`
101  * is supported on device.
102  *
103  * @todo : we need to implement proper error handling mechanism to propogate errors
104  * from GPU to CPU. For example, error code can be returned like original
105  * neuron implementation. For now we use `assert(0==1)` pattern which is
106  * used for OpenACC.
107  */
108 void CodegenAccVisitor::print_abort_routine() const {
109  printer->add_newline(2);
110  printer->push_block("static inline void coreneuron_abort()");
111  printer->add_line(R"(printf("Error : Issue while running OpenACC kernel \n");)");
112  printer->add_line("assert(0==1);");
113  printer->pop_block();
114 }

151 /**
152  * `INITIAL` block from `NET_RECEIVE` generates `net_init` function. The `net_init`
153  * function pointer is registered with the coreneuron and called from the CPU.
154  * As the data is on GPU, we need to launch `net_init` on the GPU.
155  *
156  * \todo: With the current code structure for NMODL and MOD2C, we use `serial`
157  * construct to launch serial kernels. This is during initialization
158  * but still inefficient. This should be improved when we drop MOD2C.
159  */

163  printer->add_line("{");

174 void CodegenAccVisitor::print_nrn_cur_matrix_shadow_update() {
175  auto rhs_op = operator_for_rhs();
176  auto d_op = operator_for_d();
177  if (info.point_process) {
178  print_atomic_reduction_pragma();
179  }
180  printer->fmt_line("vec_rhs[node_id] {} rhs;", rhs_op);
181  if (info.point_process) {
182  print_atomic_reduction_pragma();
183  }
184  printer->fmt_line("vec_d[node_id] {} g;", d_op);

269 void CodegenAccVisitor::print_instance_struct_transfer_routines(
270  std::vector<std::string> const& ptr_members) {
271  if (info.artificial_cell) {
272  return;
273  }
274  printer->fmt_push_block(
275  "static inline void copy_instance_to_device(NrnThread* nt, Memb_list* ml, {} const* inst)",
276  instance_struct());
277  printer->push_block("if (!nt->compute_gpu)");
278  printer->add_line("return;");
279  printer->pop_block();
280  printer->fmt_line("auto tmp = *inst;");
281  printer->add_line("auto* d_inst = cnrn_target_is_present(inst);");
282  printer->push_block("if (!d_inst)");
283  printer->add_line("d_inst = cnrn_target_copyin(inst);");
284  printer->pop_block();
285  for (auto const& ptr_mem: ptr_members) {
286  printer->fmt_line("tmp.{0} = cnrn_target_deviceptr(tmp.{0});", ptr_mem);
287  }
288  printer->add_multi_line(R"CODE(
289  cnrn_target_memcpy_to_device(d_inst, &tmp);
290  auto* d_ml = cnrn_target_deviceptr(ml);
291  void* d_inst_void = d_inst;
292  cnrn_target_memcpy_to_device(&(d_ml->instance), &d_inst_void);
293  )CODE");
294  printer->pop_block(); // copy_instance_to_device
295  printer->add_newline();
296 
297  printer->fmt_push_block("static inline void delete_instance_from_device({}* inst)",
298  instance_struct());
299  printer->push_block("if (cnrn_target_is_present(inst))");
300  printer->add_line("cnrn_target_delete(inst);");
301  printer->pop_block();
302  printer->pop_block(); // delete_instance_from_device
303  printer->add_newline();
304 }
