34 /** Reindent nmodl text for text-to-text comparison
35  *
36  * Nmodl constructs defined in test database has extra leading whitespace.
37  * This is done for readability reason in nmodl_constructs.cpp. For example,
38  * we have following nmodl text with 8 leading whitespaces:
39 
40 
41  NEURON {
42  RANGE x
43  }
44 
45  * We convert above paragraph to:
46 
47 NEURON {
48  RANGE x
49 }
50 
51  * i.e. we get first non-empty line and count number of leading whitespaces (X).
52  * Then for every sub-sequent line, we remove first X characters (assuming those
53  * all are whitespaces). This is done because when ast is transformed back to
54  * nmodl, the nmodl output is without "extra" whitespaces in the provided input.
55  */
56 
57 std::string reindent_text(const std::string& text) {
58  std::string indented_text;
59  int num_whitespaces = 0;
60  bool flag = false;
61  std::string line;
62  std::stringstream stream(text);
63 
64  while (std::getline(stream, line)) {
65  if (!line.empty()) {
66  /// count whitespaces for first non-empty line only
67  if (!flag) {
68  flag = true;
69  num_whitespaces = count_leading_spaces(line);
70  }
71 
72  /// make sure we don't remove non whitespaces characters
73  if (!is_empty(line.substr(0, num_whitespaces))) {
74  throw std::runtime_error("Test nmodl input not correctly formatted");
75  }
76 
77  line.erase(0, num_whitespaces);
78  indented_text += line;
79  }
80  /// discard empty lines at very beginning
81  if (!stream.eof() && flag) {
82  indented_text += "\n";
83  }
84  }
85  return indented_text;
86 }
