36 void EmbeddedPythonLoader::load_libraries() {
37  const auto pylib_env = std::getenv("NMODL_PYLIB");
38  if (!pylib_env) {
39  logger->critical("NMODL_PYLIB environment variable must be set to load embedded python");
40  throw std::runtime_error("NMODL_PYLIB not set");
41  }
42  const auto dlopen_opts = RTLD_NOW | RTLD_GLOBAL;
43  dlerror(); // reset old error conditions
44  pylib_handle = dlopen(pylib_env, dlopen_opts);
45  if (!pylib_handle) {
46  const auto errstr = dlerror();
47  logger->critical("Tried but failed to load {}", pylib_env);
48  logger->critical(errstr);
49  throw std::runtime_error("Failed to dlopen");
50  }
51 
52  // This code is imported from PyBind11 because this is primarly in details for internal usage
53  // License of PyBind11 is BSD-style
54  {
55  std::string compiled_ver = fmt::format("{}.{}", PY_MAJOR_VERSION, PY_MINOR_VERSION);
56  const char* (*fun)(void) = (const char* (*) (void) ) dlsym(pylib_handle, "Py_GetVersion");
57  if (fun == nullptr) {
58  logger->critical("Unable to find the function `Py_GetVersion`");
59  throw std::runtime_error("Unable to find the function `Py_GetVersion`");
60  }
61  const char* runtime_ver = fun();
62  std::size_t len = compiled_ver.size();
63  if (std::strncmp(runtime_ver, compiled_ver.c_str(), len) != 0 ||
64  (runtime_ver[len] >= '0' && runtime_ver[len] <= '9')) {
65  logger->critical(
66  "nmodl has been compiled with python {} and is being run with python {}",
67  compiled_ver,
68  runtime_ver);
69  throw std::runtime_error("Python version mismatch between compile-time and runtime.");
70  }
71  }
72 
73  if (std::getenv("NMODLHOME") == nullptr) {
74  logger->critical("NMODLHOME environment variable must be set to load embedded python");
75  throw std::runtime_error("NMODLHOME not set");
76  }
77  auto pybind_wraplib_env = fs::path(std::getenv("NMODLHOME")) / "lib" / "libpywrapper";
78  pybind_wraplib_env.concat(CMakeInfo::SHARED_LIBRARY_SUFFIX);
79  if (!fs::exists(pybind_wraplib_env)) {
80  logger->critical("NMODLHOME doesn't contain libpywrapper{} library",
81  CMakeInfo::SHARED_LIBRARY_SUFFIX);
82  throw std::runtime_error("NMODLHOME doesn't have lib/libpywrapper library");
83  }
84  pybind_wrapper_handle = dlopen(pybind_wraplib_env.c_str(), dlopen_opts);
85  if (!pybind_wrapper_handle) {
86  const auto errstr = dlerror();
87  logger->critical("Tried but failed to load {}", pybind_wraplib_env.string());
88  logger->critical(errstr);
89  throw std::runtime_error("Failed to dlopen");
90  }
91 }
