34 /// Helper for creating C codegen visitor
35 std::shared_ptr<CodegenCoreneuronCppVisitor> create_coreneuron_cpp_visitor(
36  const std::shared_ptr<ast::Program>& ast,
37  const std::string& /* text */,
38  std::stringstream& ss) {
39  /// construct symbol table
40  SymtabVisitor().visit_program(*ast);
41 
42  /// run all necessary pass
43  InlineVisitor().visit_program(*ast);
44  NeuronSolveVisitor().visit_program(*ast);
45  SolveBlockVisitor().visit_program(*ast);
46 
47  /// create C code generation visitor
48  auto cv = std::make_shared<CodegenCoreneuronCppVisitor>("temp.mod", ss, "double", false);
49  cv->setup(*ast);
50  return cv;
51 }

80 /// print entire code
81 std::string get_coreneuron_cpp_code(const std::string& nmodl_text,
82  const bool generate_gpu_code = false) {
83  const auto& ast = NmodlDriver().parse_string(nmodl_text);
84  std::stringstream ss;
85  if (generate_gpu_code) {
86  auto accvisitor = create_acc_visitor(ast, nmodl_text, ss);
87  accvisitor->visit_program(*ast);
88  } else {
89  auto cvisitor = create_coreneuron_cpp_visitor(ast, nmodl_text, ss);
90  cvisitor->visit_program(*ast);
91  }
92  return reindent_text(ss.str());
93 }

304 std::string get_instance_structure(std::string nmodl_text) {
305  // parse mod file & print mechanism structure
306  auto const ast = NmodlDriver{}.parse_string(nmodl_text);
307  // add implicit arguments
308  ImplicitArgumentVisitor{}.visit_program(*ast);
309  // update the symbol table for PerfVisitor
310  SymtabVisitor{}.visit_program(*ast);
311  // we need the read/write counts so the codegen knows whether or not
312  // global variables are used
313  PerfVisitor{}.visit_program(*ast);
314  // setup codegen
315  std::stringstream ss{};
316  CodegenCoreneuronCppVisitor cv{"temp.mod", ss, "double", false};
317  cv.setup(*ast);
318  cv.print_mechanism_range_var_structure(true);
319  return ss.str();
320 }
