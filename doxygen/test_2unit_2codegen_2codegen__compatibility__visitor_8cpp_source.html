26 /// Return true if it failed and false otherwise
27 bool runCompatibilityVisitor(const std::string& nmodl_text) {
28  const auto& ast = NmodlDriver().parse_string(nmodl_text);
29  SymtabVisitor().visit_program(*ast);
30  PerfVisitor().visit_program(*ast);
31  return CodegenCompatibilityVisitor().find_unhandled_ast_nodes(*ast);
32 }
