210 void CodegenNeuronCppVisitor::print_function_or_procedure(const ast::Block& node,
211  const std::string& name) {
212  printer->add_newline(2);
213  print_function_declaration(node, name);
214  printer->add_text(" ");
215  printer->push_block();
216 
217  // function requires return variable declaration
218  if (node.is_function_block()) {
219  auto type = default_float_data_type();
220  printer->fmt_line("{} ret_{} = 0.0;", type, name);
221  } else {
222  printer->fmt_line("int ret_{} = 0;", name);
223  }
224 
225  if (node.is_procedure_block() || node.is_function_block()) {
226  const auto& parameters = node.get_parameters();
227  auto result = std::find_if(parameters.begin(), parameters.end(), [](auto var) {
228  return var.get()->get_node_name() == "v";
229  });
230  if (result == parameters.end()) {
231  printer->fmt_line("auto v = inst.{}[id];", naming::VOLTAGE_UNUSED_VARIABLE);
232  }
233  }
234 
235  print_statement_block(*node.get_statement_block(), false, false);
236  printer->fmt_line("return ret_{};", name);
237  printer->pop_block();
238 }

276 void CodegenNeuronCppVisitor::print_hoc_py_wrapper_function_body(
277  const ast::Block* function_or_procedure_block,
278  InterpreterWrapper wrapper_type) {
279  if (info.point_process && wrapper_type == InterpreterWrapper::Python) {
280  return;
281  }
282  const auto block_name = function_or_procedure_block->get_node_name();
283  if (info.point_process) {
284  printer->fmt_push_block("static double _hoc_{}(void* _vptr)", block_name);
285  } else if (wrapper_type == InterpreterWrapper::HOC) {
286  printer->fmt_push_block("static void _hoc_{}(void)", block_name);
287  } else {
288  printer->fmt_push_block("static double _npy_{}(Prop* _prop)", block_name);
289  }
290  printer->add_multi_line(R"CODE(
291  double _r{};
292  Datum* _ppvar;
293  Datum* _thread;
294  NrnThread* _nt;
295  )CODE");
296  if (info.point_process) {
297  printer->add_multi_line(R"CODE(
298  auto* const _pnt = static_cast<Point_process*>(_vptr);
299  auto* const _p = _pnt->prop;
300  if (!_p) {
301  hoc_execerror("POINT_PROCESS data instance not valid", NULL);
302  }
303  _nrn_mechanism_cache_instance _ml_real{_p};
304  auto* const _ml = &_ml_real;
305  size_t const id{};
306  _ppvar = _nrn_mechanism_access_dparam(_p);
307  _thread = _extcall_thread.data();
308  _nt = static_cast<NrnThread*>(_pnt->_vnt);
309  )CODE");
310  } else if (wrapper_type == InterpreterWrapper::HOC) {
311  if (program_symtab->lookup(block_name)->has_all_properties(NmodlType::use_range_ptr_var)) {
312  printer->push_block("if (!_prop_id)");
313  printer->fmt_line(
314  "hoc_execerror(\"No data for {}_{}. Requires prior call to setdata_{} and that the "
315  "specified mechanism instance still be in existence.\", NULL);",
316  function_or_procedure_block->get_node_name(),
317  info.mod_suffix,
318  info.mod_suffix);
319  printer->pop_block();
320  printer->add_line("Prop* _local_prop = _extcall_prop;");
321  } else {
322  printer->add_line("Prop* _local_prop = _prop_id ? _extcall_prop : nullptr;");
323  }
324  printer->add_multi_line(R"CODE(
325  _nrn_mechanism_cache_instance _ml_real{_local_prop};
326  auto* const _ml = &_ml_real;
327  size_t const id{};
328  _ppvar = _local_prop ? _nrn_mechanism_access_dparam(_local_prop) : nullptr;
329  _thread = _extcall_thread.data();
330  _nt = nrn_threads;
331  )CODE");
332  } else { // wrapper_type == InterpreterWrapper::Python
333  printer->add_multi_line(R"CODE(
334  _nrn_mechanism_cache_instance _ml_real{_prop};
335  auto* const _ml = &_ml_real;
336  size_t const id{};
337  _ppvar = _nrn_mechanism_access_dparam(_prop);
338  _thread = _extcall_thread.data();
339  _nt = nrn_threads;
340  )CODE");
341  }
342  printer->fmt_line("auto inst = make_instance_{}(_ml_real);", info.mod_suffix);
343  if (info.function_uses_table(block_name)) {
344  printer->fmt_line("_check_{}({})", block_name, internal_method_arguments());
345  }
346  const auto get_func_call_str = [&]() {
347  const auto params = function_or_procedure_block->get_parameters();
348  const auto func_proc_name = block_name + "_" + info.mod_suffix;
349  auto func_call = fmt::format("{}({}", func_proc_name, internal_method_arguments());
350  for (int i = 0; i < params.size(); ++i) {
351  func_call.append(fmt::format(", *getarg({})", i + 1));
352  }
353  func_call.append(")");
354  return func_call;
355  };
356  if (function_or_procedure_block->is_function_block()) {
357  printer->add_indent();
358  printer->fmt_text("_r = {};", get_func_call_str());
359  printer->add_newline();
360  } else {
361  printer->add_line("_r = 1.;");
362  printer->fmt_line("{};", get_func_call_str());
363  }
364  if (info.point_process || wrapper_type != InterpreterWrapper::HOC) {
365  printer->add_line("return(_r);");
366  } else if (wrapper_type == InterpreterWrapper::HOC) {
367  printer->add_line("hoc_retpushx(_r);");
368  }
369  printer->pop_block();
370 }

590 std::string CodegenNeuronCppVisitor::get_variable_name(const std::string& name,
591  bool use_instance) const {
592  const std::string& varname = update_if_ion_variable_name(name);
593 
594  auto symbol_comparator = [&varname](const SymbolType& sym) {
595  return varname == sym->get_name();
596  };
597 
598  auto index_comparator = [&varname](const IndexVariableInfo& var) {
599  return varname == var.symbol->get_name();
600  };
601 
602  auto thread_comparator = [&varname](const ThreadVariableInfo& var) {
603  return varname == var.symbol->get_name();
604  };
605 
606  if (name == naming::POINT_PROCESS_VARIABLE) {
607  if (printing_net_receive) {
608  // In net_receive blocks, the point process is passed in as an
609  // argument called:
610  return "_pnt";
611  }
612  // The "integer variable" branch will pick up the correct `_ppvar` when
613  // not printing a NET_RECEIVE block.
614  }
615 
616  // float variable
617  auto f = std::find_if(codegen_float_variables.begin(),
618  codegen_float_variables.end(),
619  symbol_comparator);
620  if (f != codegen_float_variables.end()) {
621  return float_variable_name(*f, use_instance);
622  }
623 
624  // integer variable
625  auto i =
626  std::find_if(codegen_int_variables.begin(), codegen_int_variables.end(), index_comparator);
627  if (i != codegen_int_variables.end()) {
628  return int_variable_name(*i, varname, use_instance);
629  }
630 
631  // thread variable
632  auto t = std::find_if(codegen_thread_variables.begin(),
633  codegen_thread_variables.end(),
634  thread_comparator);
635  if (t != codegen_thread_variables.end()) {
636  return thread_variable_name(*t, use_instance);
637  }
638 
639  // global variable
640  auto g = std::find_if(codegen_global_variables.begin(),
641  codegen_global_variables.end(),
642  symbol_comparator);
643  if (g != codegen_global_variables.end()) {
644  return global_variable_name(*g, use_instance);
645  }
646 
647  if (varname == naming::NTHREAD_DT_VARIABLE) {
648  return std::string("_nt->_") + naming::NTHREAD_DT_VARIABLE;
649  }

2023  RenameVisitor vr(name, fmt::format("_args[{}]", i));
