28 /**
29  * Check whether a given SOLVE block solves a PROCEDURE with any of the CVode methods
30  */
31 static bool check_procedure_has_cvode(const std::shared_ptr<const ast::Ast>& solve_node,
32  const std::shared_ptr<const ast::Ast>& procedure_node) {
33  const auto& solve_block = std::dynamic_pointer_cast<const ast::SolveBlock>(solve_node);
34  const auto& method = solve_block->get_method();
35  if (!method) {
36  return false;
37  }
38  const auto& method_name = method->get_node_name();
39 
40  return procedure_node->get_node_name() == solve_block->get_block_name()->get_node_name() &&
41  (method_name == codegen::naming::AFTER_CVODE_METHOD ||
42  method_name == codegen::naming::CVODE_T_METHOD ||
43  method_name == codegen::naming::CVODE_T_V_METHOD);
44 }

46 /**
47  * How symbols are stored in NEURON? See notes written in markdown file.
48  *
49  * Some variables get printed by iterating over symbol table in mod2c.
50  * The example of this is thread variables (and also ions?). In this
51  * case we must have to arrange order if we are going keep compatibility
52  * with NEURON.
53  *
54  * Suppose there are three global variables: bcd, abc, abd, abe
55  * They will be in the 'a' bucket in order:
56  * abe, abd, abc
57  * and in 'b' bucket
58  * bcd
59  * So when we print thread variables, we first have to sort in the opposite
60  * order in which they come and then again order by first character in increasing
61  * order.
62  *
63  * Note that variables in double array do not need this transformation
64  * and it seems like they should just follow definition order.
65  */
66 void CodegenHelperVisitor::sort_with_mod2c_symbol_order(std::vector<SymbolType>& symbols) {
67  /// first sort by global id to get in reverse order
68  std::sort(symbols.begin(),
69  symbols.end(),
70  [](const SymbolType& first, const SymbolType& second) -> bool {
71  return first->get_id() > second->get_id();
72  });
73 
74  /// now order by name (to be same as neuron's bucket)
75  std::sort(symbols.begin(),
76  symbols.end(),
77  [](const SymbolType& first, const SymbolType& second) -> bool {
78  return first->get_name()[0] < second->get_name()[0];
79  });
80 }

83 /**
84  * Find all ions used in mod file
85  */
86 // NOLINTNEXTLINE(readability-function-cognitive-complexity)
87 void CodegenHelperVisitor::find_ion_variables(const ast::Program& node) {
88  // collect all use ion statements
89  const auto& ion_nodes = collect_nodes(node, {AstNodeType::USEION});
90 
91  // ion names, read ion variables and write ion variables
92  std::vector<std::string> ion_vars;
93  std::vector<std::string> read_ion_vars;
94  std::vector<std::string> write_ion_vars;
95  std::map<std::string, double> valences;
96 
97  for (const auto& ion_node: ion_nodes) {
98  const auto& ion = std::dynamic_pointer_cast<const ast::Useion>(ion_node);
99  auto ion_name = ion->get_node_name();
100  ion_vars.push_back(ion_name);
101  for (const auto& var: ion->get_readlist()) {
102  read_ion_vars.push_back(var->get_node_name());
103  }
104  for (const auto& var: ion->get_writelist()) {
105  write_ion_vars.push_back(var->get_node_name());
106  }
107 
108  if (ion->get_valence() != nullptr) {
109  valences[ion_name] = ion->get_valence()->get_value()->to_double();
110  }
111  }
112 
113  /**
114  * Check if given variable belongs to given ion.
115  * For example, eca belongs to ca ion, nai belongs to na ion.
116  * We just check if we exclude first/last char, if that is ion name.
117  */
118  auto ion_variable = [](const std::string& var, const std::string& ion) -> bool {
119  auto len = var.size() - 1;
120  return (var.substr(1, len) == ion || var.substr(0, len) == ion);
121  };
122 
123  /// iterate over all ion types and construct the Ion objects
124  for (auto& ion_name: ion_vars) {
125  Ion ion(ion_name);
126  for (auto& read_var: read_ion_vars) {
127  if (ion_variable(read_var, ion_name)) {
128  ion.reads.push_back(read_var);
129  }
130  }
131  for (auto& write_var: write_ion_vars) {
132  if (ion_variable(write_var, ion_name)) {
133  ion.writes.push_back(write_var);
134  if (ion.is_intra_cell_conc(write_var) || ion.is_extra_cell_conc(write_var)) {
135  ion.need_style = true;
136  info.write_concentration = true;
137  }
138  }
139  }
140  if (auto it = valences.find(ion_name); it != valences.end()) {
141  ion.valence = it->second;
142  }
143 
144  info.ions.push_back(std::move(ion));
145  }
146 
147  /// once ions are populated, we can find all currents
148  auto vars = psymtab->get_variables_with_properties(NmodlType::nonspecific_cur_var);
149  for (auto& var: vars) {
150  info.currents.push_back(var->get_name());
151  }
152  vars = psymtab->get_variables_with_properties(NmodlType::electrode_cur_var);
153  for (auto& var: vars) {
154  info.currents.push_back(var->get_name());
155  }
156  for (auto& ion: info.ions) {
157  for (auto& var: ion.writes) {
158  if (ion.is_ionic_current(var)) {
159  info.currents.push_back(var);
160  }
161  }
162  }
163 
164  /// check if write_conc(...) will be needed
165  for (const auto& ion: info.ions) {
166  for (const auto& var: ion.writes) {
167  if (!ion.is_ionic_current(var) && !ion.is_rev_potential(var)) {
168  info.require_wrote_conc = true;
169  }
170  }
171  }
172 }

174 /**
175  * Find whether or not we need to emit CVODE-related code for NEURON
176  * Notes: we generate CVODE-related code if and only if:
177  * - there is exactly ONE block being SOLVEd
178  * - the block is one of the following types:
179  * - DERIVATIVE
180  * - KINETIC
181  * - PROCEDURE being solved with the `after_cvode`, `cvode_t`, or `cvode_t_v` methods
182  */
183 void CodegenHelperVisitor::check_cvode_codegen(const ast::Program& node) {
184  // find the breakpoint block
185  const auto& breakpoint_nodes = collect_nodes(node, {AstNodeType::BREAKPOINT_BLOCK});
186 
187  // do nothing if there are no BREAKPOINT nodes
188  if (breakpoint_nodes.empty()) {
189  return;
190  }
191 
192  // there can only be one BREAKPOINT block in the entire program
193  assert(breakpoint_nodes.size() == 1);
194 
195  const auto& breakpoint_node = std::dynamic_pointer_cast<const ast::BreakpointBlock>(
196  breakpoint_nodes[0]);
197 
198  // all (global) kinetic/derivative nodes
199  const auto& kinetic_or_derivative_nodes =
200  collect_nodes(node, {AstNodeType::KINETIC_BLOCK, AstNodeType::DERIVATIVE_BLOCK});
201 
202  // all (global) procedure nodes
203  const auto& procedure_nodes = collect_nodes(node, {AstNodeType::PROCEDURE_BLOCK});
204 
205  // find all SOLVE blocks in that BREAKPOINT block
206  const auto& solve_nodes = collect_nodes(*breakpoint_node, {AstNodeType::SOLVE_BLOCK});
207 
208  // check whether any of the SOLVE blocks are solving any PROCEDURE with `after_cvode`,
209  // `cvode_t`, or `cvode_t_v` methods
210  const auto using_cvode = std::any_of(
211  solve_nodes.begin(), solve_nodes.end(), [&procedure_nodes](const auto& solve_node) {
212  return std::any_of(procedure_nodes.begin(),
213  procedure_nodes.end(),
214  [&solve_node](const auto& procedure_node) {
215  return check_procedure_has_cvode(solve_node, procedure_node);
216  });
217  });
218 
219  // only case when we emit CVODE code is if we have exactly one block, and
220  // that block is either a KINETIC/DERIVATIVE with any method, or a
221  // PROCEDURE with `after_cvode` method
222  if (solve_nodes.size() == 1 && (kinetic_or_derivative_nodes.size() || using_cvode)) {
223  logger->debug("Will emit code for CVODE");
224  info.emit_cvode = true;
225  }
226 }

228 /**
229  * Find non-range variables i.e. ones that are not belong to per instance allocation
230  *
231  * Certain variables like pointers, global, parameters are not necessary to be per
232  * instance variables. NEURON apply certain rules to determine which variables become
233  * thread, static or global variables. Here we construct those variables.
234  */
235 void CodegenHelperVisitor::find_non_range_variables() {
236  /**
237  * Top local variables are local variables appear in global scope. All local
238  * variables in program symbol table are in global scope.
239  */
240  info.constant_variables = psymtab->get_variables_with_properties(NmodlType::constant_var);
241  info.top_local_variables = psymtab->get_variables_with_properties(NmodlType::local_var);
242  info.external_variables = psymtab->get_variables_with_properties(NmodlType::extern_var);
243 
244  /**
245  * All global variables remain global if mod file is not marked thread safe.
246  * Otherwise, global variables written at least once gets promoted to thread variables.
247  */
248 
249  std::string variables;
250 
251  auto vars = psymtab->get_variables_with_properties(NmodlType::global_var);
252  for (auto& var: vars) {
253  if (info.vectorize && info.thread_safe && var->get_write_count() > 0) {
254  var->mark_thread_safe();
255  info.thread_variables.push_back(var);
256  info.thread_var_data_size += var->get_length();
257  variables += " " + var->get_name();
258  } else {
259  info.global_variables.push_back(var);
260  }
261  }
262 
263  /**
264  * If parameter is not a range and used only as read variable then it becomes global
265  * variable. To qualify it as thread variable it must be be written at least once and
266  * mod file must be marked as thread safe.
267  * To exclusively get parameters only, we exclude all other variables (in without)
268  * and then sort them with neuron/mod2c order.
269  */
270  // clang-format off
271  auto with = NmodlType::param_assign;
272  auto without = NmodlType::range_var
273  | NmodlType::assigned_definition
274  | NmodlType::global_var
275  | NmodlType::pointer_var
276  | NmodlType::bbcore_pointer_var
277  | NmodlType::read_ion_var
278  | NmodlType::write_ion_var;
279  // clang-format on
280  vars = psymtab->get_variables(with, without);
281  for (auto& var: vars) {
282  // some variables like area and diam are declared in parameter
283  // block but they are not global
284  if (var->get_name() == naming::DIAM_VARIABLE || var->get_name() == naming::AREA_VARIABLE ||
285  var->has_any_property(NmodlType::extern_neuron_variable)) {
286  continue;
287  }
288 
289  // if model is thread safe and if parameter is being written then
290  // those variables should be promoted to thread safe variable
291  if (info.vectorize && info.thread_safe && var->get_write_count() > 0) {
292  var->mark_thread_safe();
293  info.thread_variables.push_back(var);
294  info.thread_var_data_size += var->get_length();
295  } else {
296  info.global_variables.push_back(var);
297  }
298  }
299  sort_with_mod2c_symbol_order(info.thread_variables);
300 
301  /**
302  * \todo Below we calculate thread related id and sizes. This will
303  * need to do from global analysis pass as here we are handling
304  * top local variables, global variables, derivimplicit method.
305  * There might be more use cases with other solver methods.
306  */
307 
308  /**
309  * If derivimplicit is used, then first three thread ids get assigned to:
310  * 1st thread is used for: deriv_advance
311  * 2nd thread is used for: dith
312  * 3rd thread is used for: newtonspace
313  *
314  * slist and dlist represent the offsets for prime variables used. For
315  * euler or derivimplicit methods its always first number.
316  */
317  if (info.derivimplicit_used()) {
318  info.derivimplicit_var_thread_id = 0;
319  info.thread_data_index = 3;
320  info.derivimplicit_list_num = 1;
321  info.thread_callback_register = true;
322  }
323 
324  /// next thread id is allocated for top local variables
325  if (info.vectorize && !info.top_local_variables.empty()) {
326  info.top_local_thread_id = info.thread_data_index++;
327  info.thread_callback_register = true;
328  }
329 
330  /// next thread id is allocated for thread promoted variables
331  if (info.vectorize && !info.thread_variables.empty()) {
332  info.thread_var_thread_id = info.thread_data_index++;
333  info.thread_callback_register = true;
334  }
335 
336  /// find total size of local variables in global scope
337  for (auto& var: info.top_local_variables) {
338  info.top_local_thread_size += var->get_length();
339  }
340 
341  /// find number of prime variables and total size
342  auto primes = psymtab->get_variables_with_properties(NmodlType::prime_name);
343  info.num_primes = static_cast<int>(primes.size());
344  for (auto& variable: primes) {
345  info.primes_size += variable->get_length();
346  }
347 
348  /// find pointer or bbcore pointer variables
349  // clang-format off
350  auto properties = NmodlType::pointer_var
351  | NmodlType::bbcore_pointer_var;
352  // clang-format on
353  info.pointer_variables = psymtab->get_variables_with_properties(properties);
354 
355  /// find RANDOM variables
356  // clang-format off
357  properties = NmodlType::random_var;
358  // clang-format on
359  info.random_variables = psymtab->get_variables_with_properties(properties);
360 
361  // find special variables like diam, area
362  // clang-format off
363  properties = NmodlType::assigned_definition
364  | NmodlType::param_assign;
365  vars = psymtab->get_variables_with_properties(properties);
366  for (auto& var : vars) {
367  if (var->get_name() == naming::AREA_VARIABLE) {
368  info.area_used = true;
369  }
370  if (var->get_name() == naming::DIAM_VARIABLE) {
371  info.diam_used = true;
372  }
373  }
374  // clang-format on
375 }

377 /**
378  * Find range variables i.e. ones that are belong to per instance allocation
379  *
380  * In order to be compatible with NEURON, we need to print all variables in
381  * exact order as NEURON/MOD2C implementation. This is important because memory
382  * for all variables is allocated in single 1-D array with certain offset
383  * for each variable. The order of variables determine the offset and hence
384  * they must be in same order as NEURON.
385  *
386  * Here is how order is determined into NEURON/MOD2C implementation:
387  *
388  * First, following three lists are created
389  * - variables with parameter and range property (List 1)
390  * - variables with state and range property (List 2)
391  * - variables with assigned and range property (List 3)
392  *
393  * Once created, we remove some variables due to the following criteria:
394  * - In NEURON/MOD2C implementation, we remove variables with NRNPRANGEIN
395  * or NRNPRANGEOUT type
396  * - So who has NRNPRANGEIN and NRNPRANGEOUT type? these are USEION read
397  * or write variables that are not ionic currents.
398  * - This is the reason for mod files CaDynamics_E2.mod or cal_mig.mod, ica variable
399  * is printed earlier in the list but other variables like cai, cao don't appear
400  * in same order.
401  *
402  * Finally we create 4th list:
403  * - variables with assigned property and not in the previous 3 lists
404  *
405  * We now print the variables in following order:
406  *
407  * - List 1 i.e. range + parameter variables are printed first
408  * - List 3 i.e. range + assigned variables are printed next
409  * - List 2 i.e. range + state variables are printed next
410  * - List 4 i.e. assigned and ion variables not present in the previous 3 lists
411  *
412  * NOTE:
413  * - State variables also have the property `assigned_definition` but these variables
414  * are not from ASSIGNED block.
415  * - Variable can not be range as well as state, it's redeclaration error
416  * - Variable can be parameter as well as range. Without range, parameter
417  * is considered as global variable i.e. one value for all instances.
418  * - If a variable is only defined as RANGE and not in assigned or parameter
419  * or state block then it's not printed.
420  * - Note that a variable property is different than the variable type. For example,
421  * if variable has range property, it doesn't mean the variable is declared as RANGE.
422  * Other variables like STATE and ASSIGNED block variables also get range property
423  * without being explicitly declared as RANGE in the mod file.
424  * - Also, there is difference between declaration order vs. definition order. For
425  * example, POINTER variable in NEURON block is just declaration and doesn't
426  * determine the order in which they will get printed. Below we query symbol table
427  * and order all instance variables into certain order.
428  */
429 void CodegenHelperVisitor::find_range_variables() {
430  /// comparator to decide the order based on definition
431  auto comparator = [](const SymbolType& first, const SymbolType& second) -> bool {
432  return first->get_definition_order() < second->get_definition_order();
433  };
434 
435  /// from symbols vector `vars`, remove all ion variables which are not ionic currents
436  auto remove_non_ioncur_vars = [](SymbolVectorType& vars, const CodegenInfo& info) -> void {
437  vars.erase(std::remove_if(vars.begin(),
438  vars.end(),
439  [&](SymbolType& s) {
440  return info.is_ion_variable(s->get_name()) &&
441  !info.is_ionic_current(s->get_name());
442  }),
443  vars.end());
444  };
445 
446  /// if `secondary` vector contains any symbol that exist in the `primary` then remove it
447  auto remove_var_exist = [](SymbolVectorType& primary, SymbolVectorType& secondary) -> void {
448  secondary.erase(std::remove_if(secondary.begin(),
449  secondary.end(),
450  [&primary](const SymbolType& tosearch) {
451  return std::find_if(primary.begin(),
452  primary.end(),
453  // compare by symbol name
454  [&tosearch](
455  const SymbolType& symbol) {
456  return tosearch->get_name() ==
457  symbol->get_name();
458  }) != primary.end();
459  }),
460  secondary.end());
461  };
462 
463  /**
464  * First come parameters which are range variables.
465  */
466  // clang-format off
467  auto with = NmodlType::range_var
468  | NmodlType::param_assign;
469  auto without = NmodlType::global_var
470  | NmodlType::pointer_var
471  | NmodlType::bbcore_pointer_var
472  | NmodlType::state_var;
473 
474  // clang-format on
475  info.range_parameter_vars = psymtab->get_variables(with, without);
476  remove_non_ioncur_vars(info.range_parameter_vars, info);
477  std::sort(info.range_parameter_vars.begin(), info.range_parameter_vars.end(), comparator);
478 
479  /**
480  * Second come assigned variables which are range variables.
481  */
482  // clang-format off
483  with = NmodlType::range_var
484  | NmodlType::assigned_definition;
485  without = NmodlType::global_var
486  | NmodlType::pointer_var
487  | NmodlType::bbcore_pointer_var
488  | NmodlType::state_var
489  | NmodlType::param_assign;
490 
491  // clang-format on
492  info.range_assigned_vars = psymtab->get_variables(with, without);
493  remove_non_ioncur_vars(info.range_assigned_vars, info);
494  std::sort(info.range_assigned_vars.begin(), info.range_assigned_vars.end(), comparator);
495 
496 
497  /**
498  * Third come state variables. All state variables are kind of range by default.
499  * Note that some mod files like CaDynamics_E2.mod use cai as state variable which
500  * appear in USEION read/write list. These variables are not considered in this
501  * variables because non ionic-current variables are removed and printed later.
502  */
503  // clang-format off
504  with = NmodlType::state_var;
505  without = NmodlType::global_var
506  | NmodlType::pointer_var
507  | NmodlType::bbcore_pointer_var;
508 
509  // clang-format on
510  info.state_vars = psymtab->get_variables(with, without);
511  std::sort(info.state_vars.begin(), info.state_vars.end(), comparator);
512 
513  /// range_state_vars is copy of state variables but without non ionic current variables
514  info.range_state_vars = info.state_vars;
515  remove_non_ioncur_vars(info.range_state_vars, info);
516 
517  /// Remaining variables are assigned and ion variables which are not in the previous 3 lists
518 
519  // clang-format off
520  with = NmodlType::assigned_definition
521  | NmodlType::read_ion_var
522  | NmodlType::write_ion_var;
523  without = NmodlType::global_var
524  | NmodlType::pointer_var
525  | NmodlType::bbcore_pointer_var
526  | NmodlType::extern_neuron_variable;
527  // clang-format on
528  const auto& variables = psymtab->get_variables_with_properties(with, false);
529  for (const auto& variable: variables) {
530  if (!variable->has_any_property(without)) {
531  info.assigned_vars.push_back(variable);
532  }
533  }
534 
535  /// make sure that variables already present in previous lists
536  /// are removed to avoid any duplication
537  remove_var_exist(info.range_parameter_vars, info.assigned_vars);
538  remove_var_exist(info.range_assigned_vars, info.assigned_vars);
539  remove_var_exist(info.range_state_vars, info.assigned_vars);
540 
541  /// sort variables with their definition order
542  std::sort(info.assigned_vars.begin(), info.assigned_vars.end(), comparator);
543 }

706 /**
707  * Visit statement block and find prime symbols appear in derivative block
708  *
709  * Equation statements in derivative block has prime on the lhs. The order
710  * of primes could be different that declaration state block. Also, not all
711  * state variables need to appear in equation block. In this case, we want
712  * to find out the the primes in the order of equation definition. This is
713  * just to keep the same order as neuron implementation.
714  *
715  * The primes are already solved and replaced by Dstate or name. And hence
716  * we need to check if the lhs variable is derived from prime name. If it's
717  * Dstate then we have to lookup state to find out corresponding symbol. This
718  * is because prime_variables_by_order should contain state variable name and
719  * not the one replaced by solver pass.
720  *
721  * \todo AST can have duplicate DERIVATIVE blocks if a mod file uses SOLVE
722  * statements in its INITIAL block (e.g. in case of kinetic schemes using
723  * `STEADYSTATE sparse` solver). Such duplicated DERIVATIVE blocks could
724  * be removed by `SolveBlockVisitor`, or we have to avoid visiting them
725  * here. See e.g. SH_na8st.mod test and original reduced_dentate .mod.
726  */
727 void CodegenHelperVisitor::visit_statement_block(const ast::StatementBlock& node) {
728  const auto& statements = node.get_statements();
729  for (auto& statement: statements) {
730  statement->accept(*this);
731  if (under_derivative_block && assign_lhs &&
732  (assign_lhs->is_name() || assign_lhs->is_var_name())) {
733  auto name = assign_lhs->get_node_name();
734  auto symbol = psymtab->lookup(name);
735  if (symbol != nullptr) {
736  auto is_prime = symbol->has_any_property(NmodlType::prime_name);
737  auto from_state = symbol->has_any_status(Status::from_state);
738  if (is_prime || from_state) {
739  if (from_state) {
740  symbol = psymtab->lookup(name.substr(1, name.size()));
741  }
742  // See the \todo note above.
743  if (std::find_if(info.prime_variables_by_order.begin(),
744  info.prime_variables_by_order.end(),
745  [&](auto const& sym) {
746  return sym->get_name() == symbol->get_name();
747  }) == info.prime_variables_by_order.end()) {
748  info.prime_variables_by_order.push_back(symbol);
749  info.num_equations++;
750  }
751  }
752  }
753  }
754  assign_lhs = nullptr;
755  }
756 }
