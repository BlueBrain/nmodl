80 /**
81  * \details Certain statements like unit, comment, solve can/need to be skipped
82  * during code generation. Note that solve block is wrapped in expression
83  * statement and hence we have to check inner expression. It's also true
84  * for the initial block defined inside net receive block.
85  */
86 bool CodegenCppVisitor::statement_to_skip(const Statement& node) {
87  // clang-format off
88  if (node.is_unit_state()
89  || node.is_line_comment()
90  || node.is_block_comment()
91  || node.is_solve_block()
92  || node.is_conductance_hint()
93  || node.is_table_statement()) {
94  return true;
95  }
96  // clang-format on
97  if (node.is_expression_statement()) {
98  auto expression = dynamic_cast<const ExpressionStatement*>(&node)->get_expression();
99  if (expression->is_solve_block()) {
100  return true;
101  }
102  if (expression->is_initial_block()) {
103  return true;
104  }
105  }
106  return false;
107 }

210 /**
211  * \details Statements like if, else etc. don't need semicolon at the end.
212  * (Note that it's valid to have "extraneous" semicolon). Also, statement
213  * block can appear as statement using expression statement which need to
214  * be inspected.
215  */
216 bool CodegenCppVisitor::need_semicolon(const Statement& node) {
217  // clang-format off
218  if (node.is_if_statement()
219  || node.is_else_if_statement()
220  || node.is_else_statement()
221  || node.is_from_statement()
222  || node.is_verbatim()
223  || node.is_conductance_hint()
224  || node.is_while_statement()
225  || node.is_protect_statement()
226  || node.is_mutex_lock()
227  || node.is_mutex_unlock()) {
228  return false;
229  }
230  if (node.is_expression_statement()) {
231  auto expression = dynamic_cast<const ExpressionStatement&>(node).get_expression();
232  if (expression->is_statement_block()
233  || expression->is_eigen_newton_solver_block()
234  || expression->is_eigen_linear_solver_block()
235  || expression->is_solution_expression()
236  || expression->is_for_netcon()) {
237  return false;
238  }
239  }
240  // clang-format on
241  return true;
242 }

244 /**
245  * \details Depending upon the block type, we have to print read/write ion variables
246  * during code generation. Depending on block/procedure being printed, this
247  * method return statements as vector. As different code backends could have
248  * different variable names, we rely on backend-specific read_ion_variable_name
249  * and write_ion_variable_name method which will be overloaded.
250  */
251 std::vector<std::string> CodegenCppVisitor::ion_read_statements(BlockType type) const {
252  if (optimize_ion_variable_copies()) {
253  return ion_read_statements_optimized(type);
254  }
255  std::vector<std::string> statements;
256  for (const auto& ion: info.ions) {
257  auto name = ion.name;
258  for (const auto& var: ion.reads) {
259  auto const iter = std::find(ion.implicit_reads.begin(), ion.implicit_reads.end(), var);
260  if (iter != ion.implicit_reads.end()) {
261  continue;
262  }
263  auto variable_names = read_ion_variable_name(var);
264  auto first = get_variable_name(variable_names.first);
265  auto second = get_variable_name(variable_names.second);
266  statements.push_back(fmt::format("{} = {};", first, second));
267  }
268  for (const auto& var: ion.writes) {
269  if (ion.is_ionic_conc(var)) {
270  auto variables = read_ion_variable_name(var);
271  auto first = get_variable_name(variables.first);
272  auto second = get_variable_name(variables.second);
273  statements.push_back(fmt::format("{} = {};", first, second));
274  }
275  }
276  }
277  return statements;
278 }

296 // NOLINTNEXTLINE(readability-function-cognitive-complexity)
297 std::vector<ShadowUseStatement> CodegenCppVisitor::ion_write_statements(BlockType type) {
298  std::vector<ShadowUseStatement> statements;
299  for (const auto& ion: info.ions) {
300  std::string concentration;
301  for (const auto& var: ion.writes) {
302  auto variable_names = write_ion_variable_name(var);
303  if (ion.is_ionic_current(var)) {
304  if (type == BlockType::Equation) {
305  auto current = breakpoint_current(var);
306  auto lhs = variable_names.first;
307  auto op = "+=";
308  auto rhs = get_variable_name(current);
309  if (info.point_process) {
310  auto area = get_variable_name(naming::NODE_AREA_VARIABLE);
311  rhs += fmt::format("*(1.e2/{})", area);
312  }
313  statements.push_back(ShadowUseStatement{lhs, op, rhs});
314  }
315  } else {
316  if (!ion.is_rev_potential(var)) {
317  concentration = var;
318  }
319  auto lhs = variable_names.first;
320  auto op = "=";
321  auto rhs = get_variable_name(variable_names.second);
322  statements.push_back(ShadowUseStatement{lhs, op, rhs});
323  }
324  }
325 
326  if (type == BlockType::Initial && !concentration.empty()) {
327  append_conc_write_statements(statements, ion, concentration);
328  }
329  }
330  return statements;
331 }

333 /**
334  * If mechanisms dependency level execution is enabled then certain updates
335  * like ionic current contributions needs to be atomically updated. In this
336  * case we first update current mechanism's shadow vector and then add statement
337  * to queue that will be used in reduction queue.
338  */
339 std::string CodegenCppVisitor::process_shadow_update_statement(const ShadowUseStatement& statement,
340  BlockType /* type */) {
341  // when there is no operator or rhs then that statement doesn't need shadow update
342  if (statement.op.empty() && statement.rhs.empty()) {
343  auto text = statement.lhs + ";";
344  return text;
345  }
346 
347  // return regular statement
348  auto lhs = get_variable_name(statement.lhs);
349  auto text = fmt::format("{} {} {};", lhs, statement.op, statement.rhs);
350  return text;
351 }

354 /**
355  * \details Current variable used in breakpoint block could be local variable.
356  * In this case, neuron has already renamed the variable name by prepending
357  * "_l". In our implementation, the variable could have been renamed by
358  * one of the pass. And hence, we search all local variables and check if
359  * the variable is renamed. Note that we have to look into the symbol table
360  * of statement block and not breakpoint.
361  */
362 std::string CodegenCppVisitor::breakpoint_current(std::string current) const {
363  auto breakpoint = info.breakpoint_node;
364  if (breakpoint == nullptr) {
365  return current;
366  }
367  auto symtab = breakpoint->get_statement_block()->get_symbol_table();
368  auto variables = symtab->get_variables_with_properties(NmodlType::local_var);
369  for (const auto& var: variables) {
370  auto renamed_name = var->get_name();
371  auto original_name = var->get_original_name();
372  if (current == original_name) {
373  current = renamed_name;
374  break;
375  }
376  }
377  return current;
378 }

381 /**
382  * \details Depending programming model and compiler, we print compiler hint
383  * for parallelization. For example:
384  *
385  * \code
386  * #pragma omp simd
387  * for(int id = 0; id < nodecount; id++) {
388  *
389  * #pragma acc parallel loop
390  * for(int id = 0; id < nodecount; id++) {
391  * \endcode
392  */
393 void CodegenCppVisitor::print_channel_iteration_block_parallel_hint(BlockType /* type */,
394  const ast::Block* block) {
395  // ivdep allows SIMD parallelisation of a block/loop but doesn't provide
396  // a standard mechanism for atomics. Also, even with openmp 5.0, openmp
397  // atomics do not enable vectorisation under "omp simd" (gives compiler
398  // error with gcc < 9 if atomic and simd pragmas are nested). So, emit
399  // ivdep/simd pragma when no MUTEXLOCK/MUTEXUNLOCK/PROTECT statements
400  // are used in the given block.
401  std::vector<std::shared_ptr<const ast::Ast>> nodes;
402  if (block) {
403  nodes = collect_nodes(*block,
404  {ast::AstNodeType::PROTECT_STATEMENT,
405  ast::AstNodeType::MUTEX_LOCK,
406  ast::AstNodeType::MUTEX_UNLOCK});
407  }
408  if (nodes.empty()) {
409  printer->add_line("#pragma omp simd");
410  printer->add_line("#pragma ivdep");
411  }
412 }

512  return {codegen::naming::RANDOM_FUNCTIONS_MAPPING[name], true};
513  }
514  return {name, false};
515  };
516  auto [function_name, is_random_function] = get_renamed_random_function(name);
517 
518  if (defined_method(name)) {
519  function_name = method_name(name);
520  }
521 
522  if (is_nrn_pointing(name)) {
523  print_nrn_pointing(node);
524  return;
525  }
526 
527  if (is_net_send(name)) {
528  print_net_send_call(node);
529  return;
530  }
531 
532  if (is_net_move(name)) {
533  print_net_move_call(node);
534  return;
535  }
536 
537  if (is_net_event(name)) {
538  print_net_event_call(node);
539  return;
540  }
541 
542  if (is_function_table_call(name)) {
543  print_function_table_call(node);
544  return;
545  }
546 
547  const auto& arguments = node.get_arguments();
548  printer->add_text(function_name, '(');

552  printer->add_text(internal_args);

731 bool CodegenCppVisitor::is_functor_const(const ast::StatementBlock& variable_block,
732  const ast::StatementBlock& functor_block) {
733  // Create complete_block with both variable declarations (done in variable_block) and solver
734  // part (done in functor_block) to be able to run the SymtabVisitor and DefUseAnalyzeVisitor
735  // then and get the proper DUChains for the variables defined in the variable_block
736  ast::StatementBlock complete_block(functor_block);
737  // Typically variable_block has only one statement, a statement containing the declaration
738  // of the local variables
739  for (const auto& statement: variable_block.get_statements()) {
740  complete_block.insert_statement(complete_block.get_statements().begin(), statement);
741  }
742 
743  // Create Symbol Table for complete_block
744  auto model_symbol_table = std::make_shared<symtab::ModelSymbolTable>();
745  SymtabVisitor(model_symbol_table.get()).visit_statement_block(complete_block);
746  // Initialize DefUseAnalyzeVisitor to generate the DUChains for the variables defined in the
747  // variable_block
748  DefUseAnalyzeVisitor v(*complete_block.get_symbol_table());
749 
750  // Check the DUChains for all the variables in the variable_block
751  // If variable is defined in complete_block don't add const quilifier in operator()
752  auto is_functor_const = true;
753  const auto& variables = collect_nodes(variable_block, {ast::AstNodeType::LOCAL_VAR});
754  for (const auto& variable: variables) {
755  const auto& chain = v.analyze(complete_block, variable->get_node_name());
756  is_functor_const = !(chain.eval() == DUState::D || chain.eval() == DUState::LD ||
757  chain.eval() == DUState::CD);
758  if (!is_functor_const) {
759  break;
760  }
761  }
762 
763  return is_functor_const;
764 }
