34 std::string parse_string(const std::string& unit_definition) {
35  nmodl::parser::UnitDriver correctness_driver;
36  correctness_driver.parse_file(nmodl::NrnUnitsLib::get_path());
37  correctness_driver.parse_string(unit_definition);
38  std::stringstream ss;
39  correctness_driver.table->print_units_sorted(ss);
40  correctness_driver.table->print_base_units(ss);
41  return ss.str();
42 }
