56  * \param nmodl_text NMODL text to be parsed and printed

59  IspcCodegenTestHelper(const std::string& nmodl_text) {
60  ast = driver.parse_string(nmodl_text);
61 
62  /// directory where units lib file is located
63  std::string units_dir(NrnUnitsLib::get_path());
64  /// parse units of text
65  UnitsVisitor(units_dir).visit_program(*ast);
66 
67  /// construct symbol table
68  SymtabVisitor().visit_program(*ast);
69 
70  /// initialize CodegenIspcVisitor
71  std::ostream oss(&strbuf);
72  codegen_ispc_visitor =
73  std::make_shared<CodegenIspcVisitor>("unit_test", oss, "double", false);
74  codegen_ispc_visitor->setup(*ast);
75  }

77  /**

117  ASSIGNED {
118  a
119  b
120  }
121  UNITS {
122  FARADAY = (faraday) (coulomb)
123  }
124  INITIAL {
125  a = 0.0
126  b = 0.0
127  }
128  BREAKPOINT {
129  a = b + FARADAY
130  }
131  )";
