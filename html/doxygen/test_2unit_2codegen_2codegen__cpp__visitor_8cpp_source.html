33 /// Helper for creating C codegen visitor
34 std::shared_ptr<CodegenCVisitor> create_c_visitor(const std::shared_ptr<ast::Program>& ast,
35  const std::string& /* text */,
36  std::stringstream& ss) {
37  /// construct symbol table
38  SymtabVisitor().visit_program(*ast);
39 
40  /// run all necessary pass
41  InlineVisitor().visit_program(*ast);
42  NeuronSolveVisitor().visit_program(*ast);
43  SolveBlockVisitor().visit_program(*ast);
44 
45  /// create C code generation visitor
46  auto cv = std::make_shared<CodegenCVisitor>("temp.mod", ss, "double", false);
47  cv->setup(*ast);
48  return cv;
49 }

78 /// print entire code

80  const auto& ast = NmodlDriver().parse_string(nmodl_text);

82  if (generate_gpu_code) {

290 std::string get_instance_structure(std::string nmodl_text) {
291  // parse mod file & print mechanism structure
292  auto const ast = NmodlDriver{}.parse_string(nmodl_text);
293  // add implicit arguments
294  ImplicitArgumentVisitor{}.visit_program(*ast);
295  // update the symbol table for PerfVisitor
296  SymtabVisitor{}.visit_program(*ast);
297  // we need the read/write counts so the codegen knows whether or not
298  // global variables are used
299  PerfVisitor{}.visit_program(*ast);
300  // setup codegen
301  std::stringstream ss{};
302  CodegenCVisitor cv{"temp.mod", ss, "double", false};
303  cv.setup(*ast);
304  cv.print_mechanism_range_var_structure(true);
305  return ss.str();
306 }
