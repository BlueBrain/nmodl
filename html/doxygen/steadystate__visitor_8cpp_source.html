21 std::shared_ptr<ast::DerivativeBlock> SteadystateVisitor::create_steadystate_block(
22  const std::shared_ptr<ast::SolveBlock>& solve_block,
23  const std::vector<std::shared_ptr<ast::Ast>>& deriv_blocks) {
24  // new block to be returned
25  std::shared_ptr<ast::DerivativeBlock> ss_block;
26 
27  // get method & derivative block
28  const auto solve_block_name = solve_block->get_block_name()->get_value()->eval();
29  const auto steadystate_method = solve_block->get_steadystate()->get_value()->eval();
30 
31  logger->debug("SteadystateVisitor :: Found STEADYSTATE SOLVE statement: using {} for {}",
32  steadystate_method,
33  solve_block_name);
34 
35  ast::DerivativeBlock* deriv_block_ptr = nullptr;
36  for (const auto& block_ptr: deriv_blocks) {
37  auto deriv_block = std::dynamic_pointer_cast<ast::DerivativeBlock>(block_ptr);
38  if (deriv_block->get_node_name() == solve_block_name) {
39  logger->debug("SteadystateVisitor :: -> found corresponding DERIVATIVE block: {}",
40  solve_block_name);
41  deriv_block_ptr = deriv_block.get();
42  break;
43  }
44  }
45 
46  if (deriv_block_ptr != nullptr) {
47  // make a clone of derivative block with "_steadystate" suffix
48  ss_block = std::shared_ptr<ast::DerivativeBlock>(deriv_block_ptr->clone());
49  auto ss_name = ss_block->get_name();
50  ss_name->set_name(ss_name->get_value()->get_value() + "_steadystate");
51  auto ss_name_clone = std::shared_ptr<ast::Name>(ss_name->clone());
52  ss_block->set_name(std::move(ss_name));
53  logger->debug("SteadystateVisitor :: -> adding new DERIVATIVE block: {}",
54  ss_block->get_node_name());
55 
56  std::string new_dt;
57  if (steadystate_method == codegen::naming::SPARSE_METHOD) {
58  new_dt = "{:.16g}"_format(STEADYSTATE_SPARSE_DT);
59  } else if (steadystate_method == codegen::naming::DERIVIMPLICIT_METHOD) {
60  new_dt += "{:.16g}"_format(STEADYSTATE_DERIVIMPLICIT_DT);
61  } else {
62  logger->warn("SteadystateVisitor :: solve method {} not supported for STEADYSTATE",
63  steadystate_method);
64  return nullptr;
65  }
66 
67  auto statement_block = ss_block->get_statement_block();
68  auto statements = statement_block->get_statements();
69 
70  // add statement for changing the timestep
71  auto update_dt_statement = std::make_shared<ast::UpdateDt>(new ast::Double(new_dt));
72  statements.insert(statements.begin(), update_dt_statement);
73 
74  // replace old set of statements in AST with new one
75  statement_block->set_statements(std::move(statements));
76 
77  // update SOLVE statement:
78  // set name to point to new DERIVATIVE block
79  solve_block->set_block_name(std::move(ss_name_clone));
80  // change from STEADYSTATE to METHOD
81  solve_block->set_method(solve_block->get_steadystate());
82  solve_block->set_steadystate(nullptr);
83  } else {
84  logger->warn("SteadystateVisitor :: Could not find derivative block {} for STEADYSTATE",
85  solve_block_name);
86  return nullptr;
87  }
88  return ss_block;
89 }

91 void SteadystateVisitor::visit_program(ast::Program& node) {
92  // get DERIVATIVE blocks
93  const auto& deriv_blocks = collect_nodes(node, {ast::AstNodeType::DERIVATIVE_BLOCK});
94 
95  // get list of STEADYSTATE solve statements with names & methods
96  const auto& solve_block_nodes = collect_nodes(node, {ast::AstNodeType::SOLVE_BLOCK});
97 
98  // create new DERIVATIVE blocks for the STEADYSTATE solves
99  for (const auto& solve_block_ptr: solve_block_nodes) {
100  if (auto solve_block = std::dynamic_pointer_cast<ast::SolveBlock>(solve_block_ptr)) {
101  if (solve_block->get_steadystate()) {
102  auto ss_block = create_steadystate_block(solve_block, deriv_blocks);
103  if (ss_block != nullptr) {
104  node.emplace_back_node(ss_block);
105  }
106  }
107  }
108  }
109 }
