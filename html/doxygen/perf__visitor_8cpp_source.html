33 /// count math operations from all binary expressions
34 void PerfVisitor::visit_binary_expression(const ast::BinaryExpression& node) {
35  bool assign_op = false;
36 
37  if (start_measurement) {
38  auto value = node.get_op().get_value();
39  switch (value) {
40  case ast::BOP_ADDITION:
41  current_block_perf.n_add++;
42  break;
43 
44  case ast::BOP_SUBTRACTION:
45  current_block_perf.n_sub++;
46  break;
47 
48  case ast::BOP_MULTIPLICATION:
49  current_block_perf.n_mul++;
50  break;
51 
52  case ast::BOP_DIVISION:
53  current_block_perf.n_div++;
54  break;
55 
56  case ast::BOP_POWER:
57  current_block_perf.n_pow++;
58  break;
59 
60  case ast::BOP_AND:
61  current_block_perf.n_and++;
62  break;
63 
64  case ast::BOP_OR:
65  current_block_perf.n_or++;
66  break;
67 
68  case ast::BOP_GREATER:
69  current_block_perf.n_gt++;
70  break;
71 
72  case ast::BOP_GREATER_EQUAL:
73  current_block_perf.n_ge++;
74  break;
75 
76  case ast::BOP_LESS:
77  current_block_perf.n_lt++;
78  break;
79 
80  case ast::BOP_LESS_EQUAL:
81  current_block_perf.n_le++;
82  break;
83 
84  case ast::BOP_ASSIGN:
85  current_block_perf.n_assign++;
86  assign_op = true;
87  break;
88 
89  case ast::BOP_NOT_EQUAL:
90  current_block_perf.n_ne++;
91  break;
92 
93  case ast::BOP_EXACT_EQUAL:
94  current_block_perf.n_ee++;
95  break;
96 
97  default:
98  throw std::logic_error("Binary operator not handled in perf visitor");
99  }
100  }
101 
102  /// if visiting assignment expression, symbols from lhs
103  /// are written and hence need flag to track
104  if (assign_op) {
105  visiting_lhs_expression = true;
106  }
107 
108  node.get_lhs()->accept(*this);
109 
110  /// lhs is done (rhs is read only)
111  visiting_lhs_expression = false;
112 
113  node.get_rhs()->accept(*this);
114 }

127 /** Helper function used by all ast nodes : visit all children
128  * recursively and performance stats get added on stack. Once
129  * all children visited, we get total performance by summing
130  * perfstat of all children.
131  */
132 void PerfVisitor::measure_performance(const ast::Ast& node) {
133  start_measurement = true;
134 
135  node.visit_children(*this);
136 
137  PerfStat perf;
138  while (!children_blocks_perf.empty()) {
139  perf = perf + children_blocks_perf.top();
140  children_blocks_perf.pop();
141  }
142 
143  auto symtab = node.get_symbol_table();
144  if (symtab == nullptr) {
145  throw std::runtime_error("Perfvisitor : symbol table not setup for " +
146  node.get_node_type_name());
147  }
148 
149  auto name = symtab->name();
150  if (node.is_derivative_block()) {
151  name = node.get_node_type_name();
152  }
153 
154  if (printer) {
155  printer->push_block(name);
156  }
157 
158  perf.title = "Performance Statistics of " + name;
159  perf.print(stream);
160 
161  if (printer) {
162  add_perf_to_printer(perf);
163  printer->pop_block();
164  }
165 
166  start_measurement = false;
167 
168  /// clear var usage map
169  for (auto& var_set: var_usage) {
170  var_set.second.clear();
171  }
172 }

174 /// count function calls and "most useful" or "commonly used" math functions
175 void PerfVisitor::visit_function_call(const ast::FunctionCall& node) {
176  under_function_call = true;
177 
178  if (start_measurement) {
179  auto name = node.get_node_name();
180  if (name == "exp") {
181  current_block_perf.n_exp++;
182  } else if (name == "log") {
183  current_block_perf.n_log++;
184  } else if (name == "pow") {
185  current_block_perf.n_pow++;
186  }
187  node.visit_children(*this);
188 
189  auto symbol = current_symtab->lookup_in_scope(name);
190  auto method_property = NmodlType::procedure_block | NmodlType::function_block;
191  if (symbol != nullptr && symbol->has_any_property(method_property)) {
192  current_block_perf.n_int_func_call++;
193  } else {
194  current_block_perf.n_ext_func_call++;
195  }
196  }
197 
198  under_function_call = false;
199 }

227 void PerfVisitor::count_variables() {
228  /// number of instance variables: range or assigned variables
229  /// one caveat is that the global variables appearing in
230  /// assigned block are not treated as range
231  num_instance_variables = 0;
232 
233  NmodlType property = NmodlType::range_var | NmodlType::assigned_definition |
234  NmodlType::state_var;
235  auto variables = current_symtab->get_variables_with_properties(property);
236 
237  for (auto& variable: variables) {
238  if (!variable->has_any_property(NmodlType::global_var)) {
239  num_instance_variables++;
240  if (variable->has_any_property(NmodlType::param_assign)) {
241  num_constant_instance_variables++;
242  }
243  if (variable->has_any_status(Status::localized)) {
244  num_localized_instance_variables++;
245  }
246  }
247  }
248 
249  /// state variables have state_var property
250  property = NmodlType::state_var;
251  variables = current_symtab->get_variables_with_properties(property);
252  num_state_variables = variables.size();
253 
254  /// pointer variables have pointer/bbcorepointer
255  property = NmodlType::pointer_var | NmodlType::bbcore_pointer_var;
256  variables = current_symtab->get_variables_with_properties(property);
257  num_pointer_variables = variables.size();
258 
259 
260  /// number of global variables : parameters and pointers could appear also
261  /// as range variables and hence need to filter out. But if anything declared
262  /// as global is always global.
263  property = NmodlType::global_var | NmodlType::param_assign | NmodlType::bbcore_pointer_var |
264  NmodlType::pointer_var;
265  variables = current_symtab->get_variables_with_properties(property);
266  num_global_variables = 0;
267  for (auto& variable: variables) {
268  auto is_global = variable->has_any_property(NmodlType::global_var);
269  property = NmodlType::range_var | NmodlType::assigned_definition;
270  if (!variable->has_any_property(property) || is_global) {
271  num_global_variables++;
272  if (variable->has_any_property(NmodlType::param_assign)) {
273  num_constant_global_variables++;
274  }
275  if (variable->has_any_status(Status::localized)) {
276  num_localized_global_variables++;
277  }
278  }
279  }

435 /** Blocks like function can have multiple statement blocks and
436  * blocks like net receive has nested initial blocks. Hence need
437  * to maintain separate stack.
438  */
439 void PerfVisitor::visit_statement_block(const ast::StatementBlock& node) {
440  /// starting new block, store current state
441  blocks_perf.push(current_block_perf);
442 
443  current_symtab = node.get_symbol_table();
444 
445  if (current_symtab == nullptr) {
446  throw std::runtime_error("Perfvisitor : symbol table not setup for " +
447  node.get_node_type_name());
448  }
449 
450  /// new block perf starts from zero
451  current_block_perf = PerfStat();
452 
453  node.visit_children(*this);
454 
455  /// add performance of all visited children
456  total_perf = total_perf + current_block_perf;
457 
458  children_blocks_perf.push(current_block_perf);
459 
460  // go back to parent block's state
461  current_block_perf = blocks_perf.top();
462  blocks_perf.pop();
463 }

494 /** Certain statements / symbols needs extra check while measuring
495  * read/write operations. For example, for expression "exp(a+b)",
496  * "exp" is an external math function and we should not increment read
497  * count for "exp" symbol. Same for solve statement where name will
498  * be derivative block name and neuron solver method.
499  */
500 bool PerfVisitor::symbol_to_skip(const std::shared_ptr<Symbol>& symbol) {
501  bool skip = false;
502 
503  auto is_method = symbol->has_any_property(NmodlType::extern_method | NmodlType::function_block);
504  if (is_method && under_function_call) {
505  skip = true;
506  }
507 
508  is_method = symbol->has_any_property(NmodlType::derivative_block | NmodlType::extern_method);
509  if (is_method && under_solve_block) {
510  skip = true;
511  }
512 
513  return skip;
514 }

536 /** Find symbol in closest scope (up to parent) and update
537  * read/write count. Also update ops count in current block.
538  */
539 void PerfVisitor::update_memory_ops(const std::string& name) {
540  if (!start_measurement || current_symtab == nullptr) {
541  return;
542  }
543 
544  auto symbol = current_symtab->lookup_in_scope(name);
545  if (symbol == nullptr || symbol_to_skip(symbol)) {
546  return;
547  }
548 
549  if (is_local_variable(symbol)) {
550  if (visiting_lhs_expression) {
551  symbol->write();
552  current_block_perf.n_local_write++;
553  } else {
554  symbol->read();
555  current_block_perf.n_local_read++;
556  }
557  return;
558  }
559 
560  /// lhs symbols get written
561  if (visiting_lhs_expression) {
562  symbol->write();
563  if (is_constant_variable(symbol)) {
564  current_block_perf.n_constant_write++;
565  if (var_usage[const_memw_key].find(name) == var_usage[const_memw_key].end()) {
566  current_block_perf.n_unique_constant_write++;
567  var_usage[const_memw_key].insert(name);
568  }
569  } else {
570  current_block_perf.n_global_write++;
571  if (var_usage[global_memw_key].find(name) == var_usage[global_memw_key].end()) {
572  current_block_perf.n_unique_global_write++;
573  var_usage[global_memw_key].insert(name);
574  }
575  }
576  return;
577  }
578 
579  /// rhs symbols get read
580  symbol->read();
581  if (is_constant_variable(symbol)) {
582  current_block_perf.n_constant_read++;
583  if (var_usage[const_memr_key].find(name) == var_usage[const_memr_key].end()) {
584  current_block_perf.n_unique_constant_read++;
585  var_usage[const_memr_key].insert(name);
586  }
587  } else {
588  current_block_perf.n_global_read++;
589  if (var_usage[global_memr_key].find(name) == var_usage[global_memr_key].end()) {
590  current_block_perf.n_unique_global_read++;
591  var_usage[global_memr_key].insert(name);
592  }
593  }
594 }
