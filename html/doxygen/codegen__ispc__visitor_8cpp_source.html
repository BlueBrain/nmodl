46 /*
47  * Rename math functions for ISPC backend
48  */
49 void CodegenIspcVisitor::visit_function_call(const ast::FunctionCall& node) {
50  if (!codegen) {
51  return;
52  }
53  if (node.get_node_name() == "printf") {
54  logger->warn("Not emitted in ispc: {}"_format(to_nmodl(node)));
55  return;
56  }
57  auto& fname = *node.get_name();
58  RenameVisitor("fabs", "abs").visit_name(fname);
59  RenameVisitor("exp", "vexp").visit_name(fname);
60  CodegenCVisitor::visit_function_call(node);
61 }

78 void CodegenIspcVisitor::visit_local_list_statement(const ast::LocalListStatement& node) {
79  if (!codegen) {
80  return;
81  }
82  /// Correct indentation
83  printer->add_newline();
84  printer->add_indent();
85 
86  /// Name of the variable _dt given by
87  /// nmodl::visitor::SteadystateVisitor::create_steadystate_block()
88  const std::string steadystate_dt_variable_name = "dt_saved_value";
89  auto type = CodegenCVisitor::local_var_type() + " ";
90  printer->add_text(type);
91  auto local_variables = node.get_variables();
92  bool dt_saved_value_exists = false;
93 
94  /// Remove dt_saved_value from local_variables if it exists
95  local_variables.erase(std::remove_if(local_variables.begin(),
96  local_variables.end(),
97  [&](const std::shared_ptr<ast::LocalVar>& local_variable) {
98  if (local_variable->get_node_name() ==
99  steadystate_dt_variable_name) {
100  dt_saved_value_exists = true;
101  return true;
102  } else {
103  return false;
104  }
105  }),
106  local_variables.end());
107 
108  /// Print the local_variables like normally
109  CodegenCVisitor::print_vector_elements(local_variables, ", ");
110 
111  /// Print dt_saved_value as uniform
112  if (dt_saved_value_exists) {
113  printer->add_text(";");
114  /// Correct indentation
115  printer->add_newline();
116  printer->add_indent();
117 
118  type = CodegenCVisitor::local_var_type() + " uniform ";
119  printer->add_text(type + steadystate_dt_variable_name);
120  }
121 }

127 /**
128  * In ISPC we have to explicitly append `d` to a floating point number
129  * otherwise it is treated as float. A value stored in the AST can be in
130  * scientific notation and hence we can't just append `d` to the string.
131  * Hence, we have to transform the value into the ISPC compliant format by
132  * replacing `e` and `E` with `d` to keep the same representation of the
133  * number as in the cpp backend.
134  */
135 std::string CodegenIspcVisitor::format_double_string(const std::string& s_value) {
136  return utils::format_double_string<CodegenIspcVisitor>(s_value);
137 }
138 
139 
140 /**
141  * For float variables we don't have to do the conversion with changing `e` and
142  * `E` with `f`, since the scientific notation numbers are already parsed as
143  * floats by ISPC. Instead we need to take care of only appending `f` to the
144  * end of floating point numbers, which is optional on ISPC.
145  */
146 std::string CodegenIspcVisitor::format_float_string(const std::string& s_value) {
147  return utils::format_float_string<CodegenIspcVisitor>(s_value);
148 }

152  if (type == BlockType::Initial) {
153  return method_name(naming::NRN_INIT_METHOD);
154  }

156  return method_name(naming::NRN_CONSTRUCTOR_METHOD);

551 bool CodegenIspcVisitor::check_incompatibilities() {
552  const auto& has_incompatible_nodes = [](const ast::Ast& node) {
553  return !collect_nodes(node, incompatible_node_types).empty();
554  };
555 
556  const auto get_name_from_symbol_type_vector = [](const SymbolType& var) -> const std::string& {
557  return var->get_name();
558  };
559 
560  // instance vars
561  if (check_incompatible_var_name<SymbolType>(codegen_float_variables,
562  get_name_from_symbol_type_vector)) {
563  return true;
564  }
565 
566  if (check_incompatible_var_name<SymbolType>(codegen_shadow_variables,
567  get_name_from_symbol_type_vector)) {
568  return true;
569  }
570  if (check_incompatible_var_name<IndexVariableInfo>(
571  codegen_int_variables, [](const IndexVariableInfo& var) -> const std::string& {
572  return var.symbol->get_name();
573  })) {
574  return true;
575  }
576 
577 
578  if (check_incompatible_var_name<std::string>(info.currents,
579  [](const std::string& var) -> const std::string& {
580  return var;
581  })) {
582  return true;
583  }
584 
585 
586  // global vars
587  // info.top_local_variables is not checked because it should be addressed by the
588  // renameIspcVisitor
589  if (check_incompatible_var_name<SymbolType>(info.global_variables,
590  get_name_from_symbol_type_vector)) {
591  return true;
592  }
593 
594 
595  if (check_incompatible_var_name<SymbolType>(info.constant_variables,
596  get_name_from_symbol_type_vector)) {
597  return true;
598  }
599 
600  // ion vars
601  for (const auto& ion: info.ions) {
602  if (check_incompatible_var_name<std::string>(
603  ion.writes, [](const std::string& var) -> const std::string& { return var; })) {
604  return true;
605  }
606  }
607 
608 
609  emit_fallback = std::vector<bool>(BlockType::BlockTypeEnd, false);
610 
611  if (info.initial_node) {
612  emit_fallback[BlockType::Initial] =
613  emit_fallback[BlockType::Initial] || has_incompatible_nodes(*info.initial_node) ||
614  visitor::calls_function(*info.initial_node, "net_send") || info.require_wrote_conc ||
615  info.net_send_used || info.net_event_used;
616  } else {
617  emit_fallback[BlockType::Initial] = emit_fallback[BlockType::Initial] ||
618  info.net_receive_initial_node ||
619  info.require_wrote_conc;
620  }
621 
622  emit_fallback[BlockType::NetReceive] =
623  emit_fallback[BlockType::NetReceive] ||
624  (info.net_receive_node && (has_incompatible_nodes(*info.net_receive_node) ||
625  visitor::calls_function(*info.net_receive_node, "net_send")));
626 
627  emit_fallback[BlockType::Equation] = emit_fallback[BlockType::Equation] ||
628  (nrn_cur_required() && info.breakpoint_node &&
629  has_incompatible_nodes(*info.breakpoint_node));
630 
631  emit_fallback[BlockType::State] = emit_fallback[BlockType::State] ||
632  (nrn_state_required() && info.nrn_state_block &&

640 void CodegenIspcVisitor::move_procs_to_wrapper() {
641  auto nameset = std::set<std::string>();
642 
643  auto populate_nameset = [&nameset](const ast::Block* block) {
644  if (block) {
645  const auto& names = collect_nodes(*block, {ast::AstNodeType::NAME});
646  for (const auto& name: names) {
647  nameset.insert(name->get_node_name());
648  }
649  }
650  };
651  populate_nameset(info.initial_node);
652  populate_nameset(info.nrn_state_block);
653  populate_nameset(info.breakpoint_node);
654 
655  const auto& has_incompatible_nodes = [](const ast::Ast& node) {
656  return !collect_nodes(node, incompatible_node_types).empty();
657  };
658 
659  auto target_procedures = std::vector<const ast::ProcedureBlock*>();
660  for (const auto& procedure: info.procedures) {
661  const auto& name = procedure->get_name()->get_node_name();
662  if (nameset.find(name) == nameset.end() || has_incompatible_nodes(*procedure)) {
663  wrapper_procedures.push_back(procedure);
664  } else {
665  target_procedures.push_back(procedure);
666  }
667  }
668  info.procedures = target_procedures;
669  auto target_functions = std::vector<const ast::FunctionBlock*>();
670  for (const auto& function: info.functions) {
671  const auto& name = function->get_name()->get_node_name();
672  if (nameset.find(name) == nameset.end() || has_incompatible_nodes(*function)) {
673  wrapper_functions.push_back(function);
674  } else {
675  target_functions.push_back(function);
676  }
677  }
678  info.functions = target_functions;
679 }
