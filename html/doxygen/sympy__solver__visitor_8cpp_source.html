29 void SympySolverVisitor::init_block_data(ast::Node* node) {
30  // clear any previous data
31  expression_statements.clear();
32  eq_system.clear();
33  state_vars_in_block.clear();
34  last_expression_statement = nullptr;
35  block_with_expression_statements = nullptr;
36  eq_system_is_valid = true;
37  conserve_equation.clear();
38  // get set of local block vars & global vars
39  vars = global_vars;
40  if (auto symtab = node->get_statement_block()->get_symbol_table()) {
41  auto localvars = symtab->get_variables_with_properties(NmodlType::local_var);
42  for (const auto& localvar: localvars) {
43  std::string var_name = localvar->get_name();
44  if (localvar->is_array()) {
45  var_name += "[" + std::to_string(localvar->get_length()) + "]";
46  }
47  vars.insert(var_name);
48  }
49  }
50  const auto& fcall_nodes = collect_nodes(*node->get_statement_block(),
51  {ast::AstNodeType::FUNCTION_CALL});
52  for (const auto& call: fcall_nodes) {
53  function_calls.insert(call->get_node_name());
54  }
55 }

88 ast::StatementVector::const_iterator SympySolverVisitor::get_solution_location_iterator(
89  const ast::StatementVector& statements) {
90  // find out where to insert solutions in statement block
91  // returns iterator pointing to the first element after the last (non)linear eq
92  // so if there are no such elements, it returns statements.end()
93  auto it = statements.begin();
94  if (last_expression_statement != nullptr) {
95  while ((it != statements.end()) &&
96  (std::dynamic_pointer_cast<ast::ExpressionStatement>(*it).get() !=
97  last_expression_statement)) {
98  logger->debug("SympySolverVisitor :: {} != {}",
99  to_nmodl(*it),
100  to_nmodl(*last_expression_statement));
101  ++it;
102  }
103  if (it != statements.end()) {
104  logger->debug("SympySolverVisitor :: {} == {}",
105  to_nmodl(std::dynamic_pointer_cast<ast::ExpressionStatement>(*it)),
106  to_nmodl(*last_expression_statement));
107  ++it;
108  }
109  }
110  return it;
111 }

113 /**
114  * Check if provided statement is local variable declaration statement
115  * @param statement AST node representing statement in the MOD file
116  * @return True if statement is local variable declaration else False
117  *
118  * Statement declaration could be wrapped into another statement type like
119  * expression statement and hence we try to look inside if it's really a
120  * variable declaration.
121  */
122 static bool is_local_statement(const std::shared_ptr<ast::Statement>& statement) {
123  if (statement->is_local_list_statement()) {
124  return true;
125  }
126  if (statement->is_expression_statement()) {
127  auto e_statement = std::dynamic_pointer_cast<ast::ExpressionStatement>(statement);
128  auto expression = e_statement->get_expression();
129  if (expression->is_local_list_statement()) {
130  return true;
131  }
132  }
133  return false;
134 }

136 std::string& SympySolverVisitor::replaceAll(std::string& context,
137  const std::string& from,
138  const std::string& to) const {
139  std::size_t lookHere = 0;
140  std::size_t foundHere;
141  while ((foundHere = context.find(from, lookHere)) != std::string::npos) {
142  context.replace(foundHere, from.size(), to);
143  lookHere = foundHere + to.size();
144  }
145  return context;
146 }

160 void SympySolverVisitor::construct_eigen_solver_block(
161  const std::vector<std::string>& pre_solve_statements,
162  const std::vector<std::string>& solutions,
163  bool linear) {
164  // Provide random string to append to X, J, Jm and F matrices that
165  // are produced by sympy
166  std::string unique_X = suffix_random_string(vars, "X");
167  std::string unique_J = suffix_random_string(vars, "J");
168  std::string unique_Jm = suffix_random_string(vars, "Jm");
169  std::string unique_F = suffix_random_string(vars, "F");
170 
171  // filter solutions for matrices named "X", "J", "Jm" and "F" and change them to
172  // unique_X, unique_J, unique_Jm and unique_F respectively
173  auto solutions_filtered = filter_string_vector(solutions, "X[", unique_X + "[");
174  solutions_filtered = filter_string_vector(solutions_filtered, "J[", unique_J + "[");
175  solutions_filtered = filter_string_vector(solutions_filtered, "Jm[", unique_Jm + "[");
176  solutions_filtered = filter_string_vector(solutions_filtered, "F[", unique_F + "[");
177 
178  for (const auto& sol: solutions_filtered) {
179  logger->debug("SympySolverVisitor :: -> adding statement: {}", sol);
180  }
181 
182  std::vector<std::string> pre_solve_statements_and_setup_x_eqs(pre_solve_statements);
183  std::vector<std::string> update_statements;
184  for (int i = 0; i < state_vars.size(); i++) {
185  auto update_state = state_vars[i] + " = " + unique_X + "[" + std::to_string(i) + "]";
186  auto setup_x = unique_X + "[" + std::to_string(i) + "] = " + state_vars[i];
187 
188  pre_solve_statements_and_setup_x_eqs.push_back(setup_x);
189  update_statements.push_back(update_state);
190  logger->debug("SympySolverVisitor :: setup_", unique_X, ": {}", setup_x);
191  logger->debug("SympySolverVisitor :: update_state: {}", update_state);
192  }
193 
194  visitor::SympyReplaceSolutionsVisitor solution_replacer(
195  pre_solve_statements_and_setup_x_eqs,
196  solutions_filtered,
197  expression_statements,
198  visitor::SympyReplaceSolutionsVisitor::ReplacePolicy::GREEDY,
199  state_vars.size() + 1,
200  "");
201  solution_replacer.visit_statement_block(*block_with_expression_statements);
202 
203  // split in the various blocks for eigen
204  auto n_state_vars = std::make_shared<ast::Integer>(state_vars.size(), nullptr);
205 
206  const auto& statements = block_with_expression_statements->get_statements();
207 
208  ast::StatementVector variable_statements; // LOCAL //
209  ast::StatementVector initialize_statements; // pre_solve_statements //
210  ast::StatementVector setup_x_statements; // old_x = x, X[0] = x //
211  ast::StatementVector functor_statements; // J[0]_row * X = F[0], additional assignments during
212  // computation //
213  ast::StatementVector finalize_statements; // assignments at the end //
214  const size_t sr_begin = solution_replacer.replaced_statements_begin();
215  const size_t sr_end = solution_replacer.replaced_statements_end();
216 
217  // initialize and edge case where the system of equations is empty
218  for (size_t idx = 0; idx < statements.size(); ++idx) {
219  auto& s = statements[idx];
220  if (is_local_statement(s)) {
221  variable_statements.push_back(s);
222  } else if (sr_begin == statements.size()) {
223  initialize_statements.push_back(s);
224  } else if (idx < sr_begin) {
225  initialize_statements.push_back(s);
226  }
227  }
228 
229  if (sr_begin != statements.size()) {
230  initialize_statements.insert(initialize_statements.end(),
231  statements.begin() + sr_begin,
232  statements.begin() + sr_begin + pre_solve_statements.size());
233  setup_x_statements =
234  ast::StatementVector(statements.begin() + sr_begin + pre_solve_statements.size(),
235  statements.begin() + sr_begin + pre_solve_statements.size() +
236  state_vars.size());
237  functor_statements = ast::StatementVector(statements.begin() + sr_begin +
238  pre_solve_statements.size() +
239  state_vars.size(),
240  statements.begin() + sr_end);
241  finalize_statements = ast::StatementVector(statements.begin() + sr_end, statements.end());
242  }
243 
244  const size_t total_statements_size = variable_statements.size() + initialize_statements.size() +
245  setup_x_statements.size() + functor_statements.size() +
246  finalize_statements.size();
247  if (statements.size() != total_statements_size) {
248  logger->error(
249  "SympySolverVisitor :: statement number missmatch ({} =/= {}) during splitting before "
250  "creation of "
251  "eigen "
252  "solver block.",
253  statements.size(),
254  total_statements_size);
255  return;
256  }
257 
258  auto variable_block = std::make_shared<ast::StatementBlock>(std::move(variable_statements));
259  auto initialize_block = std::make_shared<ast::StatementBlock>(std::move(initialize_statements));
260  auto update_state_block = create_statement_block(update_statements);
261  auto finalize_block = std::make_shared<ast::StatementBlock>(std::move(finalize_statements));
262  if (linear) {
263  /// functor and initialize block converge in the same block
264  setup_x_statements.insert(setup_x_statements.end(),
265  functor_statements.begin(),
266  functor_statements.end());
267  auto setup_x_block = std::make_shared<ast::StatementBlock>(std::move(setup_x_statements));
268  auto solver_block = std::make_shared<ast::EigenLinearSolverBlock>(n_state_vars,
269  variable_block,
270  initialize_block,
271  setup_x_block,
272  update_state_block,
273  finalize_block);
274  /// replace statement block with solver block as it contains all statements
275  ast::StatementVector solver_block_statements{
276  std::make_shared<ast::ExpressionStatement>(solver_block)};
277  block_with_expression_statements->set_statements(std::move(solver_block_statements));
278  } else {
279  /// create eigen newton solver block
280  auto setup_x_block = std::make_shared<ast::StatementBlock>(std::move(setup_x_statements));
281  auto functor_block = std::make_shared<ast::StatementBlock>(std::move(functor_statements));
282  auto solver_block = std::make_shared<ast::EigenNewtonSolverBlock>(n_state_vars,
283  variable_block,
284  initialize_block,
285  setup_x_block,
286  functor_block,
287  update_state_block,
288  finalize_block);
289  /// replace statement block with solver block as it contains all statements
290  ast::StatementVector solver_block_statements{
291  std::make_shared<ast::ExpressionStatement>(solver_block)};
292  block_with_expression_statements->set_statements(std::move(solver_block_statements));
293  }
294 }

297 void SympySolverVisitor::solve_linear_system(const std::vector<std::string>& pre_solve_statements) {
298  // construct ordered vector of state vars used in linear system
299  init_state_vars_vector();
300  // call sympy linear solver
301  bool small_system = (eq_system.size() <= SMALL_LINEAR_SYSTEM_MAX_STATES);
302  auto solver = pywrap::EmbeddedPythonLoader::get_instance().api()->create_sls_executor();
303  solver->eq_system = eq_system;
304  solver->state_vars = state_vars;
305  solver->vars = vars;
306  solver->small_system = small_system;
307  solver->elimination = elimination;
308  // this is necessary after we destroy the solver
309  const auto tmp_unique_prefix = suffix_random_string(vars, "tmp");
310  solver->tmp_unique_prefix = tmp_unique_prefix;
311  solver->function_calls = function_calls;
312  (*solver)();
313  // returns a vector of solutions, i.e. new statements to add to block:
314  auto solutions = solver->solutions;
315  // and a vector of new local variables that need to be declared in the block:
316  auto new_local_vars = solver->new_local_vars;
317  // may also return a python exception message:
318  auto exception_message = solver->exception_message;
319  // destroy solver
320  pywrap::EmbeddedPythonLoader::get_instance().api()->destroy_sls_executor(solver);
321  if (!exception_message.empty()) {
322  logger->warn("SympySolverVisitor :: solve_lin_system python exception: " +
323  exception_message);
324  return;
325  }
326  // find out where to insert solutions in statement block
327  if (small_system) {
328  // for small number of state vars, linear solver
329  // directly returns solution by solving symbolically at compile time
330  logger->debug("SympySolverVisitor :: Solving *small* linear system of eqs");
331  // declare new local vars
332  if (!new_local_vars.empty()) {
333  for (const auto& new_local_var: new_local_vars) {
334  logger->debug("SympySolverVisitor :: -> declaring new local variable: {}",
335  new_local_var);
336  add_local_variable(*block_with_expression_statements, new_local_var);
337  }
338  }
339  visitor::SympyReplaceSolutionsVisitor solution_replacer(
340  pre_solve_statements,
341  solutions,
342  expression_statements,
343  visitor::SympyReplaceSolutionsVisitor::ReplacePolicy::VALUE,
344  1,
345  tmp_unique_prefix);
346  solution_replacer.visit_statement_block(*block_with_expression_statements);
347  } else {
348  // otherwise it returns a linear matrix system to solve
349  logger->debug("SympySolverVisitor :: Constructing linear newton solve block");
350  construct_eigen_solver_block(pre_solve_statements, solutions, true);
351  }
352 }

400 void SympySolverVisitor::visit_diff_eq_expression(ast::DiffEqExpression& node) {
401  const auto& lhs = node.get_expression()->get_lhs();
402 
403  if (!lhs->is_var_name()) {
404  logger->warn("SympySolverVisitor :: LHS of differential equation is not a VariableName");
405  return;
406  }
407  auto lhs_name = std::dynamic_pointer_cast<ast::VarName>(lhs)->get_name();
408  if ((lhs_name->is_indexed_name() &&
409  !std::dynamic_pointer_cast<ast::IndexedName>(lhs_name)->get_name()->is_prime_name()) ||
410  (!lhs_name->is_indexed_name() && !lhs_name->is_prime_name())) {
411  logger->warn("SympySolverVisitor :: LHS of differential equation is not a PrimeName");
412  return;
413  }
414 
415  check_expr_statements_in_same_block();
416 
417  const auto node_as_nmodl = to_nmodl_for_sympy(node);
418  const auto deleter = [](nmodl::pybind_wrappers::DiffeqSolverExecutor* ptr) {
419  pywrap::EmbeddedPythonLoader::get_instance().api()->destroy_des_executor(ptr);
420  };
421  std::unique_ptr<nmodl::pybind_wrappers::DiffeqSolverExecutor, decltype(deleter)> diffeq_solver{
422  pywrap::EmbeddedPythonLoader::get_instance().api()->create_des_executor(), deleter};
423 
424  diffeq_solver->node_as_nmodl = node_as_nmodl;
425  diffeq_solver->dt_var = codegen::naming::NTHREAD_DT_VARIABLE;
426  diffeq_solver->vars = vars;
427  diffeq_solver->use_pade_approx = use_pade_approx;
428  diffeq_solver->function_calls = function_calls;
429  diffeq_solver->method = solve_method;
430  (*diffeq_solver)();
431  if (solve_method == codegen::naming::EULER_METHOD) {
432  // replace x' = f(x) differential equation
433  // with forwards Euler timestep:
434  // x = x + f(x) * dt
435  logger->debug("SympySolverVisitor :: EULER - solving: {}", node_as_nmodl);
436  } else if (solve_method == codegen::naming::CNEXP_METHOD) {
437  // replace x' = f(x) differential equation
438  // with analytic solution for x(t+dt) in terms of x(t)
439  // x = ...
440  logger->debug("SympySolverVisitor :: CNEXP - solving: {}", node_as_nmodl);
441  } else {
442  // for other solver methods: just collect the ODEs & return
443  std::string eq_str = to_nmodl_for_sympy(node);
444  std::string var_name = lhs_name->get_node_name();
445  if (lhs_name->is_indexed_name()) {
446  auto index_name = std::dynamic_pointer_cast<ast::IndexedName>(lhs_name);
447  var_name +=
448  "[" +
449  std::to_string(
450  std::dynamic_pointer_cast<ast::Integer>(index_name->get_length())->eval()) +
451  "]";
452  }
453  logger->debug("SympySolverVisitor :: adding ODE system: {}", eq_str);
454  eq_system.push_back(eq_str);
455  logger->debug("SympySolverVisitor :: adding state var: {}", var_name);
456  state_vars_in_block.insert(var_name);
457  expression_statements.insert(current_expression_statement);
458  last_expression_statement = current_expression_statement;
459  return;
460  }
461 
462  // replace ODE with solution in AST
463  auto solution = diffeq_solver->solution;
464  logger->debug("SympySolverVisitor :: -> solution: {}", solution);
465 
466  auto exception_message = diffeq_solver->exception_message;
467  if (!exception_message.empty()) {
468  logger->warn("SympySolverVisitor :: python exception: " + exception_message);
469  return;
470  }
471 
472  if (!solution.empty()) {
473  replace_diffeq_expression(node, solution);
474  } else {
475  logger->warn("SympySolverVisitor :: solution to differential equation not possible");
476  }
477 }

479 void SympySolverVisitor::visit_conserve(ast::Conserve& node) {
480  // Replace ODE for state variable on LHS of CONSERVE statement with
481  // algebraic expression on RHS (see p244 of NEURON book)
482  logger->debug("SympySolverVisitor :: CONSERVE statement: {}", to_nmodl(node));
483  expression_statements.insert(&node);
484  std::string conserve_equation_statevar;
485  if (node.get_react()->is_react_var_name()) {
486  conserve_equation_statevar = node.get_react()->get_node_name();
487  }
488  if (std::find(all_state_vars.cbegin(), all_state_vars.cend(), conserve_equation_statevar) ==
489  all_state_vars.cend()) {
490  logger->error(
491  "SympySolverVisitor :: Invalid CONSERVE statement for DERIVATIVE block, LHS should be "
492  "a state variable, instead found: {}. Ignoring CONSERVE statement",
493  to_nmodl(node.get_react()));
494  return;
495  }
496  auto conserve_equation_str = to_nmodl_for_sympy(*node.get_expr());
497  logger->debug("SympySolverVisitor :: --> replace ODE for state var {} with equation {}",
498  conserve_equation_statevar,
499  conserve_equation_str);
500  conserve_equation[conserve_equation_statevar] = conserve_equation_str;
501 }

503 void SympySolverVisitor::visit_derivative_block(ast::DerivativeBlock& node) {
504  /// clear information from previous block, get global vars + block local vars
505  init_block_data(&node);
506 
507  // get user specified solve method for this block
508  solve_method = derivative_block_solve_method[node.get_node_name()];
509 
510  // visit each differential equation:
511  // - for CNEXP or EULER, each equation is independent & is replaced with its solution
512  // - otherwise, each equation is added to eq_system
513  node.visit_children(*this);
514 
515  if (eq_system_is_valid && !eq_system.empty()) {
516  // solve system of ODEs in eq_system
517  logger->debug("SympySolverVisitor :: Solving {} system of ODEs", solve_method);
518 
519  // construct implicit Euler equations from ODEs
520  std::vector<std::string> pre_solve_statements;
521  for (auto& eq: eq_system) {
522  auto split_eq = stringutils::split_string(eq, '=');
523  auto x_prime_split = stringutils::split_string(split_eq[0], '\'');
524  auto x = stringutils::trim(x_prime_split[0]);
525  std::string x_array_index = "";
526  std::string x_array_index_i = "";
527  if (x_prime_split.size() > 1 && stringutils::trim(x_prime_split[1]).size() > 2) {
528  x_array_index = stringutils::trim(x_prime_split[1]);
529  x_array_index_i = "_" + x_array_index.substr(1, x_array_index.size() - 2);
530  }
531  std::string state_var_name = x + x_array_index;
532  auto var_eq_pair = conserve_equation.find(state_var_name);
533  if (var_eq_pair != conserve_equation.cend()) {
534  // replace the ODE for this state var with corresponding CONSERVE equation
535  eq = state_var_name + " = " + var_eq_pair->second;
536  logger->debug(
537  "SympySolverVisitor :: -> instead of Euler eq using CONSERVE equation: {} = {}",
538  state_var_name,
539  var_eq_pair->second);
540  } else {
541  // no CONSERVE equation, construct Euler equation
542  auto dxdt = stringutils::trim(split_eq[1]);
543 
544

638 void SympySolverVisitor::visit_program(ast::Program& node) {
639  derivative_block_solve_method.clear();
640 
641  global_vars = get_global_vars(node);
642 
643  // get list of solve statements with names & methods
644  const auto& solve_block_nodes = collect_nodes(node, {ast::AstNodeType::SOLVE_BLOCK});
645  for (const auto& block: solve_block_nodes) {
646  if (auto block_ptr = std::dynamic_pointer_cast<const ast::SolveBlock>(block)) {
647  const auto& block_name = block_ptr->get_block_name()->get_value()->eval();
648  if (block_ptr->get_method()) {
649  // Note: solve method name is an optional parameter
650  // LINEAR and NONLINEAR blocks do not have solve method specified
651  const auto& solve_method = block_ptr->get_method()->get_value()->eval();
652  logger->debug("SympySolverVisitor :: Found SOLVE statement: using {} for {}",
653  solve_method,
654  block_name);
655  derivative_block_solve_method[block_name] = solve_method;
656  }
657  }
658  }
659 
660  // get set of all state vars
661  all_state_vars.clear();
662  if (auto symtab = node.get_symbol_table()) {
663  auto statevars = symtab->get_variables_with_properties(NmodlType::state_var);
664  for (const auto& v: statevars) {
665  std::string var_name = v->get_name();
666  if (v->is_array()) {
667  for (int i = 0; i < v->get_length(); ++i) {
668  std::string var_name_i = var_name + "[" + std::to_string(i) + "]";
669  all_state_vars.push_back(var_name_i);
670  }
671  } else {
672  all_state_vars.push_back(var_name);
673  }
674  }
675  }
676 
677  node.visit_children(*this);
678 }
