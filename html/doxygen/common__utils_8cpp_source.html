55 bool make_path(const std::string& path) {
56  mode_t mode = 0755;
57  int ret = mkdir(path.c_str(), mode);
58  if (ret == 0) {
59  return true;
60  }
61 
62  switch (errno) {
63  case ENOENT:
64  // parent didn't exist, try to create it
65  {
66  int pos = path.find_last_of('/');
67  if (pos == std::string::npos) {
68  return false;
69  }
70  if (!make_path(path.substr(0, pos))) {
71  return false;
72  }
73  }
74  // now, try to create again
75  return 0 == mkdir(path.c_str(), mode);
76 
77  case EEXIST:
78  // done!
79  return is_dir_exist(path);
80 
81  default:
82  auto msg = "Can not create directory " + path;
83  throw std::runtime_error(msg);
84  }
85 }

104 std::string generate_random_string(const int len, UseNumbersInString use_numbers) {
105  std::string s(len, 0);
106  static const char alphanum[] =
107  "0123456789"
108  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
109  "abcdefghijklmnopqrstuvwxyz";
110  std::random_device dev;
111  std::mt19937 rng(dev());
112  std::uniform_int_distribution<std::mt19937::result_type> dist(use_numbers ? 0 : 10,
113  sizeof(alphanum) - 2);
114  for (int i = 0; i < len; ++i) {
115  s[i] = alphanum[dist(rng)];
116  }
117  return s;
118 }
