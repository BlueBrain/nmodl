35 void CheckParentVisitor::check_parent(const ast::Ast &node) const {
36  if (!parent) {
37  if (is_root_with_null_parent && node.get_parent()) {
38  const auto &parent_type = parent->get_node_type_name();
39  throw std::runtime_error(
40  "root->parent: {} is set when it should be nullptr"_format(
41  parent_type));
42  }
43  } else {
44  if (parent != node.get_parent()) {
45  const std::string parent_type =
46  (parent == nullptr) ? "nullptr" : parent->get_node_type_name();
47  const std::string node_parent_type =
48  (node.get_parent() == nullptr)
49  ? "nullptr"
50  : node.get_parent()->get_node_type_name();
51  throw std::runtime_error(
52  "parent: {} and child->parent: {} missmatch"_format(
53  parent_type, node_parent_type));
54  }
55  }
56 }
