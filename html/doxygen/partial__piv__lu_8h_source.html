22 // We want to declare a function template that is callable from OpenMP and
23 // OpenACC code, but whose instantiations are compiled by nvc++ -cuda. This is to avoid
24 // Eigen internals having to be digested by OpenACC/OpenMP compilers. The
25 // problem is that it is apparently not sufficient to declare a template in a OpenMP
26 // declare target region and have the OpenMP compiler assume that device
27 // versions of instantations of it will be available. The convoluted approach
28 // here has two ingredients:
29 // - partialPivLu<N>(...), a function template that has explicit
30 // specialisations visible in this header file that call:
31 // - partialPivLuN(...), functions that are declared in this header but defined
32 // in the C++/CUDA file partial_piv_lu.cpp.
33 //
34 // It does not seem to work to generically define the template function
35 // partialPivLu<N> in the .cpp file and then explicitly instantate it for 1..16
36 // there; the relevant device functions are not generated. A possible workaround
37 // is to define the template + a dummy function that calls many different
38 // instantiations of the template in the .cpp file, but this seems more fragile
39 // than the present approach.
40 #ifdef NMODL_EIGEN_NO_OPENACC
41 #define NMODL_EIGEN_ATTR __host__ __device__
42 #define NMODL_EIGEN_ROUTINE_SEQ
43 #else
44 nrn_pragma_omp(declare target)

46 #define NMODL_EIGEN_ROUTINE_SEQ nrn_pragma_acc(routine seq)
47 #endif
