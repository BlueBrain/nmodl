30 std::string suffix_random_string(const std::set<std::string>& vars,
31  const std::string& original_string,
32  const UseNumbersInString use_num) {
33  std::string new_string = original_string;
34  std::string random_string;
35  auto& singleton_random_string_class = nmodl::utils::SingletonRandomString<4>::instance();
36  // Check if there is a variable defined in the mod file as original_string or that
37  // has original_string as prefix and, if yes,
38  // try to use a different string in the form "original_string"_"random_string"
39 
40  const auto it = vars.lower_bound(new_string);
41  while ((it != vars.end()) && new_string == it->substr(0, new_string.size())) {
42  random_string = singleton_random_string_class.reset_random_string(original_string, use_num);
43  new_string = original_string;
44  new_string += "_" + random_string;
45  }
46  return new_string;
47 }

104 /**
105  * Convert given code statement (in string format) to corresponding ast node
106  *
107  * We create dummy nmodl procedure containing given code statement and then
108  * parse it using NMODL parser. As there will be only one block with single
109  * statement, we return first statement.
110  *
111  * \todo Need to revisit this during code generation passes to make sure
112  * if all statements can be part of procedure block.
113  */
114 std::shared_ptr<Statement> create_statement(const std::string& code_statement) {
115  nmodl::parser::NmodlDriver driver;
116  auto nmodl_text = "PROCEDURE dummy() { " + code_statement + " }";
117  auto ast = driver.parse_string(nmodl_text);
118  auto procedure = std::dynamic_pointer_cast<ProcedureBlock>(ast->get_blocks().front());
119  auto statement = std::shared_ptr<Statement>(
120  procedure->get_statement_block()->get_statements()[0]->clone());
121  return statement;
122 }

124 std::vector<std::shared_ptr<Statement>> create_statements(
125  const std::vector<std::string>::const_iterator& code_statements_beg,
126  const std::vector<std::string>::const_iterator& code_statements_end) {
127  std::vector<std::shared_ptr<Statement>> statements;
128  statements.reserve(code_statements_end - code_statements_beg);
129  std::transform(code_statements_beg,
130  code_statements_end,

132  [](const std::string& s) { return create_statement(s); });
133  return statements;

136 /**
137  * Convert given code statement (in string format) to corresponding ast node
138  *
139  * We create dummy nmodl procedure containing given code statement and then
140  * parse it using NMODL parser. As there will be only one block with single
141  * statement, we return first statement.
142  */
143 std::shared_ptr<StatementBlock> create_statement_block(
144  const std::vector<std::string>& code_statements) {
145  nmodl::parser::NmodlDriver driver;
146  std::string nmodl_text = "PROCEDURE dummy() {\n";
147  for (auto& statement: code_statements) {
148  nmodl_text += statement + "\n";
149  }
150  nmodl_text += "}";
151  auto ast = driver.parse_string(nmodl_text);
152  auto procedure = std::dynamic_pointer_cast<ProcedureBlock>(ast->get_blocks().front());
153  auto statement_block = std::shared_ptr<StatementBlock>(
154  procedure->get_statement_block()->clone());
155  return statement_block;
156 }

158 std::set<std::string> get_global_vars(const Program& node) {
159  std::set<std::string> vars;
160  if (auto* symtab = node.get_symbol_table()) {
161  // NB: local_var included here as locals can be declared at global scope
162  NmodlType property = NmodlType::global_var | NmodlType::local_var | NmodlType::range_var |
163  NmodlType::param_assign | NmodlType::extern_var |
164  NmodlType::prime_name | NmodlType::assigned_definition |
165  NmodlType::read_ion_var | NmodlType::write_ion_var |
166  NmodlType::nonspecific_cur_var | NmodlType::electrode_cur_var |
167  NmodlType::section_var | NmodlType::constant_var |
168  NmodlType::extern_neuron_variable | NmodlType::state_var |
169  NmodlType::factor_def;
170  for (const auto& globalvar: symtab->get_variables_with_properties(property)) {
171  std::string var_name = globalvar->get_name();
172  if (globalvar->is_array()) {
173  var_name += "[" + std::to_string(globalvar->get_length()) + "]";
174  }
175  vars.insert(var_name);
176  }
177  }
178  return vars;
179 }

237 std::pair<std::string, std::unordered_set<std::string>> statement_dependencies(
238  const std::shared_ptr<ast::Expression>& lhs,
239  const std::shared_ptr<ast::Expression>& rhs) {
240  std::string key;
241  std::unordered_set<std::string> out;
242 
243  if (!lhs->is_var_name()) {
244  return {key, out};
245  }
246 
247  key = to_nmodl(lhs);
248  key.erase(std::remove(key.begin(), key.end(), '\''),
249  key.end()); // we want to match derivatives and variables
250  visitor::AstLookupVisitor lookup_visitor;
251  lookup_visitor.lookup(*rhs, ast::AstNodeType::VAR_NAME);
252  auto rhs_nodes = lookup_visitor.get_nodes();
253  std::for_each(rhs_nodes.begin(),
254  rhs_nodes.end(),
255  [&out](const std::shared_ptr<ast::Ast>& node) { out.emplace(to_nmodl(node)); });
256 
257 
258  return {key, out};
259 }
