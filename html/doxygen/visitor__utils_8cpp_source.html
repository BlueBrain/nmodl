31 std::string suffix_random_string(const std::set<std::string>& vars,
32  const std::string& original_string,
33  const UseNumbersInString use_num) {
34  // If the "original_string" is not in the set of the variables to check then
35  // return the "original_string" without suffix
36  if (vars.find(original_string) == vars.end()) {
37  return original_string;
38  }
39  std::string new_string = original_string;
40  auto& singleton_random_string_class = nmodl::utils::SingletonRandomString<4>::instance();
41  // Check if there is a variable defined in the mod file and, if yes, try to use
42  // a different string in the form "original_string"_"random_string"
43  // If there is already a "random_string" assigned to the "originl_string" return it
44  if (singleton_random_string_class.random_string_exists(original_string)) {
45  const auto random_suffix = "_" +
46  singleton_random_string_class.get_random_string(original_string);
47  new_string = original_string + random_suffix;
48  } else {
49  // Check if the "random_string" already exists in the set of variables and if it does try
50  // to find another random string to add as suffix
51  while (vars.find(new_string) != vars.end()) {
52  const auto random_suffix =
53  "_" + singleton_random_string_class.reset_random_string(original_string, use_num);
54  new_string = original_string + random_suffix;
55  }
56  }
57  return new_string;
58 }

115 /**
116  * Convert given code statement (in string format) to corresponding ast node
117  *
118  * We create dummy nmodl procedure containing given code statement and then
119  * parse it using NMODL parser. As there will be only one block with single
120  * statement, we return first statement.
121  *
122  * \todo Need to revisit this during code generation passes to make sure
123  * if all statements can be part of procedure block.
124  */
125 std::shared_ptr<Statement> create_statement(const std::string& code_statement) {
126  nmodl::parser::NmodlDriver driver;
127  auto nmodl_text = "PROCEDURE dummy() { " + code_statement + " }";
128  auto ast = driver.parse_string(nmodl_text);
129  auto procedure = std::dynamic_pointer_cast<ProcedureBlock>(ast->get_blocks().front());
130  auto statement = std::shared_ptr<Statement>(
131  procedure->get_statement_block()->get_statements()[0]->clone());
132  return statement;
133 }

135 std::vector<std::shared_ptr<Statement>> create_statements(
136  const std::vector<std::string>::const_iterator& code_statements_beg,
137  const std::vector<std::string>::const_iterator& code_statements_end) {
138  std::vector<std::shared_ptr<Statement>> statements;
139  statements.reserve(code_statements_end - code_statements_beg);
140  std::transform(code_statements_beg,
141  code_statements_end,
142  std::back_inserter(statements),
143  [](const std::string& s) { return create_statement(s); });
144  return statements;
145 }

147 /**
148  * Convert given code statement (in string format) to corresponding ast node
149  *
150  * We create dummy nmodl procedure containing given code statement and then
151  * parse it using NMODL parser. As there will be only one block with single
152  * statement, we return first statement.
153  */
154 std::shared_ptr<StatementBlock> create_statement_block(
155  const std::vector<std::string>& code_statements) {
156  nmodl::parser::NmodlDriver driver;
157  std::string nmodl_text = "PROCEDURE dummy() {\n";
158  for (auto& statement: code_statements) {
159  nmodl_text += statement + "\n";
160  }
161  nmodl_text += "}";
162  auto ast = driver.parse_string(nmodl_text);
163  auto procedure = std::dynamic_pointer_cast<ProcedureBlock>(ast->get_blocks().front());
164  auto statement_block = std::shared_ptr<StatementBlock>(
165  procedure->get_statement_block()->clone());
166  return statement_block;
167 }

169 std::set<std::string> get_global_vars(const Program& node) {
170  std::set<std::string> vars;
171  if (auto* symtab = node.get_symbol_table()) {
172  // NB: local_var included here as locals can be declared at global scope
173  NmodlType property = NmodlType::global_var | NmodlType::local_var | NmodlType::range_var |
174  NmodlType::param_assign | NmodlType::extern_var |
175  NmodlType::prime_name | NmodlType::assigned_definition |
176  NmodlType::read_ion_var | NmodlType::write_ion_var |
177  NmodlType::nonspecific_cur_var | NmodlType::electrode_cur_var |
178  NmodlType::constant_var | NmodlType::extern_neuron_variable |
179  NmodlType::state_var | NmodlType::factor_def;
180  for (const auto& globalvar: symtab->get_variables_with_properties(property)) {
181  std::string var_name = globalvar->get_name();
182  if (globalvar->is_array()) {
183  var_name += "[" + std::to_string(globalvar->get_length()) + "]";
184  }
185  vars.insert(var_name);
186  }
187  }
188  return vars;
189 }

245 std::pair<std::string, std::unordered_set<std::string>> statement_dependencies(
246  const std::shared_ptr<ast::Expression>& lhs,
247  const std::shared_ptr<ast::Expression>& rhs) {
248  std::string key;
249  std::unordered_set<std::string> out;
250 
251  if (!lhs->is_var_name()) {
252  return {key, out};
253  }
254 
255  const auto& lhs_var_name = std::dynamic_pointer_cast<ast::VarName>(lhs);
256  key = get_full_var_name(*lhs_var_name);
257 
258  visitor::AstLookupVisitor lookup_visitor;
259  lookup_visitor.lookup(*rhs, ast::AstNodeType::VAR_NAME);
260  auto rhs_nodes = lookup_visitor.get_nodes();
261  std::for_each(rhs_nodes.begin(),
262  rhs_nodes.end(),
263  [&out](const std::shared_ptr<ast::Ast>& node) { out.emplace(to_nmodl(node)); });
264 
265 
266  return {key, out};
267 }
