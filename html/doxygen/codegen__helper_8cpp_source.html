28 std::string run_codegen_helper_visitor(const std::string& text) {
29  NmodlDriver driver;
30  const auto& ast = driver.parse_string(text);
31 
32  /// construct symbol table and run codegen helper visitor
33  SymtabVisitor().visit_program(*ast);
34  CodegenHelperVisitor v;
35 
36  /// symbols/variables are collected in info object
37  const auto& info = v.analyze(*ast);
38 
39  /// semicolon separated list of variables
40  std::string variables;
41 
42  /// range variables in order of code generation
43  for (const auto& var: info.range_parameter_vars) {
44  variables += var->get_name() + ";";
45  }
46  for (const auto& var: info.range_assigned_vars) {
47  variables += var->get_name() + ";";
48  }
49  for (const auto& var: info.range_state_vars) {
50  variables += var->get_name() + ";";
51  }
52  for (const auto& var: info.assigned_vars) {
53  variables += var->get_name() + ";";
54  }
55 
56  return variables;
57 }

65  cai = 50.e-6 (mM)
66  cao = 2 (mM)
67  q10 = 5

71  SUFFIX cal

75  GLOBAL USEGHK
76  }

78  m
