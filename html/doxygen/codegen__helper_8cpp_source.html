24 std::string run_inline_visitor(const std::string& text) {
25  NmodlDriver driver;
26  const auto& ast = driver.parse_string(text);
27 
28  /// construct symbol table and run codegen helper visitor
29  SymtabVisitor().visit_program(*ast);
30  CodegenHelperVisitor v;
31 
32  /// symbols/variables are collected in info object
33  const auto& info = v.analyze(*ast);
34 
35  /// semicolon separated list of variables
36  std::string variables;
37 
38  /// range variables in order of code generation
39  for (const auto& var: info.range_parameter_vars) {
40  variables += var->get_name() + ";";
41  }
42  for (const auto& var: info.range_assigned_vars) {
43  variables += var->get_name() + ";";
44  }
45  for (const auto& var: info.range_state_vars) {
46  variables += var->get_name() + ";";
47  }
48  for (const auto& var: info.assigned_vars) {
49  variables += var->get_name() + ";";
50  }
51 
52  return variables;
53 }
