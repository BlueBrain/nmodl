84 static inline double egm1(double x, double n) {
85  // this cannot be reordered for the double-double trick to work
86  // i.e., it cannot be re-written as g = x - n * (C1+C2)
87  // the loss of accuracy comes from the different magnitudes of ln(2) and n
88  // max(|n|) ~ 2^9
89  // ln(2) ~ 2^-1
90  volatile double g = x - n * C1;
91  g -= n * C2;
92 
93  const double gg = g * g;
94 
95  double px = PX1exp;
96  px *= gg;
97  px += PX2exp;
98  px *= gg;
99  px += PX3exp;
100  px *= g;
101 
102  double qx = QX1exp;
103  qx *= gg;
104  qx += QX2exp;
105  qx *= gg;
106  qx += QX3exp;
107  qx *= gg;
108  qx += QX4exp;
109 
110  return 2.0 * px / (qx - px);
111 }

113 /// double precision exp function
114 static inline double vexp(double initial_x) {
115  double x = initial_x;
116  double px = std::floor(LOG2E * x + 0.5);
117  const int32_t n = px;
118 
119  x = 1.0 + egm1(x, px);
120  x *= uint642dp((((uint64_t) n) + 1023) << 52);
121 
122  if (initial_x > EXP_LIMIT)
123  x = inf();
124  if (initial_x < -EXP_LIMIT)
125  x = 0.0;
126 
127  return x;
128 }

130 /// double precision exp(x) - 1 function, used for small x.
131 static inline double vexpm1(double initial_x) {
132  double x = initial_x;
133  double px = std::floor(LOG2E * x + 0.5);
134  const int32_t n = px;
135 
136  const uint64_t twopnm1 = (static_cast<uint64_t>(n - 1) + 1023) << 52;
137  x = 2 * (uint642dp(twopnm1) * egm1(x, px) + uint642dp(twopnm1)) - 1;

141  if (initial_x < -EXP_LIMIT)
142  x = -1.0;

157 static inline float egm1(float x) {
158  float z = x * PX1expf;
159  z += PX2expf;
160  z *= x;
161  z += PX3expf;

164  z *= x;

169  z += x;

174 /// single precision exp function
175 static inline float vexp(float x) {
176  float z = std::floor(LOG2EF * x + 0.5f);
177 
178  // this cannot be reordered for the double-double trick to work

195 /// single precision exp function
196 static inline float vexpm1(float x) {
197  float z = std::floor(LOG2EF * x + 0.5f);
198 
199  // this cannot be reordered for the double-double trick to work
200  volatile float g = x - z * C1F;
201  g -= z * C2F;
202 
203  const int32_t n = z;
204 
205  const int32_t twopnm1 = ((n - 1) + 0x7f) << 23;
206  float ret = 2 * (int322sp(twopnm1) * egm1(g) + int322sp(twopnm1)) - 1;
207 
208  if (x > MAXLOGF)
209  ret = f_inf();
210  if (x < MINLOGF)
211  ret = -1.0f;
