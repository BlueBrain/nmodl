42 SCENARIO("LOCAL to ASSIGNED variable transformer", "[visitor][localtoassigned]") {
43  GIVEN("mod file with LOCAL variables that are written") {
44  std::string input_nmodl = R"(
45  NEURON {
46  SUFFIX test
47  }
48 
49  LOCAL x, y, z
50 
51  INITIAL {
52  x = 1
53  }
54 
55  BREAKPOINT {
56  z = 2
57  }
58  )";
59 
60  auto ast = run_local_to_assigned_visitor(input_nmodl);
61  auto symtab = ast->get_symbol_table();
62 
63  THEN("LOCAL variables that are written are turned to ASSIGNED") {
64  /// check for all ASSIGNED variables : old ones + newly converted ones
65  auto vars = symtab->get_variables_with_properties(NmodlType::assigned_definition);
66  REQUIRE(vars.size() == 2);
67 
68  /// x and z should be converted from LOCAL to ASSIGNED
69  auto x = symtab->lookup("x");
70  REQUIRE(x != nullptr);
71  REQUIRE(x->has_any_property(NmodlType::assigned_definition) == true);
72  REQUIRE(x->has_any_property(NmodlType::local_var) == false);
73 
74  auto z = symtab->lookup("z");
75  REQUIRE(z != nullptr);
76  REQUIRE(z->has_any_property(NmodlType::assigned_definition) == true);
77  REQUIRE(z->has_any_property(NmodlType::local_var) == false);
78  }
79 
80  THEN("LOCAL variables that are read only remain LOCAL") {
81  auto vars = symtab->get_variables_with_properties(NmodlType::local_var);
82  REQUIRE(vars.size() == 1);
83  REQUIRE(vars[0]->get_name() == "y");
84  }
85  }
86 }
