38 /**
39  * \brief Newton method with user-provided Jacobian
40  *
41  * Newton method with user-provided Jacobian: given initial vector X and a
42  * functor that calculates `F(X)`, `J(X)` where `J(X)` is the Jacobian of `F(X)`,
43  * solves for \f$F(X) = 0\f$, starting with initial value of `X` by iterating:
44  *
45  * \f[
46  * X_{n+1} = X_n - J(X_n)^{-1} F(X_n)
47  * \f]
48  * when \f$|F|^2 < eps^2\f$, solution has converged.
49  *
50  * @return number of iterations (-1 if failed to converge)
51  */
52 template <int N, typename FUNC>
53 EIGEN_DEVICE_FUNC int newton_solver(Eigen::Matrix<double, N, 1>& X,
54  FUNC functor,
55  double eps = EPS,
56  int max_iter = MAX_ITER) {
57  // Vector to store result of function F(X):
58  Eigen::Matrix<double, N, 1> F;
59  // Matrix to store jacobian of F(X):
60  Eigen::Matrix<double, N, N> J;
61  // Solver iteration count:
62  int iter = -1;
63  while (++iter < max_iter) {
64  // calculate F, J from X using user-supplied functor
65  functor(X, F, J);
66  // get error norm: here we use sqrt(|F|^2)
67  double error = F.norm();
68  if (error < eps) {
69  // we have converged: return iteration count
70  return iter;
71  }
72  // update X use in-place LU decomposition of J with partial pivoting
73  // (suitable for any N, but less efficient than .inverse() for N <=4)
74  X -= Eigen::PartialPivLU<Eigen::Ref<Eigen::Matrix<double, N, N>>>(J).solve(F);
75  }
76  // If we fail to converge after max_iter iterations, return -1
77  return -1;
78 }

83 /**
84  * \brief Newton method without user-provided Jacobian
85  *
86  * Newton method without user-provided Jacobian: given initial vector X and a
87  * functor that calculates `F(X)`, solves for \f$F(X) = 0\f$, starting with
88  * initial value of `X` by iterating:
89  *
90  * \f[
91  * X_{n+1} = X_n - J(X_n)^{-1} F(X_n)
92  * \f]
93  *
94  * where `J(X)` is the Jacobian of `F(X)`, which is approximated numerically
95  * using a symmetric finite difference approximation to the derivative
96  * when \f$|F|^2 < eps^2\f$, solution has converged/
97  *
98  * @return number of iterations (-1 if failed to converge)
99  */
100 template <int N, typename FUNC>
101 EIGEN_DEVICE_FUNC int newton_numerical_diff_solver(Eigen::Matrix<double, N, 1>& X,
102  FUNC functor,
103  double eps = EPS,
104  int max_iter = MAX_ITER) {
105  // Vector to store result of function F(X):
106  Eigen::Matrix<double, N, 1> F;
107  // Temporary storage for F(X+dx)
108  Eigen::Matrix<double, N, 1> F_p;
109  // Temporary storage for F(X-dx)
110  Eigen::Matrix<double, N, 1> F_m;
111  // Matrix to store jacobian of F(X):
112  Eigen::Matrix<double, N, N> J;
113  // Solver iteration count:
114  int iter = 0;
115  while (iter < max_iter) {
116  // calculate F from X using user-supplied functor
117  functor(X, F);
118  // get error norm: here we use sqrt(|F|^2)
119  double error = F.norm();
120  if (error < eps) {
121  // we have converged: return iteration count
122  return iter;
123  }
124  ++iter;
125  // calculate approximate Jacobian
126  for (int i = 0; i < N; ++i) {
127  // symmetric finite difference approximation to derivative
128  // df/dx ~= ( f(x+dx) - f(x-dx) ) / (2*dx)
129  // choose dx to be ~(ULP)^{1/3}*X[i]
130  // https://aip.scitation.org/doi/pdf/10.1063/1.4822971
131  // also enforce a lower bound ~sqrt(ULP) to avoid dx being too small
132  double dX = std::max(CUBIC_ROOT_ULP * X[i], SQUARE_ROOT_ULP);
133  // F(X + dX)
134  X[i] += dX;
135  functor(X, F_p);
136  // F(X - dX)
137  X[i] -= 2.0 * dX;
138  functor(X, F_m);
139  F_p -= F_m;
140  // J = (F(X + dX) - F(X - dX)) / (2*dX)
141  J.col(i) = F_p / (2.0 * dX);
142  // restore X
143  X[i] += dX;
144  }
145  // update X
146  // use in-place LU decomposition of J with partial pivoting
147  // (suitable for any N, but less efficient than .inverse() for N <=4)
148  X -= Eigen::PartialPivLU<Eigen::Ref<Eigen::Matrix<double, N, N>>>(J).solve(F);
149  }
150  // If we fail to converge after max_iter iterations, return -1
151  return -1;
152 }

154 /**
155  * Newton method template specializations for \f$N <= 4\f$ Use explicit inverse
156  * of `F` instead of LU decomposition. This is faster, as there is no pivoting
157  * and therefore no branches, but it is not numerically safe for \f$N > 4\f$.
158  */
159 
160 template <typename FUNC, int N>
161 EIGEN_DEVICE_FUNC int newton_solver_small_N(Eigen::Matrix<double, N, 1>& X,
162  FUNC functor,
163  double eps,
164  int max_iter) {
165  Eigen::Matrix<double, N, 1> F;
166  Eigen::Matrix<double, N, N> J;
167  int iter = -1;
168  while (++iter < max_iter) {
169  functor(X, F, J);
170  double error = F.norm();
171  if (error < eps) {
172  return iter;
173  }
174  X -= J.inverse() * F;
175  }
176  return -1;
177 }
