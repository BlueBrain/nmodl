41 /**
42  * \brief Newton method with user-provided Jacobian
43  *
44  * Newton method with user-provided Jacobian: given initial vector X and a
45  * functor that calculates `F(X)`, `J(X)` where `J(X)` is the Jacobian of `F(X)`,
46  * solves for \f$F(X) = 0\f$, starting with initial value of `X` by iterating:
47  *
48  * \f[
49  * X_{n+1} = X_n - J(X_n)^{-1} F(X_n)
50  * \f]
51  * when \f$|F|^2 < eps^2\f$, solution has converged.
52  *
53  * @return number of iterations (-1 if failed to converge)
54  */
55 template <int N, typename FUNC>
56 EIGEN_DEVICE_FUNC int newton_solver(Eigen::Matrix<double, N, 1>& X,
57  FUNC functor,
58  double eps = EPS,
59  int max_iter = MAX_ITER) {
60  // Vector to store result of function F(X):
61  Eigen::Matrix<double, N, 1> F;
62  // Matrix to store jacobian of F(X):
63  Eigen::Matrix<double, N, N> J;
64  // Solver iteration count:
65  int iter = -1;
66  while (++iter < max_iter) {
67  // calculate F, J from X using user-supplied functor
68  functor(X, F, J);
69  // get error norm: here we use sqrt(|F|^2)
70  double error = F.norm();
71  if (error < eps) {
72  // we have converged: return iteration count
73  return iter;
74  }
75  // In Eigen the default storage order is ColMajor.
76  // Crout's implementation requires matrices stored in RowMajor order (C-style arrays).
77  // Therefore, the transposeInPlace is critical such that the data() method to give the rows
78  // instead of the columns.
79  if (!J.IsRowMajor)
80  J.transposeInPlace();
81  Eigen::Matrix<int, N, 1> pivot;
82  Eigen::Matrix<double, N, 1> rowmax;
83  // Check if J is singular
84  if (nmodl::crout::Crout<double>(N, J.data(), pivot.data(), rowmax.data()) < 0)
85  return -1;
86  Eigen::Matrix<double, N, 1> X_solve;
87  nmodl::crout::solveCrout<double>(N, J.data(), F.data(), X_solve.data(), pivot.data());
88  X -= X_solve;
89  }
90  // If we fail to converge after max_iter iterations, return -1
91  return -1;
92 }

97 /**
98  * \brief Newton method without user-provided Jacobian
99  *
100  * Newton method without user-provided Jacobian: given initial vector X and a
101  * functor that calculates `F(X)`, solves for \f$F(X) = 0\f$, starting with
102  * initial value of `X` by iterating:
103  *
104  * \f[
105  * X_{n+1} = X_n - J(X_n)^{-1} F(X_n)
106  * \f]
107  *
108  * where `J(X)` is the Jacobian of `F(X)`, which is approximated numerically
109  * using a symmetric finite difference approximation to the derivative
110  * when \f$|F|^2 < eps^2\f$, solution has converged/
111  *
112  * @return number of iterations (-1 if failed to converge)
113  */
114 template <int N, typename FUNC>
115 EIGEN_DEVICE_FUNC int newton_numerical_diff_solver(Eigen::Matrix<double, N, 1>& X,
116  FUNC functor,
117  double eps = EPS,
118  int max_iter = MAX_ITER) {
119  // Vector to store result of function F(X):
120  Eigen::Matrix<double, N, 1> F;
121  // Temporary storage for F(X+dx)
122  Eigen::Matrix<double, N, 1> F_p;
123  // Temporary storage for F(X-dx)
124  Eigen::Matrix<double, N, 1> F_m;
125  // Matrix to store jacobian of F(X):
126  Eigen::Matrix<double, N, N> J;
127  // Solver iteration count:
128  int iter = 0;
129  while (iter < max_iter) {
130  // calculate F from X using user-supplied functor
131  functor(X, F);
132  // get error norm: here we use sqrt(|F|^2)
133  double error = F.norm();
134  if (error < eps) {
135  // we have converged: return iteration count
136  return iter;
137  }
138  ++iter;
139  // calculate approximate Jacobian
140  for (int i = 0; i < N; ++i) {
141  // symmetric finite difference approximation to derivative
142  // df/dx ~= ( f(x+dx) - f(x-dx) ) / (2*dx)
143  // choose dx to be ~(ULP)^{1/3}*X[i]
144  // https://aip.scitation.org/doi/pdf/10.1063/1.4822971
145  // also enforce a lower bound ~sqrt(ULP) to avoid dx being too small
146  double dX = std::max(CUBIC_ROOT_ULP * X[i], SQUARE_ROOT_ULP);
147  // F(X + dX)
148  X[i] += dX;
149  functor(X, F_p);
150  // F(X - dX)
151  X[i] -= 2.0 * dX;
152  functor(X, F_m);
153  F_p -= F_m;
154  // J = (F(X + dX) - F(X - dX)) / (2*dX)
155  J.col(i) = F_p / (2.0 * dX);
156  // restore X
157  X[i] += dX;
158  }
159  if (!J.IsRowMajor)
160  J.transposeInPlace();
161  Eigen::Matrix<int, N, 1> pivot;
162  Eigen::Matrix<double, N, 1> rowmax;
163  // Check if J is singular
164  if (nmodl::crout::Crout<double>(N, J.data(), pivot.data(), rowmax.data()) < 0)
165  return -1;
166  Eigen::Matrix<double, N, 1> X_solve;
167  nmodl::crout::solveCrout<double>(N, J.data(), F.data(), X_solve.data(), pivot.data());
168  X -= X_solve;
169  }
170  // If we fail to converge after max_iter iterations, return -1
171  return -1;
172 }

174 /**
175  * Newton method template specializations for \f$N <= 4\f$ Use explicit inverse
176  * of `F` instead of LU decomposition. This is faster, as there is no pivoting
177  * and therefore no branches, but it is not numerically safe for \f$N > 4\f$.
178  */
179 
180 template <typename FUNC, int N>
181 EIGEN_DEVICE_FUNC int newton_solver_small_N(Eigen::Matrix<double, N, 1>& X,
182  FUNC functor,
183  double eps,
184  int max_iter) {
185  bool invertible;
186  Eigen::Matrix<double, N, 1> F;
187  Eigen::Matrix<double, N, N> J, J_inv;
188  int iter = -1;
189  while (++iter < max_iter) {
190  functor(X, F, J);
191  double error = F.norm();
192  if (error < eps) {
193  return iter;
194  }
195  // The inverse can be called from within OpenACC regions without any issue, as opposed to
196  // Eigen::PartialPivLU.
197  J.computeInverseWithCheck(J_inv, invertible);
198  if (invertible)
199  X -= J_inv * F;
200  else
201  return -1;
202  }
203  return -1;
204 }
