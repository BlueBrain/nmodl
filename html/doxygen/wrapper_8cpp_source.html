22 void SolveLinearSystemExecutor::operator()() {
23  const auto locals = py::dict("eq_strings"_a = eq_system,
24  "state_vars"_a = state_vars,
25  "vars"_a = vars,
26  "small_system"_a = small_system,
27  "do_cse"_a = elimination,
28  "function_calls"_a = function_calls,
29  "tmp_unique_prefix"_a = tmp_unique_prefix);
30  py::exec(R"(
31  from nmodl.ode import solve_lin_system
32  exception_message = ""
33  try:
34  solutions, new_local_vars = solve_lin_system(eq_strings,
35  state_vars,
36  vars,
37  function_calls,
38  tmp_unique_prefix,
39  small_system,
40  do_cse)
41  except Exception as e:
42  # if we fail, fail silently and return empty string
43  solutions = [""]
44  new_local_vars = [""]
45  exception_message = str(e)
46  )",
47  py::globals(),
48  locals);
49  // returns a vector of solutions, i.e. new statements to add to block:
50  solutions = locals["solutions"].cast<std::vector<std::string>>();
51  // and a vector of new local variables that need to be declared in the block:
52  new_local_vars = locals["new_local_vars"].cast<std::vector<std::string>>();
53  // may also return a python exception message:
54  exception_message = locals["exception_message"].cast<std::string>();
55 }

85 void DiffeqSolverExecutor::operator()() {
86  const auto locals = py::dict("equation_string"_a = node_as_nmodl,
87  "dt_var"_a = dt_var,
88  "vars"_a = vars,
89  "use_pade_approx"_a = use_pade_approx,
90  "function_calls"_a = function_calls);
91 
92  if (method == codegen::naming::EULER_METHOD) {
93  // replace x' = f(x) differential equation
94  // with forwards Euler timestep:
95  // x = x + f(x) * dt
96  py::exec(R"(
97  from nmodl.ode import forwards_euler2c
98  exception_message = ""
99  try:
100  solution = forwards_euler2c(equation_string, dt_var, vars, function_calls)
101  except Exception as e:
102  # if we fail, fail silently and return empty string
103  solution = ""
104  exception_message = str(e)
105  )",
106  py::globals(),
107  locals);
108  } else if (method == codegen::naming::CNEXP_METHOD) {
109  // replace x' = f(x) differential equation
110  // with analytic solution for x(t+dt) in terms of x(t)
111  // x = ...
112  py::exec(R"(
113  from nmodl.ode import integrate2c
114  exception_message = ""
115  try:
116  solution = integrate2c(equation_string, dt_var, vars,
117  use_pade_approx)
118  except Exception as e:
119  # if we fail, fail silently and return empty string
120  solution = ""
121  exception_message = str(e)
122  )",
123  py::globals(),
124  locals);
125  } else {
126  // nothing to do, but the caller should know.
127  return;
128  }
129  solution = locals["solution"].cast<std::string>();
130  exception_message = locals["exception_message"].cast<std::string>();
131 }
