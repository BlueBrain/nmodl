32 std::string CodegenCompatibilityVisitor::return_error_if_solve_method_is_unhandled(
33  ast::Ast& /* node */,
34  const std::shared_ptr<ast::Ast>& ast_node) const {
35  auto solve_block_ast_node = std::dynamic_pointer_cast<ast::SolveBlock>(ast_node);
36  std::stringstream unhandled_method_error_message;
37  auto method = solve_block_ast_node->get_method();
38  if (!method) {
39  return {};
40  }
41  auto unhandled_solver_method = handled_solvers.find(method->get_node_name()) ==
42  handled_solvers.end();
43  if (unhandled_solver_method) {
44  unhandled_method_error_message << fmt::format(
45  "\"{}\" solving method used at [{}] not handled. Supported methods are cnexp, euler, "
46  "derivimplicit and sparse\n",
47  method->get_node_name(),
48  method->get_token()->position());
49  }
50  return unhandled_method_error_message.str();
51 }

57  auto global_var = std::dynamic_pointer_cast<ast::GlobalVar>(ast_node);
58  std::stringstream error_message_global_var;
59  if (node.get_symbol_table()->lookup(global_var->get_node_name())->get_write_count() > 0) {
60  error_message_global_var << fmt::format(
61  "\"{}\" variable found at [{}] should be defined as a RANGE variable instead of GLOBAL "
62  "to enable backend transformations\n",
63  global_var->get_node_name(),
64  global_var->get_token()->position());
65  }
66  return error_message_global_var.str();

71  ast::Ast& node,
72  const std::shared_ptr<ast::Ast>& ast_node) const {
73  auto param_assign = std::dynamic_pointer_cast<ast::ParamAssign>(ast_node);
74  std::stringstream error_message_global_var;
75  auto symbol = node.get_symbol_table()->lookup(param_assign->get_node_name());
76  if (!symbol->is_writable() && symbol->get_write_count() > 0) {
77  error_message_global_var << fmt::format(
78  "\"{}\" variable found at [{}] should be writable if it needs to be written\n",

80  symbol->get_token().position());

82  return error_message_global_var.str();

85 // NOLINTNEXTLINE(readability-convert-member-functions-to-static)
86 std::string CodegenCompatibilityVisitor::return_error_if_no_bbcore_read_write(
87  ast::Ast& node,
88  const std::shared_ptr<ast::Ast>& /* ast_node */) const {
89  std::stringstream error_message_no_bbcore_read_write;
90  const auto& verbatim_nodes = collect_nodes(node, {AstNodeType::VERBATIM});
91  auto found_bbcore_read = false;
92  auto found_bbcore_write = false;
93  for (const auto& it: verbatim_nodes) {
94  auto verbatim = std::dynamic_pointer_cast<ast::Verbatim>(it);
95 
96  auto verbatim_statement_string = verbatim->get_statement()->get_value();
97 
98  // Parse verbatim block to find out if there is a token "bbcore_read" or
99  // "bbcore_write". If there is not, then the function is not defined or
100  // is commented.
101  parser::CDriver driver;
102 
103  driver.scan_string(verbatim_statement_string);
104  auto tokens = driver.all_tokens();
105 
106  for (const auto& token: tokens) {
107  if (token == "bbcore_read") {
108  found_bbcore_read = true;
109  }
110  if (token == "bbcore_write") {
111  found_bbcore_write = true;
112  }
113  }
114  }
115  if (!found_bbcore_read) {
116  error_message_no_bbcore_read_write
117  << "\"bbcore_read\" function not defined in any VERBATIM block\n";
118  }
119  if (!found_bbcore_write) {
120  error_message_no_bbcore_read_write
121  << "\"bbcore_write\" function not defined in any VERBATIM block\n";
122  }
123  return error_message_no_bbcore_read_write.str();
124 }

128  * generation backend for CoreNEURON and prints related messages. If there is
129  * some kind of incompatibility return false.

132 bool CodegenCompatibilityVisitor::find_unhandled_ast_nodes(Ast& node) const {
133  std::vector<ast::AstNodeType> unhandled_ast_types;
134  unhandled_ast_types.reserve(unhandled_ast_types_func.size());
135  for (auto [node_type, _]: unhandled_ast_types_func) {
136  unhandled_ast_types.push_back(node_type);
137  }
138  const auto& unhandled_ast_nodes = collect_nodes(node, unhandled_ast_types);
139 
140  std::ostringstream ss;
141  for (const auto& it: unhandled_ast_nodes) {
142  auto node_type = it->get_node_type();
143  ss << (this->*unhandled_ast_types_func.find(node_type)->second)(node, it);
144  }
145  if (!ss.str().empty()) {
146  logger->error("Code incompatibility detected");
147  logger->error("Cannot translate mod file to .cpp file");
148  logger->error("Fix the following errors and try again");
149  std::string line;
150  std::istringstream ss_stringstream(ss.str());
151  while (std::getline(ss_stringstream, line)) {
152  if (!line.empty()) {
153  logger->error(fmt::format("Code Incompatibility :: {}", line));
154  }
155  }
156  return true;
157  }
158  return false;
159 }
