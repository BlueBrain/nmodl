37 /**
38  * \class CodegenCompatibilityVisitor
39  * \brief %Visitor for printing compatibility issues of the mod file
40  *
41  * INDEPENDENT_BLOCK is ignored (no error raised) as stated in:
42  * https://www.neuron.yale.edu/neuron/static/py_doc/modelspec/programmatic/mechanisms/nmodl.html
43  */
44 class CodegenCompatibilityVisitor: public visitor::AstVisitor {
45  /// Typedef for defining FunctionPointer that points to the

73  /// \}

77  ///
78  /// \param node Ast

80  bool find_unhandled_ast_nodes(Ast& node);

82  private:
83  /// Takes as parameter an std::shared_ptr<ast::Ast>,
84  /// searches if the method used for solving is supported
85  /// and if it is not it returns a relative error message
86  ///
87  /// \param node Not used by the function
88  /// \param ast_node Ast node which is checked
89  /// \return std::string error
90  std::string return_error_if_solve_method_is_unhandled(
91  ast::Ast& node,
92  const std::shared_ptr<ast::Ast>& ast_node);
93 
94  /// Takes as parameter an std::shared_ptr<ast::Ast> node
95  /// and returns a relative error with the name, the type
96  /// and the location of the unhandled statement
97  ///
98  /// \tparam T Type of node parameter in the ast::Ast
99  /// \param node Not used by the function
100  /// \param ast_node Ast node which is checked
101  /// \return std::string error
102  template <typename T>
103  std::string return_error_with_name(ast::Ast& node, const std::shared_ptr<ast::Ast>& ast_node) {
104  auto real_type_block = std::dynamic_pointer_cast<T>(ast_node);
105  return "\"{}\" {}construct found at [{}] is not handled\n"_format(
106  ast_node->get_node_name(),
107  real_type_block->get_nmodl_name(),
108  real_type_block->get_token()->position());
109  }
110 
111  /// Takes as parameter an std::shared_ptr<ast::Ast> node
112  /// and returns a relative error with the type and the
113  /// location of the unhandled statement
114  ///
115  /// \tparam T Type of node parameter in the ast::Ast
116  /// \param node Not used by the function
117  /// \param ast_node Ast node which is checked
118  /// \return std::string error
119  template <typename T>
120  std::string return_error_without_name(ast::Ast& node,
121  const std::shared_ptr<ast::Ast>& ast_node) {
122  auto real_type_block = std::dynamic_pointer_cast<T>(ast_node);
123  return "{}construct found at [{}] is not handled\n"_format(
124  real_type_block->get_nmodl_name(), real_type_block->get_token()->position());
125  }
126 
127  /// Takes as parameter the ast::Ast to read the symbol table
128  /// and an std::shared_ptr<ast::Ast> node and returns relative
129  /// error if a variable that is writen in the mod file is
130  /// defined as GLOBAL instead of RANGE
131  ///
132  /// \param node Ast
133  /// \param ast_node Ast node which is checked
134  /// \return std::string error
135  std::string return_error_global_var(ast::Ast& node, const std::shared_ptr<ast::Ast>& ast_node);
136 
137  std::string return_error_param_var(ast::Ast& node, const std::shared_ptr<ast::Ast>& ast_node);
138 
139  /// Takes as parameter the ast::Ast and checks if the
140  /// functions "bbcore_read" and "bbcore_write" are defined
141  /// in any of the ast::Ast VERBATIM blocks. The function is
142  /// called if there is a BBCORE_POINTER defined in the mod
143  /// file
144  ///
145  /// \param node Ast
146  /// \param ast_node Not used by the function
147  /// \return std::string error
148  std::string return_error_if_no_bbcore_read_write(ast::Ast& node,
149  const std::shared_ptr<ast::Ast>& ast_node);
150 };
