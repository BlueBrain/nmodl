29 void EmbeddedPythonLoader::load_libraries() {
30  const auto pylib_env = std::getenv("NMODL_PYLIB");
31  if (!pylib_env) {
32  logger->critical("NMODL_PYLIB environment variable must be set to load embedded python");
33  throw std::runtime_error("NMODL_PYLIB not set");
34  }
35  const auto dlopen_opts = RTLD_NOW | RTLD_GLOBAL;
36  dlerror(); // reset old error conditions
37  pylib_handle = dlopen(pylib_env, dlopen_opts);
38  if (!pylib_handle) {
39  const auto errstr = dlerror();
40  logger->critical("Tried but failed to load {}", pylib_env);
41  logger->critical(errstr);
42  throw std::runtime_error("Failed to dlopen");
43  }
44  const auto pybind_wraplib_env = std::getenv("NMODL_WRAPLIB");
45  if (!pybind_wraplib_env) {
46  logger->critical(
47  "NMODL_WRAPLIB environment variable must be set to load the pybind wrapper library");
48  throw std::runtime_error("NMODL_WRAPLIB not set");
49  }
50  pybind_wrapper_handle = dlopen(pybind_wraplib_env, dlopen_opts);
51  if (!pybind_wrapper_handle) {
52  const auto errstr = dlerror();
53  logger->critical("Tried but failed to load {}", pybind_wraplib_env);
54  logger->critical(errstr);
55  throw std::runtime_error("Failed to dlopen");
56  }
57 }
