26 SCENARIO("Symbol properties can be added and converted to string") {
27  NmodlType prop1{NmodlType::empty};
28  NmodlType prop2 = NmodlType::local_var;
29  NmodlType prop3 = NmodlType::global_var;
30 
31  GIVEN("A empty property") {
32  WHEN("converted to string") {
33  THEN("returns empty string") {
34  REQUIRE(to_string(prop1).empty());
35  }
36  }
37  WHEN("checked for property") {
38  THEN("doesn't have any property") {
39  REQUIRE_FALSE(has_property(prop1, NmodlType::local_var));
40  }
41  }
42  WHEN("adding another empty property") {
43  NmodlType result = prop1 | prop1; // NOLINT(misc-redundant-expression)
44  THEN("to_string still returns empty string") {
45  REQUIRE(to_string(result).empty());
46  }
47  }
48  WHEN("added some other property") {
49  NmodlType result = prop1 | prop2;
50  THEN("to_string returns added property") {
51  REQUIRE(to_string(result) == "local");
52  }
53  WHEN("checked for property") {
54  THEN("has required property") {
55  REQUIRE(has_property(result, NmodlType::local_var) == true);
56  }
57  }
58  }
59  WHEN("added multiple properties") {
60  NmodlType result = prop1 | prop2 | prop3;
61  result |= NmodlType::write_ion_var;
62  THEN("to_string returns all added properties") {
63  REQUIRE_THAT(to_string(result), Catch::Contains("local"));
64  REQUIRE_THAT(to_string(result), Catch::Contains("global"));
65  REQUIRE_THAT(to_string(result), Catch::Contains("write_ion"));
66  }
67  WHEN("checked for property") {
68  THEN("has all added properties") {
69  REQUIRE(has_property(result, NmodlType::local_var) == true);
70  REQUIRE(has_property(result, NmodlType::global_var) == true);
71  REQUIRE(has_property(result, NmodlType::write_ion_var) == true);
72  REQUIRE_FALSE(has_property(result, NmodlType::read_ion_var));
73  }
74  }
75  }
76  WHEN("properties manipulated with bitwise operators") {
77  THEN("& applied correctly") {
78  NmodlType result1 = prop2 & prop2; // NOLINT(misc-redundant-expression)
79  NmodlType result2 = prop1 & prop2;
80  NmodlType result3 = prop1 & prop2 & prop3;
81  REQUIRE(to_string(result1) == "local");
82  REQUIRE(to_string_vector(result2).empty());
83  REQUIRE(result2 == NmodlType::empty);
84  REQUIRE(result3 == NmodlType::empty);
85  }
86  }
87  }
88 }

94 SCENARIO("Multiple properties can be added to Symbol") {
95  NmodlType property1 = NmodlType::argument;
96  NmodlType property2 = NmodlType::range_var;
97  NmodlType property3 = NmodlType::discrete_block;
98  GIVEN("A symbol") {
99  ModToken token(true);
100  Symbol symbol("alpha", token);
101  WHEN("added external property") {
102  symbol.add_property(NmodlType::extern_neuron_variable);
103  THEN("symbol becomes external") {
104  REQUIRE(symbol.is_external_variable() == true);
105  }
106  }
107  WHEN("added multiple properties to symbol") {
108  symbol.add_property(property1);
109  symbol.add_property(property2);
110  THEN("symbol has multiple properties") {
111  REQUIRE(symbol.has_any_property(property1) == true);
112 
113  REQUIRE(symbol.has_any_property(property3) == false);
114 
115  symbol.add_property(property3);
116  REQUIRE(symbol.has_any_property(property3) == true);
117 
118  auto property = property1 | property2;
119  REQUIRE(symbol.has_all_properties(property) == true);
120 
121  property |= property3;
122  REQUIRE(symbol.has_all_properties(property) == true);
123 
124  property = property2 | property3;
125  REQUIRE(symbol.has_all_properties(property) == true);
126 
127  property |= NmodlType::to_solve;
128  REQUIRE(symbol.has_all_properties(property) == false);
129  }
130  }
131  WHEN("remove properties from symbol") {
132  symbol.add_property(property1);
133  symbol.add_property(property2);
134  THEN("remove property that exists") {
135  REQUIRE(symbol.has_any_property(property2) == true);
136  symbol.remove_property(property2);
137  REQUIRE(symbol.has_any_property(property2) == false);
138  }
139  THEN("remove property that doesn't exist") {
140  REQUIRE(symbol.has_any_property(property3) == false);
141  symbol.remove_property(property3);
142  REQUIRE(symbol.has_any_property(property3) == false);
143  auto properties = property1 | property2;
144  REQUIRE(symbol.has_all_properties(properties) == true);
145  }
146  }
147  WHEN("combined properties") {
148  NmodlType property = NmodlType::factor_def | NmodlType::global_var;
149  THEN("symbol has union of all properties") {
150  REQUIRE(symbol.has_any_property(property) == false);
151  symbol.add_properties(property);
152  REQUIRE(symbol.has_any_property(property) == true);
153  property |= symbol.get_properties();
154  REQUIRE(symbol.get_properties() == property);
155  }
156  }
157  }

164 SCENARIO("Symbol table allows operations like insert, lookup") {
165  GIVEN("A global SymbolTable") {
166  auto program = std::make_shared<ast::Program>();
167  auto table = std::make_shared<SymbolTable>("Na", program.get(), true);
168  auto symbol = std::make_shared<Symbol>("alpha", ModToken());
169 
170  WHEN("checked methods and member variables") {
171  THEN("all members are initialized") {
172  REQUIRE(table->under_global_scope());
173  REQUIRE_THAT(table->name(), Catch::Contains("Na"));
174  REQUIRE_THAT(table->get_parent_table_name(), Catch::Contains("None"));
175  REQUIRE_THAT(table->position(), Catch::Contains("UNKNOWN"));
176  }
177  }
178  WHEN("insert symbol") {
179  table->insert(symbol);
180  THEN("table size increases") {
181  REQUIRE(table->symbol_count() == 1);
182  }
183  THEN("lookup returns a inserted symbol") {
184  REQUIRE(table->lookup("alpha") != nullptr);
185  REQUIRE(table->lookup("beta") == nullptr);
186  }
187  WHEN("re-inserting the same symbol") {
188  THEN("throws an exception") {
189  REQUIRE_THROWS_WITH(table->insert(symbol), Catch::Contains("re-insert"));
190  }
191  }
192  WHEN("inserting another symbol") {
193  auto next_symbol = std::make_shared<Symbol>("beta", ModToken());
194  table->insert(next_symbol);
195  THEN("symbol gets added and table size increases") {
196  REQUIRE(table->symbol_count() == 2);
197  REQUIRE(table->lookup("beta") != nullptr);
198  }
199  }
200  }
201  WHEN("checked for global variables") {
202  table->insert(symbol);
203  auto variables = table->get_variables_with_properties(NmodlType::range_var);
204  THEN("table doesn't have any global variables") {
205  REQUIRE(variables.empty());
206  WHEN("added global symbol") {
207  auto next_symbol = std::make_shared<Symbol>("gamma", ModToken());
208  next_symbol->add_property(NmodlType::assigned_definition);
209  table->insert(next_symbol);
210  auto variables = table->get_variables_with_properties(
211  NmodlType::assigned_definition);
212  THEN("table has global variable") {
213  REQUIRE(variables.size() == 1);
214  }
215  }
216  }
217  }
218  WHEN("added another symbol table as children") {
219  table->insert(symbol);
220  auto next_program = std::make_shared<ast::Program>();
221  auto next_table = std::make_shared<SymbolTable>("Ca", next_program.get(), true);
222  next_table->set_parent_table(table.get());
223  THEN("children symbol table can lookup into parent table scope") {
224  REQUIRE(next_table->lookup("alpha") == nullptr);
225  REQUIRE(next_table->lookup_in_scope("alpha") != nullptr);
226  }
227  }
228  WHEN("query for symbol with and without properties") {
229  auto symbol1 = std::make_shared<Symbol>("alpha", ModToken());
230  auto symbol2 = std::make_shared<Symbol>("beta", ModToken());
231  auto symbol3 = std::make_shared<Symbol>("gamma", ModToken());
232  auto symbol4 = std::make_shared<Symbol>("delta", ModToken());
233 
234  symbol1->add_property(NmodlType::range_var | NmodlType::param_assign);
235  symbol2->add_property(NmodlType::range_var | NmodlType::param_assign |
236  NmodlType::state_var);
237  symbol3->add_property(NmodlType::range_var | NmodlType::assigned_definition |
238  NmodlType::pointer_var);
239  symbol4->add_property(NmodlType::range_var);
240 
241  table->insert(symbol1);
242  table->insert(symbol2);
243  table->insert(symbol3);
244  table->insert(symbol4);
245 
246  auto result = table->get_variables_with_properties(NmodlType::range_var);
247  REQUIRE(result.size() == 4);
248 
249  result = table->get_variables_with_properties(NmodlType::range_var |
250  NmodlType::pointer_var);
251  REQUIRE(result.size() == 4);
252 
253  auto with = NmodlType::range_var | NmodlType::param_assign;
254  auto without = NmodlType::state_var | NmodlType::pointer_var;
255  result = table->get_variables(with, without);
256  REQUIRE(result.size() == 1);
257  REQUIRE(result[0]->get_name() == "alpha");
258 
259 
260  with = NmodlType::range_var;
261  without = NmodlType::param_assign | NmodlType::assigned_definition;
262  result = table->get_variables(with, without);
263  REQUIRE(result.size() == 1);
264  REQUIRE(result[0]->get_name() == "delta");
265 
266  with = NmodlType::range_var;
267  without = NmodlType::range_var;
268  result = table->get_variables(with, without);
269  REQUIRE(result.empty());
270  }
271  }
272 }

278 SCENARIO("Global symbol table (ModelSymbol) allows scope based operations") {
279  GIVEN("A Model symbolTable") {
280  ModelSymbolTable mod_symtab;
281 
282  auto program = std::make_shared<ast::Program>();
283  auto symbol1 = std::make_shared<Symbol>("alpha", ModToken());
284  auto symbol2 = std::make_shared<Symbol>("alpha", ModToken());
285  auto symbol3 = std::make_shared<Symbol>("alpha", ModToken());
286 
287  symbol1->add_property(NmodlType::param_assign);
288  symbol2->add_property(NmodlType::range_var);
289  symbol3->add_property(NmodlType::range_var);
290 
291  SymbolTable* old_symtab = nullptr;
292 
293  WHEN("trying to exit scope without entering") {
294  THEN("throws an exception") {
295  REQUIRE_THROWS_WITH(mod_symtab.leave_scope(), Catch::Contains("without entering"));
296  }
297  }
298  WHEN("trying to enter scope without valid node") {
299  THEN("throws an exception") {
300  REQUIRE_THROWS_WITH(mod_symtab.enter_scope("scope", nullptr, true, old_symtab),
301  Catch::Contains("empty node"));
302  }
303  }
304  WHEN("trying to insert without entering scope") {
305  THEN("throws an exception") {
306  auto symbol = std::make_shared<Symbol>("alpha", ModToken());
307  REQUIRE_THROWS_WITH(mod_symtab.insert(symbol), Catch::Contains("Can not insert"));
308  }
309  }
310  WHEN("enter scope multipel times") {
311  auto program1 = std::make_shared<ast::Program>();
312  auto program2 = std::make_shared<ast::Program>();
313  mod_symtab.enter_scope("scope1", program1.get(), false, old_symtab);
314  mod_symtab.enter_scope("scope2", program2.get(), false, old_symtab);
315  THEN("can leave scope multiple times") {
316  mod_symtab.leave_scope();
317  mod_symtab.leave_scope();
318  }
319  }
320  WHEN("added same symbol with different properties in global scope") {
321  mod_symtab.enter_scope("scope", program.get(), true, old_symtab);
322  mod_symtab.insert(symbol1);
323  mod_symtab.insert(symbol2);
324  THEN("only one symbol gets added with combined properties") {
325  auto symbol = mod_symtab.lookup("alpha");
326  auto properties = NmodlType::param_assign | NmodlType::range_var;
327  REQUIRE(symbol->get_properties() == properties);
328  }
329  }
330  WHEN("added same symbol with exisiting property") {
331  mod_symtab.enter_scope("scope", program.get(), true, old_symtab);
332  mod_symtab.insert(symbol1);
333  mod_symtab.insert(symbol2);
334  THEN("throws an exception") {
335  REQUIRE_THROWS_WITH(mod_symtab.insert(symbol3), Catch::Contains("Re-declaration"));
336  }
337  }
338  WHEN("added same symbol in children scope") {
339  mod_symtab.enter_scope("scope1", program.get(), true, old_symtab);
340  mod_symtab.insert(symbol2);
341  THEN("it's ok, just get overshadow warning") {
342  mod_symtab.enter_scope("scope2", program.get(), false, old_symtab);
343  mod_symtab.insert(symbol3);
344  ///\todo : not sure how to capture std::cout
345  }
346  }
347  }
