37 /// Trim from start
38 static inline std::string& ltrim(std::string& s) {
39  s.erase(s.begin(), std::find_if(s.begin(), s.end(), [](int c) { return !std::isspace(c); }));
40  return s;
41 }
42 
43 /// Trim from end
44 static inline std::string& rtrim(std::string& s) {
45  s.erase(std::find_if(s.rbegin(), s.rend(), [](int c) { return !std::isspace(c); }).base(),
46  s.end());
47  return s;
48 }

97 /// Left/Right/Center-aligns string within a field of width "width"
98 static inline std::string align_text(std::string text, int width, text_alignment type) {
99  /// left and right spacing
100  std::string left, right;
101 
102  /// count excess room to pad
103  int padding = width - text.size();
104 
105  if (padding > 0) {
106  if (type == text_alignment::left) {
107  right = std::string(padding, ' ');
108  } else if (type == text_alignment::right) {
109  left = std::string(padding, ' ');
110  } else {
111  left = std::string(padding / 2, ' ');
112  right = std::string(padding / 2, ' ');
113  /// if odd #, add one more space
114  if (padding > 0 && padding % 2 != 0) {
115  right += " ";
116  }
117  }
118  }
119  return left + text + right;
120 }
