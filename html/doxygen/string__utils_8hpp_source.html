39 /// Trim from start
40 static inline std::string& ltrim(std::string& s) {
41  s.erase(s.begin(), std::find_if(s.begin(), s.end(), [](int c) { return !std::isspace(c); }));
42  return s;
43 }
44 
45 /// Trim from end
46 static inline std::string& rtrim(std::string& s) {
47  s.erase(std::find_if(s.rbegin(), s.rend(), [](int c) { return !std::isspace(c); }).base(),
48  s.end());
49  return s;
50 }

52 /// Trim from both ends

99 /// Left/Right/Center-aligns string within a field of width "width"
100 static inline std::string align_text(std::string text, int width, text_alignment type) {
101  /// left and right spacing
102  std::string left, right;
103 
104  /// count excess room to pad
105  int padding = width - text.size();
106 
107  if (padding > 0) {
108  if (type == text_alignment::left) {
109  right = std::string(padding, ' ');
110  } else if (type == text_alignment::right) {
111  left = std::string(padding, ' ');
112  } else {
113  left = std::string(padding / 2, ' ');
114  right = std::string(padding / 2, ' ');
115  /// if odd #, add one more space
116  if (padding > 0 && padding % 2 != 0) {
117  right += " ";
118  }
119  }
120  }
121  return left + text + right;
122 }

130 /**
131  * Convert double value to string without trailing zeros
132  *
133  * When we use std::to_string with double value 1 then it gets
134  * printed as `1.000000`. This is not convenient for testing
135  * and testing/validation. To avoid this issue, we use to_string
136  * for integer values and stringstream for the rest.
137  */
138 static inline std::string to_string(double value, const std::string& format_spec = "{:.16g}") {
139  // double containing integer value
140  if (std::ceil(value) == value) {
141  return std::to_string(static_cast<int>(value));
142  }
143 
144  // actual float value
145  return fmt::format(format_spec, value);
146 }
