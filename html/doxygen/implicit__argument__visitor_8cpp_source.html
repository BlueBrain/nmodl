17 void ImplicitArgumentVisitor::visit_function_call(ast::FunctionCall& node) {
18  auto function_name = node.get_node_name();
19  auto const& arguments = node.get_arguments();
20  if (function_name == "nrn_ghk") {
21  // This function is traditionally used in MOD files with four arguments, but
22  // its value also depends on the global celsius variable so the real
23  // function in CoreNEURON has a 5th argument for that.
24  if (arguments.size() == 4) {
25  auto new_arguments = arguments;
26  new_arguments.insert(new_arguments.end(),
27  std::make_shared<ast::Name>(std::make_shared<ast::String>(
28  codegen::naming::CELSIUS_VARIABLE)));
29  node.set_arguments(std::move(new_arguments));
30  }
31  } else if (nmodl::details::needs_neuron_thread_first_arg(function_name)) {
32  // We need to insert `nt` as the first argument if it's not already
33  // there
34  auto const is_nt = [](auto const& arg) {
35  // Not all node types implement get_node_name(); for example if the
36  // first argument is `a+b` then calling `get_node_name` would throw
37  auto const node_type = arg.get_node_type();
38  using ast::AstNodeType;
39  if (node_type == AstNodeType::NAME || node_type == AstNodeType::STRING ||
40  node_type == AstNodeType::CONSTANT_VAR || node_type == AstNodeType::VAR_NAME ||
41  node_type == AstNodeType::LOCAL_VAR) {
42  return arg.get_node_name() == "nt";
43  }
44  return false;
45  };
46  if (arguments.empty() || !is_nt(*arguments.front())) {
47  auto new_arguments = arguments;
48  new_arguments.insert(new_arguments.begin(), std::make_shared<ast::String>("nt"));
49  node.set_arguments(std::move(new_arguments));
50  }
51  }
52  node.visit_children(*this);
53 }
