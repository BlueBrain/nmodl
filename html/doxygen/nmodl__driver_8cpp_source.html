33 std::shared_ptr<ast::Program> NmodlDriver::parse_file(const std::string& filename,
34  const location* loc) {
35  std::ifstream in(filename.c_str());
36  if (!in.good()) {
37  std::ostringstream oss;
38  if (loc == nullptr) {
39  oss << "NMODL Parser Error : ";
40  }
41  oss << "can not open file : " << filename;
42  if (loc != nullptr) {
43  parse_error(*loc, oss.str());
44  } else {
45  throw std::runtime_error(oss.str());
46  }
47  }
48 
49  auto current_stream_name = stream_name;
50  stream_name = filename;
51  auto absolute_path = utils::cwd() + utils::pathsep + filename;
52  {
53  const auto last_slash = filename.find_last_of(utils::pathsep);
54  if (utils::file_is_abs(filename)) {
55  const auto path_prefix = filename.substr(0, last_slash + 1);
56  library.push_current_directory(path_prefix);
57  absolute_path = filename;
58  } else if (last_slash == std::string::npos) {
59  library.push_current_directory(utils::cwd());
60  } else {
61  const auto path_prefix = filename.substr(0, last_slash + 1);
62  const auto path = utils::cwd() + utils::pathsep + path_prefix;
63  library.push_current_directory(path);
64  }
65  }
66 
67  open_files.emplace(absolute_path, loc);
68  parse_stream(in);
69  open_files.erase(absolute_path);
70  library.pop_current_directory();
71  stream_name = current_stream_name;
72 
73  return astRoot;
74 }

82 std::shared_ptr<ast::Include> NmodlDriver::parse_include(const std::string& name,
83  const location& loc) {
84  if (name.empty()) {
85  parse_error(loc, "empty filename");
86  }
87 
88  // Try to find directory containing the file to import
89  const auto directory_path = library.find_file(name);
90 
91  // Complete path of file (directory + filename).
92  std::string absolute_path = name;
93 
94  if (!directory_path.empty()) {
95  absolute_path = directory_path + std::string(1, utils::pathsep) + name;
96  }
97 
98  // Detect recursive inclusion.
99  auto already_included = open_files.find(absolute_path);
100  if (already_included != open_files.end()) {
101  std::ostringstream oss;
102  oss << name << ": recursive inclusion.\n";
103  if (already_included->second != nullptr) {
104  oss << *already_included->second << ": initial inclusion was here.";
105  }
106  parse_error(loc, oss.str());
107  }
108 
109  std::shared_ptr<ast::Program> program;
110  program.swap(astRoot);
111 
112  parse_file(absolute_path, &loc);
113 
114  program.swap(astRoot);
115  auto filename_node = std::shared_ptr<ast::String>(
116  new ast::String(std::string(1, '"') + name + std::string(1, '"')));
117  return std::shared_ptr<ast::Include>(new ast::Include(filename_node, program->get_blocks()));
118 }

142 std::string NmodlDriver::check_include_argument(const location& location,
143  const std::string& filename) {
144  if (filename.empty()) {
145  parse_error(location, "empty filename in INCLUDE directive");
146  } else if (filename.front() != '"' && filename.back() != '"') {
147  parse_error(location, "filename may start and end with \" character");
148  } else if (filename.size() == 3) {
149  parse_error(location, "filename is empty");
150  }
151  return filename.substr(1, filename.size() - 2);
152 }
