36 std::shared_ptr<ast::Program> NmodlDriver::parse_file(const fs::path& filename,
37  const location* loc) {
38  std::ifstream in(filename);
39  if (!in.good()) {
40  std::ostringstream oss;
41  if (loc == nullptr) {
42  oss << "NMODL Parser Error : ";
43  }
44  oss << "can not open file : " << filename;
45  if (loc != nullptr) {
46  parse_error(*loc, oss.str());
47  } else {
48  throw std::runtime_error(oss.str());
49  }
50  }
51 
52  auto current_stream_name = stream_name;
53  stream_name = filename.string();
54  auto absolute_path = fs::absolute(filename);
55  {
56  if (filename.is_absolute()) {
57  const auto path_prefix = filename.parent_path();
58  library.push_current_directory(path_prefix);
59  absolute_path = filename.string();
60  } else if (!filename.has_parent_path()) {
61  library.push_current_directory(fs::current_path());
62  } else {
63  const auto path_prefix = filename.parent_path();
64  const auto path = fs::absolute(path_prefix);
65  library.push_current_directory(path);
66  }
67  }
68 
69  open_files.emplace(absolute_path.string(), loc);
70  parse_stream(in);
71  open_files.erase(absolute_path.string());
72  library.pop_current_directory();
73  stream_name = current_stream_name;
74 
75  return astRoot;
76 }

84 std::shared_ptr<ast::Include> NmodlDriver::parse_include(const fs::path& name,
85  const location& loc) {
86  if (name.empty()) {
87  parse_error(loc, "empty filename");
88  }
89 
90  // Try to find directory containing the file to import
91  const auto directory_path = fs::path{library.find_file(name)};
92 
93  // Complete path of file (directory + filename).
94  auto absolute_path = name;
95 
96  if (!directory_path.empty()) {
97  absolute_path = directory_path / name;
98  }
99 
100  // Detect recursive inclusion.
101  auto already_included = open_files.find(absolute_path.string());
102  if (already_included != open_files.end()) {
103  std::ostringstream oss;
104  oss << name << ": recursive inclusion.\n";
105  if (already_included->second != nullptr) {
106  oss << *already_included->second << ": initial inclusion was here.";
107  }
108  parse_error(loc, oss.str());
109  }
110 
111  std::shared_ptr<ast::Program> program;
112  program.swap(astRoot);
113 
114  parse_file(absolute_path, &loc);
115 
116  program.swap(astRoot);
117  auto filename_node = std::shared_ptr<ast::String>(
118  new ast::String(fmt::format("\"{}\"", name.string())));
119  return std::shared_ptr<ast::Include>(new ast::Include(filename_node, program->get_blocks()));
120 }

155 std::string NmodlDriver::check_include_argument(const location& location,
156  const std::string& filename) {
157  if (filename.empty()) {
158  parse_error(location, "empty filename in INCLUDE directive");
159  } else if (filename.front() != '"' && filename.back() != '"') {
160  parse_error(location, "filename may start and end with \" character");
161  } else if (filename.size() == 3) {
162  parse_error(location, "filename is empty");
163  }
164  return filename.substr(1, filename.size() - 2);
165 }
