32 std::vector<std::string> run_kinetic_block_visitor(const std::string& text) {
33  std::vector<std::string> results;
34 
35  // construct AST from text including KINETIC block(s)
36  NmodlDriver driver;
37  const auto& ast = driver.parse_string(text);
38 
39  // construct symbol table from AST
40  SymtabVisitor().visit_program(*ast);
41 
42  // unroll loops and fold constants
43  ConstantFolderVisitor().visit_program(*ast);
44  LoopUnrollVisitor().visit_program(*ast);
45  ConstantFolderVisitor().visit_program(*ast);
46  SymtabVisitor().visit_program(*ast);
47 
48  // run KineticBlock visitor on AST
49  KineticBlockVisitor().visit_program(*ast);
50 
51  // run lookup visitor to extract DERIVATIVE block(s) from AST
52  const auto& blocks = collect_nodes(*ast, {AstNodeType::DERIVATIVE_BLOCK});
53  results.reserve(blocks.size());
54  for (const auto& block: blocks) {
55  results.push_back(to_nmodl(block));
56  }
57 
58 
59  // check that, after visitor rearrangement, parents are still up-to-date
60  CheckParentVisitor().check_ast(*ast);
61 
62  return results;
63 }

65 SCENARIO("Convert KINETIC to DERIVATIVE using KineticBlock visitor", "[kinetic][visitor]") {
66  GIVEN("KINETIC block with << reaction statement, 1 state var") {
67  static const std::string input_nmodl_text = R"(

69  x
70  }
