27 /**
28  * \details units::Unit definition is based only on pre-defined units, parse only the
29  * new unit and the pre-defined units. <br>
30  * Example:
31  * \code
32  * (nA) = (nanoamp) => nA nanoamp)
33  * \endcode
34  * The ast::UnitDef is converted to a string that is able to be parsed by the
35  * unit parser which was used for parsing the \c nrnunits.lib file.
36  * On \c nrnunits.lib constant "1" is defined as "fuzz", so it must be converted.
37  */
38 void UnitsVisitor::visit_unit_def(ast::UnitDef& node) {
39  std::ostringstream ss;
40  /*
41  * In nrnunits.lib file "1" is defined as "fuzz", so there
42  * must be a conversion to be able to parse "1" as unit
43  */
44  if (node.get_unit2()->get_node_name() == "1") {
45  ss << node.get_unit1()->get_node_name() << '\t';
46  ss << UNIT_FUZZ;
47  } else {
48  ss << node.get_unit1()->get_node_name() << '\t' << node.get_unit2()->get_node_name();
49  }
50 
51  // Parse the generated string for the defined unit using the units::UnitParser
52  units_driver.parse_string(ss.str());
53 }
54 
55 /**
56  * \details The new unit definition is based on a factor combined with
57  * units or other defined units.
58  * In the first case the factor saved to the ast::FactorDef node and
59  * printed to \c .cpp file is the one defined on the modfile. The factor
60  * and the dimensions saved to the units::UnitTable are based on the
61  * factor and the units defined in the modfile, so this factor will be
62  * calculated based on the base units of the units::UnitTable. <br>
63  * Example:
64  * \code
65  * R = 8.314 (volt-coul/degC))
66  * \endcode
67  * In the second case, the factor and the dimensions that are inserted
68  * to the units::UnitTable are based on the ast::FactorDef::unit1, like
69  * in MOD2C.
70  * \b However, the factor that is saved in the ast::FactorDef and printed
71  * in the \c .cpp file is the factor of the ast::FactorDef::unit1 divided
72  * by the factor of ast::FactorDef::unit2. <br>
73  * Example:
74  * \code
75  * R = (mole k) (mV-coulomb/degC)
76  * \endcode
77  * `unit1` is `mole k` and unit2 is `mV-coulomb/degC` <br>
78  * To parse the units defined in modfiles there are stringstreams
79  * created that are passed to the string parser, to be parsed by the
80  * unit parser used for parsing the \c nrnunits.lib file, which takes
81  * care of all the units calculations.
82  */
83 void UnitsVisitor::visit_factor_def(ast::FactorDef& node) {
84  std::ostringstream ss;
85  const auto node_has_value_defined_in_modfile = node.get_value() != nullptr;
86  if (node_has_value_defined_in_modfile) {
87  /*
88  * In nrnunits.lib file "1" is defined as "fuzz", so
89  * there must be a conversion to be able to parse "1" as unit
90  */
91  if (node.get_unit1()->get_node_name() == "1") {
92  ss << node.get_node_name() << "\t" << node.get_value()->eval() << " ";
93  ss << UNIT_FUZZ;
94  } else {
95  ss << node.get_node_name() << "\t" << node.get_value()->eval() << " ";
96  ss << node.get_unit1()->get_node_name();
97  }
98  // Parse the generated string for the defined unit using the units::UnitParser
99  units_driver.parse_string(ss.str());
100  } else {
101  std::ostringstream ss_unit1, ss_unit2;
102  std::string unit1_name, unit2_name;
103  /*
104  * In nrnunits.lib file "1" is defined as "fuzz", so
105  * there must be a conversion to be able to parse "1" as unit
106  */
107  if (node.get_unit1()->get_node_name() == "1") {
108  unit1_name = UNIT_FUZZ;
109  } else {
110  unit1_name = node.get_unit1()->get_node_name();
111  }
112  if (node.get_unit2()->get_node_name() == "1") {
113  unit2_name = UNIT_FUZZ;
114  } else {
115  unit2_name = node.get_unit2()->get_node_name();
116  }
117  /*
118  * Create dummy unit "node.get_node_name()_unit1" and parse
119  * it to calculate its factor
120  */
121  ss_unit1 << node.get_node_name() << "_unit1\t" << unit1_name;
122  units_driver.parse_string(ss_unit1.str());
123  /*
124  * Create dummy unit "node.get_node_name()_unit2" and parse
125  * it to calculate its factor
126  */
127  ss_unit2 << node.get_node_name() << "_unit2\t" << unit2_name;
128  units_driver.parse_string(ss_unit2.str());
129 
130  // Parse the generated string for the defined unit using the units::UnitParser
131  ss << node.get_node_name() << "\t" << unit1_name;
132  units_driver.parse_string(ss.str());
133 
134  /**
135  * \note If the ast::FactorDef was made by using two units (second case),
136  * the factors of both of them must be calculated based on the
137  * units::UnitTable and then they must be divided to produce the unit's
138  * factor that will be printed to the \c .cpp file. <br>
139  * Example:
140  * \code
141  * FARADAY = (faraday) (10000 coulomb)
142  * \endcode
143  * In the \c .cpp file the printed factor will be `9.64853090` but in the
144  * units::UnitTable the factor of `FARADAY` will be `96485.30900000`
145  */
146 
147  auto node_unit_name = node.get_node_name();
148  auto unit1_factor = units_driver.table->get_unit(node_unit_name + "_unit1")->get_factor();
149  auto unit2_factor = units_driver.table->get_unit(node_unit_name + "_unit2")->get_factor();
150 
151 #ifdef USE_LEGACY_UNITS
152  auto unit_factor = stringutils::to_string(unit1_factor / unit2_factor, "{:g}");
153 #else
154  auto unit_factor = stringutils::to_string(unit1_factor / unit2_factor, "{:.18g}");
155 #endif
156 
157  auto double_value_ptr = std::make_shared<ast::Double>(ast::Double(unit_factor));
158  node.set_value(std::move(double_value_ptr));
159  }
160 }
