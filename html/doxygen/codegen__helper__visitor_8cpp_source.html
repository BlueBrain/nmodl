26 /**
27  * How symbols are stored in NEURON? See notes written in markdown file.
28  *
29  * Some variables get printed by iterating over symbol table in mod2c.
30  * The example of this is thread variables (and also ions?). In this
31  * case we must have to arrange order if we are going keep compatibility
32  * with NEURON.
33  *
34  * Suppose there are three global variables: bcd, abc, abd, abe
35  * They will be in the 'a' bucket in order:
36  * abe, abd, abc
37  * and in 'b' bucket
38  * bcd
39  * So when we print thread variables, we first have to sort in the opposite
40  * order in which they come and then again order by first character in increasing
41  * order.
42  *
43  * Note that variables in double array do not need this transformation
44  * and it seems like they should just follow definition order.
45  */
46 void CodegenHelperVisitor::sort_with_mod2c_symbol_order(std::vector<SymbolType>& symbols) const {
47  /// first sort by global id to get in reverse order
48  std::sort(symbols.begin(),
49  symbols.end(),
50  [](const SymbolType& first, const SymbolType& second) -> bool {
51  return first->get_id() > second->get_id();
52  });
53 
54  /// now order by name (to be same as neuron's bucket)
55  std::sort(symbols.begin(),
56  symbols.end(),
57  [](const SymbolType& first, const SymbolType& second) -> bool {
58  return first->get_name()[0] < second->get_name()[0];
59  });
60 }

63 /**
64  * Find all ions used in mod file
65  */
66 void CodegenHelperVisitor::find_ion_variables(const ast::Program& node) {
67  // collect all use ion statements
68  const auto& ion_nodes = collect_nodes(node, {AstNodeType::USEION});
69 
70  // ion names, read ion variables and write ion variables
71  std::vector<std::string> ion_vars;
72  std::vector<std::string> read_ion_vars;
73  std::vector<std::string> write_ion_vars;
74 
75  for (const auto& ion_node: ion_nodes) {
76  const auto& ion = std::dynamic_pointer_cast<const ast::Useion>(ion_node);
77  ion_vars.push_back(ion->get_node_name());
78  for (const auto& var: ion->get_readlist()) {
79  read_ion_vars.push_back(var->get_node_name());
80  }
81  for (const auto& var: ion->get_writelist()) {
82  write_ion_vars.push_back(var->get_node_name());
83  }
84  }
85 
86  /**
87  * Check if given variable belongs to given ion.
88  * For example, eca belongs to ca ion, nai belongs to na ion.
89  * We just check if we exclude first/last char, if that is ion name.
90  */
91  auto ion_variable = [](const std::string& var, const std::string& ion) -> bool {
92  auto len = var.size() - 1;
93  return (var.substr(1, len) == ion || var.substr(0, len) == ion);
94  };
95 
96  /// iterate over all ion types and construct the Ion objects
97  for (auto& ion_name: ion_vars) {
98  Ion ion(ion_name);
99  for (auto& read_var: read_ion_vars) {
100  if (ion_variable(read_var, ion_name)) {
101  ion.reads.push_back(read_var);
102  }
103  }
104  for (auto& write_var: write_ion_vars) {
105  if (ion_variable(write_var, ion_name)) {
106  ion.writes.push_back(write_var);
107  if (ion.is_intra_cell_conc(write_var) || ion.is_extra_cell_conc(write_var)) {
108  ion.need_style = true;
109  info.write_concentration = true;
110  }
111  }
112  }
113  info.ions.push_back(std::move(ion));
114  }
115 
116  /// once ions are populated, we can find all currents
117  auto vars = psymtab->get_variables_with_properties(NmodlType::nonspecific_cur_var);
118  for (auto& var: vars) {
119  info.currents.push_back(var->get_name());
120  }
121  vars = psymtab->get_variables_with_properties(NmodlType::electrode_cur_var);
122  for (auto& var: vars) {
123  info.currents.push_back(var->get_name());
124  }
125  for (auto& ion: info.ions) {
126  for (auto& var: ion.writes) {
127  if (ion.is_ionic_current(var)) {
128  info.currents.push_back(var);
129  }
130  }
131  }
132 
133  /// check if write_conc(...) will be needed
134  for (const auto& ion: info.ions) {
135  for (const auto& var: ion.writes) {
136  if (!ion.is_ionic_current(var) && !ion.is_rev_potential(var)) {
137  info.require_wrote_conc = true;
138  }
139  }
140  }
141 }

144 /**
145  * Find non-range variables i.e. ones that are not belong to per instance allocation
146  *
147  * Certain variables like pointers, global, parameters are not necessary to be per
148  * instance variables. NEURON apply certain rules to determine which variables become
149  * thread, static or global variables. Here we construct those variables.
150  */
151 void CodegenHelperVisitor::find_non_range_variables() {
152  /**
153  * Top local variables are local variables appear in global scope. All local
154  * variables in program symbol table are in global scope.
155  */
156  info.constant_variables = psymtab->get_variables_with_properties(NmodlType::constant_var);
157  info.top_local_variables = psymtab->get_variables_with_properties(NmodlType::local_var);
158 
159  /**
160  * All global variables remain global if mod file is not marked thread safe.
161  * Otherwise, global variables written at least once gets promoted to thread variables.
162  */
163 
164  std::string variables;
165 
166  auto vars = psymtab->get_variables_with_properties(NmodlType::global_var);
167  for (auto& var: vars) {
168  if (info.thread_safe && var->get_write_count() > 0) {
169  var->mark_thread_safe();
170  info.thread_variables.push_back(var);
171  info.thread_var_data_size += var->get_length();
172  variables += " " + var->get_name();
173  } else {
174  info.global_variables.push_back(var);
175  }
176  }
177 
178  /**
179  * If parameter is not a range and used only as read variable then it becomes global
180  * variable. To qualify it as thread variable it must be be written at least once and
181  * mod file must be marked as thread safe.
182  * To exclusively get parameters only, we exclude all other variables (in without)
183  * and then sort them with neuron/mod2c order.
184  */
185  // clang-format off
186  auto with = NmodlType::param_assign;
187  auto without = NmodlType::range_var
188  | NmodlType::assigned_definition
189  | NmodlType::global_var
190  | NmodlType::pointer_var
191  | NmodlType::bbcore_pointer_var
192  | NmodlType::read_ion_var
193  | NmodlType::write_ion_var;
194  // clang-format on
195  vars = psymtab->get_variables(with, without);
196  for (auto& var: vars) {
197  // some variables like area and diam are declared in parameter
198  // block but they are not global
199  if (var->get_name() == naming::DIAM_VARIABLE || var->get_name() == naming::AREA_VARIABLE ||
200  var->has_any_property(NmodlType::extern_neuron_variable)) {
201  continue;
202  }
203 
204  // if model is thread safe and if parameter is being written then
205  // those variables should be promoted to thread safe variable
206  if (info.thread_safe && var->get_write_count() > 0) {
207  var->mark_thread_safe();
208  info.thread_variables.push_back(var);
209  info.thread_var_data_size += var->get_length();
210  } else {
211  info.global_variables.push_back(var);
212  }
213  }
214  sort_with_mod2c_symbol_order(info.thread_variables);
215 
216  /**
217  * \todo Below we calculate thread related id and sizes. This will
218  * need to do from global analysis pass as here we are handling
219  * top local variables, global variables, derivimplicit method.
220  * There might be more use cases with other solver methods.
221  */
222 
223  /**
224  * If derivimplicit is used, then first three thread ids get assigned to:
225  * 1st thread is used for: deriv_advance
226  * 2nd thread is used for: dith
227  * 3rd thread is used for: newtonspace
228  *
229  * slist and dlist represent the offsets for prime variables used. For
230  * euler or derivimplicit methods its always first number.
231  */
232  if (info.derivimplicit_used()) {
233  info.derivimplicit_var_thread_id = 0;
234  info.thread_data_index = 3;
235  info.derivimplicit_list_num = 1;
236  info.thread_callback_register = true;
237  }
238 
239  /// next thread id is allocated for top local variables
240  if (info.vectorize && !info.top_local_variables.empty()) {
241  info.top_local_thread_id = info.thread_data_index++;
242  info.thread_callback_register = true;
243  }
244 
245  /// next thread id is allocated for thread promoted variables
246  if (info.vectorize && !info.thread_variables.empty()) {
247  info.thread_var_thread_id = info.thread_data_index++;
248  info.thread_callback_register = true;
249  }
250 
251  /// find total size of local variables in global scope
252  for (auto& var: info.top_local_variables) {
253  info.top_local_thread_size += var->get_length();
254  }
255 
256  /// find number of prime variables and total size
257  auto primes = psymtab->get_variables_with_properties(NmodlType::prime_name);
258  info.num_primes = primes.size();
259  for (auto& variable: primes) {
260  info.primes_size += variable->get_length();
261  }
262 
263  /// find pointer or bbcore pointer variables
264  // clang-format off
265  auto properties = NmodlType::pointer_var
266  | NmodlType::bbcore_pointer_var;
267  // clang-format on
268  info.pointer_variables = psymtab->get_variables_with_properties(properties);
269 
270  // find special variables like diam, area
271  // clang-format off
272  properties = NmodlType::assigned_definition
273  | NmodlType::param_assign;
274  vars = psymtab->get_variables_with_properties(properties);
275  for (auto& var : vars) {
276  if (var->get_name() == naming::AREA_VARIABLE) {
277  info.area_used = true;
278  }
279  if (var->get_name() == naming::DIAM_VARIABLE) {
280  info.diam_used = true;
281  }
282  }
283  // clang-format on
284 }

286 /**
287  * Find range variables i.e. ones that are belong to per instance allocation
288  *
289  * In order to be compatible with NEURON, we need to print all variables in
290  * exact order as NEURON/MOD2C implementation. This is important because memory
291  * for all variables is allocated in single 1-D array with certain offset
292  * for each variable. The order of variables determine the offset and hence
293  * they must be in same order as NEURON.
294  *
295  * Here is how order is determined into NEURON/MOD2C implementation:
296  *
297  * First, following three lists are created
298  * - variables with parameter and range property (List 1)
299  * - variables with state and range property (List 2)
300  * - variables with assigned and range property (List 3)
301  *
302  * Once created, we remove some variables due to the following criteria:
303  * - In NEURON/MOD2C implementation, we remove variables with NRNPRANGEIN
304  * or NRNPRANGEOUT type
305  * - So who has NRNPRANGEIN and NRNPRANGEOUT type? these are USEION read
306  * or write variables that are not ionic currents.
307  * - This is the reason for mod files CaDynamics_E2.mod or cal_mig.mod, ica variable
308  * is printed earlier in the list but other variables like cai, cao don't appear
309  * in same order.
310  *
311  * Finally we create 4th list:
312  * - variables with assigned property and not in the previous 3 lists
313  *
314  * We now print the variables in following order:
315  *
316  * - List 1 i.e. range + parameter variables are printed first
317  * - List 3 i.e. range + assigned variables are printed next
318  * - List 2 i.e. range + state variables are printed next
319  * - List 4 i.e. assigned and ion variables not present in the previous 3 lists
320  *
321  * NOTE:
322  * - State variables also have the property `assigned_definition` but these variables
323  * are not from ASSIGNED block.
324  * - Variable can not be range as well as state, it's redeclaration error
325  * - Variable can be parameter as well as range. Without range, parameter
326  * is considered as global variable i.e. one value for all instances.
327  * - If a variable is only defined as RANGE and not in assigned or parameter
328  * or state block then it's not printed.
329  * - Note that a variable property is different than the variable type. For example,
330  * if variable has range property, it doesn't mean the variable is declared as RANGE.
331  * Other variables like STATE and ASSIGNED block variables also get range property
332  * without being explicitly declared as RANGE in the mod file.
333  * - Also, there is difference between declaration order vs. definition order. For
334  * example, POINTER variable in NEURON block is just declaration and doesn't
335  * determine the order in which they will get printed. Below we query symbol table
336  * and order all instance variables into certain order.
337  */
338 void CodegenHelperVisitor::find_range_variables() {
339  /// comparator to decide the order based on definition
340  auto comparator = [](const SymbolType& first, const SymbolType& second) -> bool {
341  return first->get_definition_order() < second->get_definition_order();
342  };
343 
344  /// from symbols vector `vars`, remove all ion variables which are not ionic currents
345  auto remove_non_ioncur_vars = [](SymbolVectorType& vars, const CodegenInfo& info) -> void {
346  vars.erase(std::remove_if(vars.begin(),
347  vars.end(),
348  [&](SymbolType& s) {
349  return info.is_ion_variable(s->get_name()) &&
350  !info.is_ionic_current(s->get_name());
351  }),
352  vars.end());
353  };
354 
355  /// if `secondary` vector contains any symbol that exist in the `primary` then remove it
356  auto remove_var_exist = [](SymbolVectorType& primary, SymbolVectorType& secondary) -> void {
357  secondary.erase(std::remove_if(secondary.begin(),
358  secondary.end(),
359  [&primary](const SymbolType& tosearch) {
360  return std::find_if(primary.begin(),
361  primary.end(),
362  // compare by symbol name
363  [&tosearch](
364  const SymbolType& symbol) {
365  return tosearch->get_name() ==
366  symbol->get_name();
367  }) != primary.end();
368  }),
369  secondary.end());
370  };
371 
372  /**
373  * First come parameters which are range variables.
374  */
375  // clang-format off
376  auto with = NmodlType::range_var
377  | NmodlType::param_assign;
378  auto without = NmodlType::global_var
379  | NmodlType::pointer_var
380  | NmodlType::bbcore_pointer_var
381  | NmodlType::state_var;
382 
383  // clang-format on
384  info.range_parameter_vars = psymtab->get_variables(with, without);
385  remove_non_ioncur_vars(info.range_parameter_vars, info);
386  std::sort(info.range_parameter_vars.begin(), info.range_parameter_vars.end(), comparator);
387 
388  /**
389  * Second come assigned variables which are range variables.
390  */
391  // clang-format off
392  with = NmodlType::range_var
393  | NmodlType::assigned_definition;
394  without = NmodlType::global_var
395  | NmodlType::pointer_var
396  | NmodlType::bbcore_pointer_var
397  | NmodlType::state_var
398  | NmodlType::param_assign;
399 
400  // clang-format on
401  info.range_assigned_vars = psymtab->get_variables(with, without);
402  remove_non_ioncur_vars(info.range_assigned_vars, info);
403  std::sort(info.range_assigned_vars.begin(), info.range_assigned_vars.end(), comparator);
404 
405 
406  /**
407  * Third come state variables. All state variables are kind of range by default.
408  * Note that some mod files like CaDynamics_E2.mod use cai as state variable which
409  * appear in USEION read/write list. These variables are not considered in this
410  * variables because non ionic-current variables are removed and printed later.
411  */
412  // clang-format off
413  with = NmodlType::state_var;
414  without = NmodlType::global_var
415  | NmodlType::pointer_var
416  | NmodlType::bbcore_pointer_var;
417 
418  // clang-format on
419  info.state_vars = psymtab->get_variables(with, without);
420  std::sort(info.state_vars.begin(), info.state_vars.end(), comparator);
421 
422  /// range_state_vars is copy of state variables but without non ionic current variables
423  info.range_state_vars = info.state_vars;
424  remove_non_ioncur_vars(info.range_state_vars, info);
425 
426  /// Remaining variables are assigned and ion variables which are not in the previous 3 lists
427 
428  // clang-format off
429  with = NmodlType::assigned_definition
430  | NmodlType::read_ion_var
431  | NmodlType::write_ion_var;
432  without = NmodlType::global_var
433  | NmodlType::pointer_var
434  | NmodlType::bbcore_pointer_var
435  | NmodlType::extern_neuron_variable;
436  // clang-format on
437  const auto& variables = psymtab->get_variables_with_properties(with, false);
438  for (const auto& variable: variables) {
439  if (!variable->has_any_property(without)) {
440  info.assigned_vars.push_back(variable);
441  }
442  }
443 
444  /// make sure that variables already present in previous lists
445  /// are removed to avoid any duplication
446  remove_var_exist(info.range_parameter_vars, info.assigned_vars);
447  remove_var_exist(info.range_assigned_vars, info.assigned_vars);
448  remove_var_exist(info.range_state_vars, info.assigned_vars);
449 
450  /// sort variables with their definition order
451  std::sort(info.assigned_vars.begin(), info.assigned_vars.end(), comparator);
452 }

591 /**
592  * Visit statement block and find prime symbols appear in derivative block
593  *
594  * Equation statements in derivative block has prime on the lhs. The order
595  * of primes could be different that declaration state block. Also, not all
596  * state variables need to appear in equation block. In this case, we want
597  * to find out the the primes in the order of equation definition. This is
598  * just to keep the same order as neuron implementation.
599  *
600  * The primes are already solved and replaced by Dstate or name. And hence
601  * we need to check if the lhs variable is derived from prime name. If it's
602  * Dstate then we have to lookup state to find out corresponding symbol. This
603  * is because prime_variables_by_order should contain state variable name and
604  * not the one replaced by solver pass.
605  */
606 void CodegenHelperVisitor::visit_statement_block(const ast::StatementBlock& node) {
607  const auto& statements = node.get_statements();
608  for (auto& statement: statements) {
609  statement->accept(*this);
610  if (under_derivative_block && assign_lhs &&
611  (assign_lhs->is_name() || assign_lhs->is_var_name())) {
612  auto name = assign_lhs->get_node_name();
613  auto symbol = psymtab->lookup(name);
614  if (symbol != nullptr) {
615  auto is_prime = symbol->has_any_property(NmodlType::prime_name);
616  auto from_state = symbol->has_any_status(Status::from_state);
617  if (is_prime || from_state) {
618  if (from_state) {
619  symbol = psymtab->lookup(name.substr(1, name.size()));
620  }
621  info.prime_variables_by_order.push_back(symbol);
622  info.num_equations++;
623  }
