25 /**
26  * How symbols are stored in NEURON? See notes written in markdown file.
27  *
28  * Some variables get printed by iterating over symbol table in mod2c.
29  * The example of this is thread variables (and also ions?). In this
30  * case we must have to arrange order if we are going keep compatibility
31  * with NEURON.
32  *
33  * Suppose there are three global variables: bcd, abc, abd, abe
34  * They will be in the 'a' bucket in order:
35  * abe, abd, abc
36  * and in 'b' bucket
37  * bcd
38  * So when we print thread variables, we first have to sort in the opposite
39  * order in which they come and then again order by first character in increasing
40  * order.
41  *
42  * Note that variables in double array do not need this transformation
43  * and it seems like they should just follow definition order.
44  */
45 void CodegenHelperVisitor::sort_with_mod2c_symbol_order(std::vector<SymbolType>& symbols) const {
46  /// first sort by global id to get in reverse order
47  std::sort(symbols.begin(),
48  symbols.end(),
49  [](const SymbolType& first, const SymbolType& second) -> bool {
50  return first->get_id() > second->get_id();
51  });
52 
53  /// now order by name (to be same as neuron's bucket)
54  std::sort(symbols.begin(),
55  symbols.end(),
56  [](const SymbolType& first, const SymbolType& second) -> bool {
57  return first->get_name()[0] < second->get_name()[0];
58  });
59 }

62 /**
63  * Find all ions used in mod file
64  */
65 void CodegenHelperVisitor::find_ion_variables() {
66  /// name of the ions used
67  auto ion_vars = psymtab->get_variables_with_properties(NmodlType::useion);
68  /// read variables from all ions
69  auto read_ion_vars = psymtab->get_variables_with_properties(NmodlType::read_ion_var);
70  /// write variables from all ions
71  auto write_ion_vars = psymtab->get_variables_with_properties(NmodlType::write_ion_var);
72 
73  /**
74  * Check if given variable belongs to given ion.
75  * For example, eca belongs to ca ion, nai belongs to na ion.
76  * We just check if we exclude first/last char, if that is ion name.
77  */
78  auto ion_variable = [](const std::string& var, const std::string& ion) -> bool {
79  auto len = var.size() - 1;
80  return (var.substr(1, len) == ion || var.substr(0, len) == ion);
81  };
82 
83  /// iterate over all ion types and construct the Ion objects
84  for (auto& ion_var: ion_vars) {
85  auto ion_name = ion_var->get_name();
86  Ion ion(ion_name);
87  for (auto& read_var: read_ion_vars) {
88  auto var = read_var->get_name();
89  if (ion_variable(var, ion_name)) {
90  ion.reads.push_back(var);
91  }
92  }
93  for (auto& write_var: write_ion_vars) {
94  auto varname = write_var->get_name();
95  if (ion_variable(varname, ion_name)) {
96  ion.writes.push_back(varname);
97  if (ion.is_intra_cell_conc(varname) || ion.is_extra_cell_conc(varname)) {
98  ion.need_style = true;
99  info.write_concentration = true;
100  }
101  }
102  }
103  info.ions.push_back(std::move(ion));
104  }
105 
106  /// once ions are populated, we can find all currents
107  auto vars = psymtab->get_variables_with_properties(NmodlType::nonspecific_cur_var);
108  for (auto& var: vars) {
109  info.currents.push_back(var->get_name());
110  }
111  vars = psymtab->get_variables_with_properties(NmodlType::electrode_cur_var);
112  for (auto& var: vars) {
113  info.currents.push_back(var->get_name());
114  }
115  for (auto& ion: info.ions) {
116  for (auto& var: ion.writes) {
117  if (ion.is_ionic_current(var)) {
118  info.currents.push_back(var);
119  }
120  }
121  }
122 
123  /// check if write_conc(...) will be needed
124  for (const auto& ion: info.ions) {
125  for (const auto& var: ion.writes) {
126  if (!ion.is_ionic_current(var) && !ion.is_rev_potential(var)) {
127  info.require_wrote_conc = true;
128  }
129  }
130  }
131 }

134 /**
135  * Find non-range variables i.e. ones that are not belong to per instance allocation
136  *
137  * Certain variables like pointers, global, parameters are not necessary to be per
138  * instance variables. NEURON apply certain rules to determine which variables become
139  * thread, static or global variables. Here we construct those variables.
140  */
141 void CodegenHelperVisitor::find_non_range_variables() {
142  /**
143  * Top local variables are local variables appear in global scope. All local
144  * variables in program symbol table are in global scope.
145  */
146  info.constant_variables = psymtab->get_variables_with_properties(NmodlType::constant_var);
147  info.top_local_variables = psymtab->get_variables_with_properties(NmodlType::local_var);
148 
149  /**
150  * All global variables remain global if mod file is not marked thread safe.
151  * Otherwise, global variables written at least once gets promoted to thread variables.
152  */
153 
154  std::string variables;
155 
156  auto vars = psymtab->get_variables_with_properties(NmodlType::global_var);
157  for (auto& var: vars) {
158  if (info.thread_safe && var->get_write_count() > 0) {
159  var->mark_thread_safe();
160  info.thread_variables.push_back(var);
161  info.thread_var_data_size += var->get_length();
162  variables += " " + var->get_name();
163  } else {
164  info.global_variables.push_back(var);
165  }
166  }
167 
168  /**
169  * If parameter is not a range and used only as read variable then it becomes global
170  * variable. To qualify it as thread variable it must be be written at least once and
171  * mod file must be marked as thread safe.
172  * To exclusively get parameters only, we exclude all other variables (in without)
173  * and then sort them with neuron/mod2c order.
174  */
175  // clang-format off
176  auto with = NmodlType::param_assign;
177  auto without = NmodlType::range_var
178  | NmodlType::assigned_definition
179  | NmodlType::global_var
180  | NmodlType::pointer_var
181  | NmodlType::bbcore_pointer_var
182  | NmodlType::read_ion_var
183  | NmodlType::write_ion_var;
184  // clang-format on
185  vars = psymtab->get_variables(with, without);
186  for (auto& var: vars) {
187  // some variables like area and diam are declared in parameter
188  // block but they are not global
189  if (var->get_name() == naming::DIAM_VARIABLE || var->get_name() == naming::AREA_VARIABLE ||
190  var->has_any_property(NmodlType::extern_neuron_variable)) {
191  continue;
192  }
193 
194  // if model is thread safe and if parameter is being written then
195  // those variables should be promoted to thread safe variable
196  if (info.thread_safe && var->get_write_count() > 0) {
197  var->mark_thread_safe();
198  info.thread_variables.push_back(var);
199  info.thread_var_data_size += var->get_length();
200  } else {
201  info.global_variables.push_back(var);
202  }
203  }
204  sort_with_mod2c_symbol_order(info.thread_variables);
205 
206  /**
207  * \todo Below we calculate thread related id and sizes. This will
208  * need to do from global analysis pass as here we are handling
209  * top local variables, global variables, derivimplicit method.
210  * There might be more use cases with other solver methods.
211  */
212 
213  /**
214  * If derivimplicit is used, then first three thread ids get assigned to:
215  * 1st thread is used for: deriv_advance
216  * 2nd thread is used for: dith
217  * 3rd thread is used for: newtonspace
218  *
219  * slist and dlist represent the offsets for prime variables used. For
220  * euler or derivimplicit methods its always first number.
221  */
222  if (info.derivimplicit_used()) {
223  info.derivimplicit_var_thread_id = 0;
224  info.thread_data_index = 3;
225  info.derivimplicit_list_num = 1;
226  info.thread_callback_register = true;
227  }
228 
229  /// next thread id is allocated for top local variables
230  if (info.vectorize && !info.top_local_variables.empty()) {
231  info.top_local_thread_id = info.thread_data_index++;
232  info.thread_callback_register = true;
233  }
234 
235  /// next thread id is allocated for thread promoted variables
236  if (info.vectorize && !info.thread_variables.empty()) {
237  info.thread_var_thread_id = info.thread_data_index++;
238  info.thread_callback_register = true;
239  }
240 
241  /// find total size of local variables in global scope
242  for (auto& var: info.top_local_variables) {
243  info.top_local_thread_size += var->get_length();
244  }
245 
246  /// find number of prime variables and total size
247  auto primes = psymtab->get_variables_with_properties(NmodlType::prime_name);
248  info.num_primes = primes.size();
249  for (auto& variable: primes) {
250  info.primes_size += variable->get_length();
251  }
252 
253  /// find pointer or bbcore pointer variables
254  // clang-format off
255  auto properties = NmodlType::pointer_var
256  | NmodlType::bbcore_pointer_var;
257  // clang-format on
258  info.pointer_variables = psymtab->get_variables_with_properties(properties);
259 
260  // find special variables like diam, area
261  // clang-format off
262  properties = NmodlType::assigned_definition
263  | NmodlType::param_assign;
264  vars = psymtab->get_variables_with_properties(properties);
265  for (auto& var : vars) {
266  if (var->get_name() == naming::AREA_VARIABLE) {
267  info.area_used = true;
268  }
269  if (var->get_name() == naming::DIAM_VARIABLE) {
270  info.diam_used = true;
271  }
272  }
273  // clang-format on
274 }

276 /**
277  * Find range variables i.e. ones that are belong to per instance allocation
278  *
279  * In order to be compatible with NEURON, we need to print all variables in
280  * exact order as NEURON/MOD2C implementation. This is important because memory
281  * for all variables is allocated in single 1-D array with certain offset
282  * for each variable. The order of variables determine the offset and hence
283  * they must be in same order as NEURON.
284  *
285  * Here is how order is determined into NEURON/MOD2C implementation:
286  *
287  * First, following three lists are created
288  * - variables with parameter and range property (List 1)
289  * - variables with state and range property (List 2)
290  * - variables with assigned and range property (List 3)
291  *
292  * Once created, we remove some variables due to the following criteria:
293  * - In NEURON/MOD2C implementation, we remove variables with NRNPRANGEIN
294  * or NRNPRANGEOUT type
295  * - So who has NRNPRANGEIN and NRNPRANGEOUT type? these are USEION read
296  * or write variables that are not ionic currents.
297  * - This is the reason for mod files CaDynamics_E2.mod or cal_mig.mod, ica variable
298  * is printed earlier in the list but other variables like cai, cao don't appear
299  * in same order.
300  *
301  * Finally we create 4th list:
302  * - variables with assigned property and not in the previous 3 lists
303  *
304  * We now print the variables in following order:
305  *
306  * - List 1 i.e. range + parameter variables are printed first
307  * - List 3 i.e. range + assigned variables are printed next
308  * - List 2 i.e. range + state variables are printed next
309  * - List 4 i.e. assigned and ion variables not present in the previous 3 lists
310  *
311  * NOTE:
312  * - State variables also have the property `assigned_definition` but these variables
313  * are not from ASSIGNED block.
314  * - Variable can not be range as well as state, it's redeclaration error
315  * - Variable can be parameter as well as range. Without range, parameter
316  * is considered as global variable i.e. one value for all instances.
317  * - If a variable is only defined as RANGE and not in assigned or parameter
318  * or state block then it's not printed.
319  * - Note that a variable property is different than the variable type. For example,
320  * if variable has range property, it doesn't mean the variable is declared as RANGE.
321  * Other variables like STATE and ASSIGNED block variables also get range property
322  * without being explicitly declared as RANGE in the mod file.
323  * - Also, there is difference between declaration order vs. definition order. For
324  * example, POINTER variable in NEURON block is just declaration and doesn't
325  * determine the order in which they will get printed. Below we query symbol table
326  * and order all instance variables into certain order.
327  */
328 void CodegenHelperVisitor::find_range_variables() {
329  /// comparator to decide the order based on definition
330  auto comparator = [](const SymbolType& first, const SymbolType& second) -> bool {
331  return first->get_definition_order() < second->get_definition_order();
332  };
333 
334  /// from symbols vector `vars`, remove all ion variables which are not ionic currents
335  auto remove_non_ioncur_vars = [](SymbolVectorType& vars, const CodegenInfo& info) -> void {
336  vars.erase(std::remove_if(vars.begin(),
337  vars.end(),
338  [&](SymbolType& s) {
339  return info.is_ion_variable(s->get_name()) &&
340  !info.is_ionic_current(s->get_name());
341  }),
342  vars.end());
343  };
344 
345  /// if `secondary` vector contains any symbol that exist in the `primary` then remove it
346  auto remove_var_exist = [](SymbolVectorType& primary, SymbolVectorType& secondary) -> void {
347  secondary.erase(std::remove_if(secondary.begin(),
348  secondary.end(),
349  [&primary](const SymbolType& tosearch) {
350  return std::find_if(primary.begin(),
351  primary.end(),
352  // compare by symbol name
353  [&tosearch](
354  const SymbolType& symbol) {
355  return tosearch->get_name() ==
356  symbol->get_name();
357  }) != primary.end();
358  }),
359  secondary.end());
360  };
361 
362  /**
363  * First come parameters which are range variables.
364  */
365  // clang-format off
366  auto with = NmodlType::range_var
367  | NmodlType::param_assign;
368  auto without = NmodlType::global_var
369  | NmodlType::pointer_var
370  | NmodlType::bbcore_pointer_var
371  | NmodlType::state_var;
372 
373  // clang-format on
374  info.range_parameter_vars = psymtab->get_variables(with, without);
375  remove_non_ioncur_vars(info.range_parameter_vars, info);
376  std::sort(info.range_parameter_vars.begin(), info.range_parameter_vars.end(), comparator);
377 
378  /**
379  * Second come assigned variables which are range variables.
380  */
381  // clang-format off
382  with = NmodlType::range_var
383  | NmodlType::assigned_definition;
384  without = NmodlType::global_var
385  | NmodlType::pointer_var
386  | NmodlType::bbcore_pointer_var
387  | NmodlType::state_var
388  | NmodlType::param_assign;
389 
390  // clang-format on
391  info.range_assigned_vars = psymtab->get_variables(with, without);
392  remove_non_ioncur_vars(info.range_assigned_vars, info);
393  std::sort(info.range_assigned_vars.begin(), info.range_assigned_vars.end(), comparator);
394 
395 
396  /**
397  * Third come state variables. All state variables are kind of range by default.
398  * Note that some mod files like CaDynamics_E2.mod use cai as state variable which
399  * appear in USEION read/write list. These variables are not considered in this
400  * variables because non ionic-current variables are removed and printed later.
401  */
402  // clang-format off
403  with = NmodlType::state_var;
404  without = NmodlType::global_var
405  | NmodlType::pointer_var
406  | NmodlType::bbcore_pointer_var;
407 
408  // clang-format on
409  info.state_vars = psymtab->get_variables(with, without);
410  std::sort(info.state_vars.begin(), info.state_vars.end(), comparator);
411 
412  /// range_state_vars is copy of state variables but without non ionic current variables
413  info.range_state_vars = info.state_vars;
414  remove_non_ioncur_vars(info.range_state_vars, info);
415 
416  /// Remaining variables are assigned and ion variables which are not in the previous 3 lists
417 
418  // clang-format off
419  with = NmodlType::assigned_definition
420  | NmodlType::read_ion_var
421  | NmodlType::write_ion_var;
422  without = NmodlType::global_var
423  | NmodlType::pointer_var
424  | NmodlType::bbcore_pointer_var
425  | NmodlType::extern_neuron_variable;
426  // clang-format on
427  const auto& variables = psymtab->get_variables_with_properties(with, false);
428  for (const auto& variable: variables) {
429  if (!variable->has_any_property(without)) {
430  info.assigned_vars.push_back(variable);
431  }
432  }
433 
434  /// make sure that variables already present in previous lists
435  /// are removed to avoid any duplication
436  remove_var_exist(info.range_parameter_vars, info.assigned_vars);
437  remove_var_exist(info.range_assigned_vars, info.assigned_vars);
438  remove_var_exist(info.range_state_vars, info.assigned_vars);
439 
440  /// sort variables with their definition order
441  std::sort(info.assigned_vars.begin(), info.assigned_vars.end(), comparator);
442 }

567 /**
568  * Visit statement block and find prime symbols appear in derivative block
569  *
570  * Equation statements in derivative block has prime on the lhs. The order
571  * of primes could be different that declaration state block. Also, not all
572  * state variables need to appear in equation block. In this case, we want
573  * to find out the the primes in the order of equation definition. This is
574  * just to keep the same order as neuron implementation.
575  *
576  * The primes are already solved and replaced by Dstate or name. And hence
577  * we need to check if the lhs variable is derived from prime name. If it's
578  * Dstate then we have to lookup state to find out corresponding symbol. This
579  * is because prime_variables_by_order should contain state variable name and
580  * not the one replaced by solver pass.
581  */
582 void CodegenHelperVisitor::visit_statement_block(const ast::StatementBlock& node) {
583  const auto& statements = node.get_statements();
584  for (auto& statement: statements) {
585  statement->accept(*this);
586  if (under_derivative_block && assign_lhs &&
587  (assign_lhs->is_name() || assign_lhs->is_var_name())) {
588  auto name = assign_lhs->get_node_name();
589  auto symbol = psymtab->lookup(name);
590  if (symbol != nullptr) {
591  auto is_prime = symbol->has_any_property(NmodlType::prime_name);
592  auto from_state = symbol->has_any_status(Status::from_state);
593  if (is_prime || from_state) {
594  if (from_state) {
595  symbol = psymtab->lookup(name.substr(1, name.size()));
596  }
597  info.prime_variables_by_order.push_back(symbol);
598  info.num_equations++;
599  }
600  }
601  }
602  assign_lhs = nullptr;
603  }
604 }
