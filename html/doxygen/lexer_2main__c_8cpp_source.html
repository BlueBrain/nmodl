28 void scan_c_code(std::istream& in) {
29  nmodl::parser::CDriver driver;
30  nmodl::parser::CLexer scanner(driver, &in);
31 
32  /// parse C file and print token until EOF
33  while (true) {
34  auto sym = scanner.next_token();
35  auto token_type = sym.type_get();
36  if (token_type == parser::CParser::by_type(Token::END).type_get()) {
37  break;
38  }
39  std::cout << sym.value.as<std::string>() << std::endl;
40  }
41 }
