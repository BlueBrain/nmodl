51 std::string CodegenCompatibilityVisitor::return_error_if_solve_method_is_unhandled(
52  ast::Ast& node,
53  const std::shared_ptr<ast::Ast>& ast_node) {
54  auto solve_block_ast_node = std::dynamic_pointer_cast<ast::SolveBlock>(ast_node);
55  std::stringstream unhandled_method_error_message;
56  auto method = solve_block_ast_node->get_method();
57  if (method == nullptr)
58  return "";
59  auto unhandled_solver_method = handled_solvers.find(method->get_node_name()) ==
60  handled_solvers.end();
61  if (unhandled_solver_method) {
62  unhandled_method_error_message
63  << "\"{}\" solving method used at [{}] not handled. Supported methods are cnexp, euler, derivimplicit and sparse\n"_format(
64  method->get_node_name(), method->get_token()->position());
65  }
66  return unhandled_method_error_message.str();
67 }

90 std::string CodegenCompatibilityVisitor::return_error_if_no_bbcore_read_write(
91  ast::Ast& node,
92  const std::shared_ptr<ast::Ast>& ast_node) {
93  std::stringstream error_message_no_bbcore_read_write;
94  const auto& verbatim_nodes = collect_nodes(node, {AstNodeType::VERBATIM});
95  auto found_bbcore_read = false;
96  auto found_bbcore_write = false;
97  for (const auto& it: verbatim_nodes) {
98  auto verbatim = std::dynamic_pointer_cast<ast::Verbatim>(it);
99 
100  auto verbatim_statement_string = verbatim->get_statement()->get_value();
101 
102  // Parse verbatim block to find out if there is a token "bbcore_read" or
103  // "bbcore_write". If there is not, then the function is not defined or
104  // is commented.
105  parser::CDriver driver;
106 
107  driver.scan_string(verbatim_statement_string);
108  auto tokens = driver.all_tokens();
109 
110  for (const auto& token: tokens) {
111  if (token == "bbcore_read") {
112  found_bbcore_read = true;
113  }
114  if (token == "bbcore_write") {
115  found_bbcore_write = true;
116  }
117  }
118  }
119  if (!found_bbcore_read) {
120  error_message_no_bbcore_read_write
121  << "\"bbcore_read\" function not defined in any VERBATIM block\n";
122  }
123  if (!found_bbcore_write) {
124  error_message_no_bbcore_read_write
125  << "\"bbcore_write\" function not defined in any VERBATIM block\n";
126  }
127  return error_message_no_bbcore_read_write.str();
128 }

131  * \details Checks all the ast::AstNodeType that are not handled in NMODL code
132  * generation backend for CoreNEURON and prints related messages. If there is
133  * some kind of incompatibility return false.

136 bool CodegenCompatibilityVisitor::find_unhandled_ast_nodes(Ast& node) {
137  std::vector<ast::AstNodeType> unhandled_ast_types;
138  unhandled_ast_types.reserve(unhandled_ast_types_func.size());
139  for (auto kv: unhandled_ast_types_func) {
140  unhandled_ast_types.push_back(kv.first);
141  }
142  unhandled_ast_nodes = collect_nodes(node, unhandled_ast_types);
143 
144  std::stringstream ss;
145  for (const auto& it: unhandled_ast_nodes) {
146  auto node_type = it->get_node_type();
147  ss << (this->*unhandled_ast_types_func.find(node_type)->second)(node, it);
148  }
149  if (!ss.str().empty()) {
150  logger->error("Code incompatibility detected");
151  logger->error("Cannot translate mod file to .cpp file");
152  logger->error("Fix the following errors and try again");
153  std::string line;
154  std::istringstream ss_stringstream(ss.str());
155  while (std::getline(ss_stringstream, line)) {
156  if (!line.empty())
157  logger->error("Code Incompatibility :: {}"_format(line));
158  }
159  return true;
160  }
161  return false;
162 }
