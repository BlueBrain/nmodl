168 std::vector<std::string> to_string_vector(const Status& obj) {
169  std::vector<std::string> status;
170 
171  // do nothing for Status::empty
172 
173  if (has_status(obj, Status::localized)) {
174  status.emplace_back("localized");
175  }
176 
177  if (has_status(obj, Status::globalized)) {
178  status.emplace_back("globalized");
179  }
180 
181  if (has_status(obj, Status::inlined)) {
182  status.emplace_back("inlined");
183  }
184 
185  if (has_status(obj, Status::renamed)) {
186  status.emplace_back("renamed");
187  }
188 
189  if (has_status(obj, Status::created)) {
190  status.emplace_back("created");
191  }
192 
193  if (has_status(obj, Status::from_state)) {
194  status.emplace_back("from_state");
195  }
196 
197  if (has_status(obj, Status::thread_safe)) {
198  status.emplace_back("thread_safe");
199  }
200 
201  return status;
202 }
