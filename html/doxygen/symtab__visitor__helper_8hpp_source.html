22 // create symbol for given node
23 static std::shared_ptr<Symbol> create_symbol_for_node(ast::Node* node,
24  NmodlType property,
25  bool under_state_block) {
26  ModToken token;
27  auto token_ptr = node->get_token();
28  if (token_ptr != nullptr) {
29  token = *token_ptr;
30  }
31 
32  /// add new symbol
33  auto symbol = std::make_shared<Symbol>(node->get_node_name(), node, token);
34  symbol->add_property(property);
35 
36  // non specific variable is range
37  if (property == NmodlType::nonspecific_cur_var) {
38  symbol->add_property(NmodlType::range_var);
39  }
40 
41  /// extra property for state variables
42  if (under_state_block) {
43  symbol->add_property(NmodlType::state_var);
44  }
45  return symbol;
46 }

49 /// helper function to setup/insert symbol into symbol table
50 /// for the ast nodes which are of variable types
51 void SymtabVisitor::setup_symbol(ast::Node* node, NmodlType property) {
52  std::shared_ptr<Symbol> symbol;
53  auto name = node->get_node_name();
54 
55  /// if prime variable is already exist in symbol table
56  /// then just update the order
57  if (node->is_prime_name()) {
58  auto prime = dynamic_cast<ast::PrimeName*>(node);
59  symbol = modsymtab->lookup(name);
60  if (symbol) {
61  symbol->set_order(prime->get_order()->eval());
62  symbol->add_property(property);
63  return;
64  }
65  }
66 
67  /// range and non_spec_cur can appear in any order in neuron block.
68  /// for both properties, we have to check if symbol is already exist.
69  /// if so we have to return to avoid duplicate definition error.
70  if (property == NmodlType::range_var || property == NmodlType::nonspecific_cur_var) {
71  auto s = modsymtab->lookup(name);
72  if (s && s->has_any_property(NmodlType::nonspecific_cur_var | NmodlType::range_var)) {
73  s->add_property(property);
74  return;
75  }
76  }
77 
78  symbol = create_symbol_for_node(node, property, under_state_block);
79 
80  /// insert might return different symbol if already exist in the same scope
81  symbol = modsymtab->insert(symbol);
82 
83  if (node->is_param_assign()) {
84  auto parameter = dynamic_cast<ast::ParamAssign*>(node);
85  const auto& value = parameter->get_value();
86  const auto& name = parameter->get_name();
87  if (value) {
88  symbol->set_value(value->to_double());
89  }
90  if (name->is_indexed_name()) {
91  auto index_name = dynamic_cast<ast::IndexedName*>(name.get());
92  auto length = dynamic_cast<ast::Integer*>(index_name->get_length().get());
93  symbol->set_as_array(length->eval());
94  }
95  }
96 
97  if (node->is_assigned_definition()) {
98  auto variable = dynamic_cast<ast::AssignedDefinition*>(node);
99  auto length = variable->get_length();
100  if (length) {
101  symbol->set_as_array(length->eval());
102  }
103  }
104 
105  if (node->is_constant_var()) {
106  auto constant = dynamic_cast<ast::ConstantVar*>(node);
107  auto value = constant->get_value();
108  if (value) {
109  symbol->set_value(value->to_double());
110  }
111  }
112 
113  if (node->is_local_var()) {
114  auto variable = dynamic_cast<ast::LocalVar*>(node);
115  auto name = variable->get_name();
116  if (name->is_indexed_name()) {
117  auto index_name = dynamic_cast<ast::IndexedName*>(name.get());
118  auto length = dynamic_cast<ast::Integer*>(index_name->get_length().get());
119  symbol->set_as_array(length->eval());
120  }
121  }
122 
123  if (node->is_define()) {
124  auto define = dynamic_cast<ast::Define*>(node);
125  symbol->set_value(define->get_value()->to_double());
126  }
127 
128  /// visit children, most likely variables are already
129  /// leaf nodes, not necessary to visit
130  node->visit_children(*this);
131 }

135  auto token = node->get_token();

148 static void add_external_symbols(symtab::ModelSymbolTable* symtab) {
149  ModToken tok(true);
150  auto variables = nmodl::get_external_variables();
151  for (auto variable : variables) {
152  auto symbol = std::make_shared<Symbol>(variable, nullptr, tok);
153  symbol->add_property(NmodlType::extern_neuron_variable);
154  symtab->insert(symbol);
155  }
156  auto methods = nmodl::get_external_functions();
157  for (auto method : methods) {
158  auto symbol = std::make_shared<Symbol>(method, nullptr, tok);
159  symbol->add_property(NmodlType::extern_method);
160  symtab->insert(symbol);
161  }
162 }

165 void SymtabVisitor::setup_symbol_table(ast::Ast* node, const std::string& name, bool is_global) {
166  /// entering into new nmodl block
167  auto symtab = modsymtab->enter_scope(name, node, is_global, node->get_symbol_table());
168 
169  if (node->is_state_block()) {
170  under_state_block = true;
171  }
172 
173  /// there is only one solve statement allowed in mod file
174  if (node->is_solve_block()) {
175  auto solve_block = dynamic_cast<ast::SolveBlock*>(node);
176  block_to_solve.insert(solve_block->get_block_name()->get_node_name());
177  }
178 
179  /// not required at the moment but every node
180  /// has pointer to associated symbol table
181  node->set_symbol_table(symtab);
182 
183  /// when visiting highest level node i.e. Program, we insert
184  /// all global variables to the global symbol table
185  if (node->is_program()) {
186  add_external_symbols(modsymtab);
187  }
188 
189  /// look for all children blocks recursively
190  node->visit_children(*this);
191 
192  /// existing nmodl block
193  modsymtab->leave_scope();
194 
195  if (node->is_state_block()) {
196  under_state_block = false;
197  }
198 }

222 /**
223  * Visit table statement and update symbol in symbol table
224  *
225  * @todo we assume table statement follows variable declaration
226  */
227 void SymtabVisitor::visit_table_statement(ast::TableStatement& node) {
228  auto update_symbol = [this](const ast::NameVector& variables, NmodlType property, int num_values) {
229  for (auto& var : variables) {
230  auto name = var->get_node_name();
231  auto symbol = modsymtab->lookup(name);
232  if (symbol) {
233  symbol->add_property(property);
234  symbol->set_num_values(num_values);
235  }
236  }
237  };
238  int num_values = node.get_with()->eval() + 1;
239  update_symbol(node.get_table_vars(), NmodlType::table_statement_var, num_values);
240  update_symbol(node.get_depend_vars(), NmodlType::table_assigned_var, num_values);
241 }
