28 /// retrieve token type from lexer and check if it's of given type
29 bool check_token_type(const std::string& name, TokenType type) {
30  std::istringstream ss(name);
31  std::istream& in = ss;
32 
33  NmodlDriver driver;
34  NmodlLexer scanner(driver, &in);
35 
36  SymbolType sym = scanner.next_token();
37  int token_type = sym.type_get();
38 
39  auto get_token_type = [](TokenType token) {
40  return parser::NmodlParser::by_type(token).type_get();
41  };
42 
43  /**
44  * Lexer returns raw pointers for some AST types
45  * and we need to clean-up memory for those.
46  * Todo: add tests later for checking values
47  */
48  // clang-format off
49  if (token_type == get_token_type(Token::NAME) ||
50  token_type == get_token_type(Token::METHOD) ||
51  token_type == get_token_type(Token::SUFFIX) ||
52  token_type == get_token_type(Token::VALENCE) ||
53  token_type == get_token_type(Token::DEL) ||
54  token_type == get_token_type(Token::DEL2)) {
55  auto value = sym.value.as<ast::Name>();
56  REQUIRE(value.get_node_name() != "");
57  }
58  // clang-format on
59  // prime variable
60  else if (token_type == get_token_type(Token::PRIME)) {
61  auto value = sym.value.as<ast::PrimeName>();
62  REQUIRE(value.get_node_name() != "");
63  }
64  // integer constant
65  else if (token_type == get_token_type(Token::INTEGER)) {
66  auto value = sym.value.as<ast::Integer>();
67  REQUIRE(value.get_value() != 0);
68  }
69  // float constant
70  else if (token_type == get_token_type(Token::REAL)) {
71  auto value = sym.value.as<ast::Double>();
72  REQUIRE(value.to_double() != 0);
73  }
74  // const char*
75  else if (token_type == get_token_type(Token::STRING)) {
76  auto value = sym.value.as<ast::String>();
77  REQUIRE(value.get_value() != "");
78  }
79  // string block representation verbatim or block comment
80  else if (token_type == get_token_type(Token::VERBATIM) ||
81  token_type == get_token_type(Token::BLOCK_COMMENT) ||
82  token_type == get_token_type(Token::LINE_PART)) {
83  auto value = sym.value.as<std::string>();
84  REQUIRE(value != "");
85  }
86  // rest of the tokens
87  else {
88  auto value = sym.value.as<ModToken>();
89  REQUIRE(value.text() != "");
90  }
91 
92  return sym.type_get() == get_token_type(type);
93 }
