57 void CodePrinter::add_multi_line(const std::string& text) {
58  const auto& lines = stringutils::split_string(text, '\n');
59 
60  int prefix_length{};
61  int start_line{};
62  while (start_line < lines.size()) {
63  const auto& line = lines[start_line];
64  // skip first empty line, if any
65  if (line.empty()) {
66  ++start_line;
67  continue;
68  }
69  // The common indentation of all blocks if the number of spaces
70  // at the beginning of the first non-empty line.
71  for (auto line_it = line.begin(); line_it != line.end() && *line_it == ' '; ++line_it) {
72  prefix_length += 1;
73  }
74  break;
75  }
76 
77  for (const auto& line: lines) {
78  if (line.size() < prefix_length) {
79  // ignore lines made of ' ' characters
80  if (std::find_if_not(line.begin(), line.end(), [](char c) { return c == ' '; }) !=
81  line.end()) {
82  throw std::invalid_argument("Indentation mismatch");
83  }
84  } else {
85  add_line(line.substr(prefix_length));
86  }
87  }
88 }
