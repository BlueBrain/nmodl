22 /**
23  * Depending programming model and compiler, we print compiler hint
24  * for parallelization. For example:
25  *
26  * #pragma ivdep
27  * for(int id=0; id<nodecount; id++) {
28  *
29  * #pragma acc parallel loop
30  * for(int id=0; id<nodecount; id++) {
31  *
32  */
33 void CodegenAccVisitor::print_channel_iteration_block_parallel_hint(BlockType type) {
34  if (info.artificial_cell) {
35  return;
36  }
37 
38  std::ostringstream present_clause;
39  present_clause << "present(inst";
40 
41  if (type == BlockType::NetReceive) {
42  present_clause << ", nrb";
43  } else {
44  present_clause << ", node_index, data, voltage, indexes, thread";
45  if (type == BlockType::Equation) {
46  present_clause << ", vec_rhs, vec_d";
47  }
48  }
49  present_clause << ')';
50  printer->add_line(
51  "#pragma acc parallel loop {} async(nt->stream_id) if(nt->compute_gpu)"_format(
52  present_clause.str()));
53 }

63 void CodegenAccVisitor::print_backend_includes() {
64  /**
65  * Artificial cells are executed on CPU. As Random123 is allocated on GPU by default,
66  * we have to disable GPU allocations using `DISABLE_OPENACC` macro.
67  */
68  if (info.artificial_cell) {
69  printer->add_line("#undef DISABLE_OPENACC");
70  printer->add_line("#define DISABLE_OPENACC");
71  } else {
72  printer->add_line("#include <cuda.h>");
73  printer->add_line("#include <cuda_runtime_api.h>");
74  printer->add_line("#include <openacc.h>");
75  }

84 void CodegenAccVisitor::print_memory_allocation_routine() const {
85  // memory for artificial cells should be allocated on CPU
86  if (info.artificial_cell) {
87  CodegenCVisitor::print_memory_allocation_routine();
88  return;
89  }
90  printer->add_newline(2);
91  auto args = "size_t num, size_t size, size_t alignment = 16";
92  printer->add_line("static inline void* mem_alloc({}) {}"_format(args, "{"));
93  printer->add_line(" void* ptr;");
94  printer->add_line(" cudaMallocManaged(&ptr, num*size);");
95  printer->add_line(" cudaMemset(ptr, 0, num*size);");
96  printer->add_line(" return ptr;");
97  printer->add_line("}");
98 
99  printer->add_newline(2);
100  printer->add_line("static inline void mem_free(void* ptr) {");
101  printer->add_line(" cudaFree(ptr);");
102  printer->add_line("}");
103 }

105 /**
106  * OpenACC kernels running on GPU doesn't support `abort()`. CUDA/OpenACC supports
107  * `assert()` in device kernel that can be used for similar purpose. Also, `printf`
108  * is supported on device.
109  *
110  * @todo : we need to implement proper error handling mechanism to propogate errors
111  * from GPU to CPU. For example, error code can be returned like original
112  * neuron implementation. For now we use `assert(0==1)` pattern which is
113  * used for OpenACC/CUDA.
114  */
115 void CodegenAccVisitor::print_abort_routine() const {
116  printer->add_newline(2);
117  printer->add_line("static inline void coreneuron_abort() {");
118  printer->add_line(" printf(\"Error : Issue while running OpenACC kernel \\n\");");
119  printer->add_line(" assert(0==1);");
120  printer->add_line("}");
121 }

130 /**
131  * Each kernel like nrn_init, nrn_state and nrn_cur could be offloaded
132  * to accelerator. In this case, at very top level, we print pragma
133  * for data present. For example:
134  *
135  * \code{.cpp}
136  * void nrn_state(...) {
137  * #pragma acc data present (nt, ml...)
138  * {
139  *
140  * }
141  * }
142  * \endcode
143  */
144 void CodegenAccVisitor::print_kernel_data_present_annotation_block_begin() {
145  if (!info.artificial_cell) {
146  auto global_variable = "{}_global"_format(info.mod_suffix);
147  printer->add_line(
148  "#pragma acc data present(nt, ml, {}) if(nt->compute_gpu)"_format(global_variable));
149  printer->add_line("{");
150  printer->increase_indent();
151  }
152 }

170  auto d_op = operator_for_d();
