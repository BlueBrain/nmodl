22 /**
23  * Depending programming model and compiler, we print compiler hint
24  * for parallelization. For example:
25  *
26  * #pragma ivdep
27  * for(int id=0; id<nodecount; id++) {
28  *
29  * #pragma acc parallel loop
30  * for(int id=0; id<nodecount; id++) {
31  *
32  */
33 void CodegenAccVisitor::print_channel_iteration_block_parallel_hint(BlockType type) {
34  if (info.artificial_cell) {
35  return;
36  }
37 
38  std::ostringstream present_clause;
39  present_clause << "present(inst";
40 
41  if (type == BlockType::NetReceive) {
42  present_clause << ", nrb";
43  } else {
44  present_clause << ", node_index, data, voltage, indexes, thread";
45  if (type == BlockType::Equation) {
46  present_clause << ", vec_rhs, vec_d";
47  }
48  }
49  present_clause << ')';
50  printer->add_line(
51  "#pragma acc parallel loop {} async(nt->stream_id)"_format(present_clause.str()));
52 }

62 void CodegenAccVisitor::print_backend_includes() {
63  /**
64  * Artificial cells are executed on CPU. As Random123 is allocated on GPU by default,
65  * we have to disable GPU allocations using `DISABLE_OPENACC` macro.
66  */
67  if (info.artificial_cell) {
68  printer->add_line("#undef DISABLE_OPENACC");
69  printer->add_line("#define DISABLE_OPENACC");
70  } else {
71  printer->add_line("#include <cuda.h>");
72  printer->add_line("#include <cuda_runtime_api.h>");
73  printer->add_line("#include <openacc.h>");
74  }

83 void CodegenAccVisitor::print_memory_allocation_routine() const {
84  // memory for artificial cells should be allocated on CPU
85  if (info.artificial_cell) {
86  CodegenCVisitor::print_memory_allocation_routine();
87  return;
88  }
89  printer->add_newline(2);
90  auto args = "size_t num, size_t size, size_t alignment = 16";
91  printer->add_line("static inline void* mem_alloc({}) {}"_format(args, "{"));
92  printer->add_line(" void* ptr;");
93  printer->add_line(" cudaMallocManaged(&ptr, num*size);");
94  printer->add_line(" cudaMemset(ptr, 0, num*size);");
95  printer->add_line(" return ptr;");
96  printer->add_line("}");
97 
98  printer->add_newline(2);
99  printer->add_line("static inline void mem_free(void* ptr) {");
100  printer->add_line(" cudaFree(ptr);");
101  printer->add_line("}");
102 }

104 /**
105  * OpenACC kernels running on GPU doesn't support `abort()`. CUDA/OpenACC supports
106  * `assert()` in device kernel that can be used for similar purpose. Also, `printf`
107  * is supported on device.
108  *
109  * @todo : we need to implement proper error handling mechanism to propogate errors
110  * from GPU to CPU. For example, error code can be returned like original
111  * neuron implementation. For now we use `assert(0==1)` pattern which is
112  * used for OpenACC/CUDA.
113  */
114 void CodegenAccVisitor::print_abort_routine() const {
115  printer->add_newline(2);
116  printer->add_line("static inline void coreneuron_abort() {");
117  printer->add_line(" printf(\"Error : Issue while running OpenACC kernel \\n\");");
118  printer->add_line(" assert(0==1);");
119  printer->add_line("}");
120 }

129 /**
130  * Each kernel like nrn_init, nrn_state and nrn_cur could be offloaded
131  * to accelerator. In this case, at very top level, we print pragma
132  * for data present. For example:
133  *
134  * \code{.cpp}
135  * void nrn_state(...) {
136  * #pragma acc data present (nt, ml...)
137  * {
138  *
139  * }
140  * }
141  * \endcode
142  */
143 void CodegenAccVisitor::print_kernel_data_present_annotation_block_begin() {
144  if (!info.artificial_cell) {
145  auto global_variable = "{}_global"_format(info.mod_suffix);
146  printer->add_line("#pragma acc data present(nt, ml, {})"_format(global_variable));
147  printer->add_line("{");
148  printer->increase_indent();
149  }
