275 std::string CodegenNeuronCppVisitor::get_variable_name(const std::string& name,
276  bool use_instance) const {
277  const std::string& varname = update_if_ion_variable_name(name);
278 
279  auto symbol_comparator = [&varname](const SymbolType& sym) {
280  return varname == sym->get_name();
281  };
282 
283  auto index_comparator = [&varname](const IndexVariableInfo& var) {
284  return varname == var.symbol->get_name();
285  };
286 
287  // float variable
288  auto f = std::find_if(codegen_float_variables.begin(),
289  codegen_float_variables.end(),
290  symbol_comparator);
291  if (f != codegen_float_variables.end()) {
292  return float_variable_name(*f, use_instance);
293  }
294 
295  // integer variable
296  auto i =
297  std::find_if(codegen_int_variables.begin(), codegen_int_variables.end(), index_comparator);
298  if (i != codegen_int_variables.end()) {
299  return int_variable_name(*i, varname, use_instance);
300  }
301 
302  // global variable
303  auto g = std::find_if(codegen_global_variables.begin(),
304  codegen_global_variables.end(),
305  symbol_comparator);
306  if (g != codegen_global_variables.end()) {
307  return global_variable_name(*g, use_instance);
308  }
309 
310  if (varname == naming::NTHREAD_DT_VARIABLE) {
311  return std::string("_nt->_") + naming::NTHREAD_DT_VARIABLE;
312  }
313 
314  // t in net_receive method is an argument to function and hence it should
315  // be used instead of nt->_t which is current time of thread
316  if (varname == naming::NTHREAD_T_VARIABLE && !printing_net_receive) {
317  return std::string("_nt->_") + naming::NTHREAD_T_VARIABLE;
318  }
319 
320  auto const iter =
321  std::find_if(info.neuron_global_variables.begin(),
322  info.neuron_global_variables.end(),
323  [&varname](auto const& entry) { return entry.first->get_name() == varname; });
324  if (iter != info.neuron_global_variables.end()) {
325  std::string ret;
326  if (use_instance) {
327  ret = "*(inst->";
328  }
329  ret.append(varname);
330  if (use_instance) {
331  ret.append(")");
332  }
333  return ret;
334  }
335 
336  // otherwise return original name
337  return varname;
338 }

432 void CodegenNeuronCppVisitor::print_mechanism_global_var_structure(bool print_initializers) {
433  const auto value_initialize = print_initializers ? "{}" : "";
434 
435  /// TODO: Print only global variables printed in NEURON
436  printer->add_newline(2);
437  printer->add_line("/* NEURON global variables */");
438  if (info.primes_size != 0) {
439  printer->fmt_line("static neuron::container::field_index _slist1[{0}], _dlist1[{0}];",
440  info.primes_size);
441  }
442  printer->add_line("static int mech_type;");
443 
444  printer->fmt_line("static int {} = {};",
445  naming::NRN_POINTERINDEX,
446  info.pointer_variables.size() > 0
447  ? static_cast<int>(info.pointer_variables.size())
448  : -1);
449 
450  // Start printing the CNRN-style global variables.
451  auto float_type = default_float_data_type();
452  printer->add_newline(2);
453  printer->add_line("/** all global variables */");
454  printer->fmt_push_block("struct {}", global_struct());
455 
456  if (!info.ions.empty()) {
457  // TODO implement these when needed.
458  }
459 
460  if (info.point_process) {
461  throw std::runtime_error("Not implemented, global point process.");
462  }
463 
464  if (!info.vectorize && !info.top_local_variables.empty()) {
465  throw std::runtime_error("Not implemented, global vectorize something.");
466  }
467 
468  if (!info.thread_variables.empty()) {
469  throw std::runtime_error("Not implemented, global thread variables.");
470  }
471 
472  if (info.table_count > 0) {
473  throw std::runtime_error("Not implemented, global table count.");
474  }
475 
476  for (const auto& var: info.state_vars) {
477  auto name = var->get_name() + "0";
478  auto symbol = program_symtab->lookup(name);
479  if (symbol == nullptr) {
480  printer->fmt_line("{} {}{};", float_type, name, value_initialize);
481  codegen_global_variables.push_back(make_symbol(name));
482  }
483  }
484 
485  for (const auto& var: info.global_variables) {
486  auto name = var->get_name();
487  auto length = var->get_length();
488  if (var->is_array()) {
489  printer->fmt_line("{} {}[{}] /* TODO init const-array */;", float_type, name, length);
490  } else {
491  double value{};
492  if (auto const& value_ptr = var->get_value()) {
493  value = *value_ptr;
494  }
495  printer->fmt_line("{} {}{};",
496  float_type,
497  name,
498  print_initializers ? fmt::format("{{{:g}}}", value) : std::string{});
499  }
500  codegen_global_variables.push_back(var);
501  }
502 
503 
504  for (const auto& f: info.function_tables) {
505  throw std::runtime_error("Not implemented, global function tables.");
506  }
507 
508  if (info.vectorize && info.thread_data_index) {
509  throw std::runtime_error("Not implemented, global vectorize something else.");
510  }
511 
512  printer->pop_block(";");
513 
514  print_global_var_struct_assertions();
515  print_global_var_struct_decl();
516 }

569 void CodegenNeuronCppVisitor::print_make_instance() const {
570  printer->add_newline(2);
571  printer->fmt_push_block("static {} make_instance_{}(_nrn_mechanism_cache_range& _ml)",
572  instance_struct(),
573  info.mod_suffix);
574  printer->fmt_push_block("return {}", instance_struct());
575 
576  std::vector<std::string> make_instance_args;
577 
578  const auto codegen_float_variables_size = codegen_float_variables.size();
579  for (int i = 0; i < codegen_float_variables_size; ++i) {
580  const auto& float_var = codegen_float_variables[i];
581  if (float_var->is_array()) {
582  make_instance_args.push_back(
583  fmt::format("_ml.template data_array_ptr<{}, {}>()", i, float_var->get_length()));
584  } else {
585  make_instance_args.push_back(fmt::format("_ml.template fpfield_ptr<{}>()", i));
586  }
587  }
588 
589  const auto codegen_int_variables_size = codegen_int_variables.size();
590  for (size_t i = 0; i < codegen_int_variables_size; ++i) {
591  const auto& var = codegen_int_variables[i];
592  auto name = var.symbol->get_name();
593  auto const variable = [&var, i]() -> std::string {
594  if (var.is_index || var.is_integer) {
595  return "";
596  } else if (var.is_vdata) {
597  return "";
598  } else {
599  return fmt::format("_ml.template dptr_field_ptr<{}>()", i);
600  }
601  }();
602  if (variable != "") {
603  make_instance_args.push_back(variable);
604  }
605  }
606 
607  printer->add_multi_line(fmt::format("{}", fmt::join(make_instance_args, ",\n")));
608 
609  printer->pop_block(";");
610  printer->pop_block();
611 }
