89 void Unit::add_fraction(const std::string& fraction_string) {
90  double nom, denom;
91  std::string nominator;
92  std::string denominator;
93  std::string::const_iterator it;
94  for (it = fraction_string.begin(); it != fraction_string.end() && *it != '|'; ++it) {
95  nominator.push_back(*it);
96  }
97  // pass "|" char
98  ++it;
99  for (auto itm = it; itm != fraction_string.end(); ++itm) {
100  denominator.push_back(*itm);
101  }
102  nom = parse_double(nominator);
103  denom = parse_double(denominator);
104  unit_factor = nom / denom;
105 }

107 double Unit::parse_double(std::string double_string) {
108  long double d_number;
109  double d_magnitude;
110  std::string s_number;
111  std::string s_magnitude;
112  std::string::const_iterator it;
113  // if double is positive sign = 1 else sign = -1
114  // to be able to be multiplied by the d_number
115  int sign = 1;
116  if (double_string.front() == '-') {
117  sign = -1;
118  double_string.erase(double_string.begin());
119  }
120  // if *it reached an exponent related char, then the whole double number is read
121  for (it = double_string.begin();
122  it != double_string.end() && *it != 'e' && *it != '+' && *it != '-';
123  ++it) {
124  s_number.push_back(*it);
125  }
126  // then read the magnitude of the double number
127  for (auto itm = it; itm != double_string.end(); ++itm) {
128  if (*itm != 'e') {
129  s_magnitude.push_back(*itm);
130  }
131  }
132  d_number = std::stold(s_number);
133  if (s_magnitude.empty()) {
134  d_magnitude = 0.0;
135  } else {
136  d_magnitude = std::stod(s_magnitude);
137  }
138  return static_cast<double>(d_number * powl(10.0, d_magnitude) * sign);
139 }

141 void UnitTable::calc_nominator_dims(const std::shared_ptr<Unit>& unit, std::string nominator_name) {
142  double nominator_prefix_factor = 1.0;
143  int nominator_power = 1;
144 
145  // if the nominator is DOUBLE, divide it from the unit factor
146  if (nominator_name.front() >= '1' && nominator_name.front() <= '9') {
147  unit->mul_factor(1 / std::stod(nominator_name));
148  return;
149  }
150 
151  std::string nom_name = nominator_name;
152  auto nominator = table.find(nominator_name);
153 
154  // if the nominator_name is not in the table, check if there are any prefixes or power
155  if (nominator == table.end()) {
156  int changed_nominator_name = 1;
157 
158  while (changed_nominator_name) {
159  changed_nominator_name = 0;
160  for (const auto& it: prefixes) {
161  auto res = std::mismatch(it.first.begin(), it.first.end(), nominator_name.begin());
162  if (res.first == it.first.end()) {
163  changed_nominator_name = 1;
164  nominator_prefix_factor *= it.second;
165  nominator_name.erase(nominator_name.begin(),
166  nominator_name.begin() + it.first.size());
167  }
168  }
169  }
170  // if the nominator is only a prefix, just multiply the prefix factor with the unit factor
171  if (nominator_name.empty()) {
172  for (const auto& it: prefixes) {
173  auto res = std::mismatch(it.first.begin(), it.first.end(), nom_name.begin());
174  if (res.first == it.first.end()) {
175  unit->mul_factor(it.second);
176  return;
177  }
178  }
179  }
180 
181  // if the nominator_back is a UNIT_POWER save the power to be able
182  // to calculate the correct factor and dimensions later
183  char nominator_back = nominator_name.back();
184  if (nominator_back >= '2' && nominator_back <= '9') {
185  nominator_power = nominator_back - '0';
186  nominator_name.pop_back();
187  }
188 
189  nominator = table.find(nominator_name);
190 
191  // delete "s" char for plural of the nominator_name
192  if (nominator == table.end()) {
193  if (nominator_name.back() == 's') {
194  nominator_name.pop_back();
195  }
196  nominator = table.find(nominator_name);
197  }
198  }
199 
200  // if the nominator is still not found in the table then output error
201  // else multiply its factor to the unit factor and calculate unit's dimensions
202  if (nominator == table.end()) {
203  std::stringstream ss;
204  ss << "Unit " << nominator_name << " not defined!" << std::endl;
205  throw std::runtime_error(ss.str());
206  } else {
207  for (int i = 0; i < nominator_power; i++) {
208  unit->mul_factor(nominator_prefix_factor * nominator->second->get_factor());
209  unit->add_nominator_dims(nominator->second->get_dimensions());
210  }
211  }
212 }

287 void UnitTable::insert(const std::shared_ptr<Unit>& unit) {
288  // check if the unit is a base unit and
289  // then add it to the base units vector
290  auto unit_nominator = unit->get_nominator_unit();
291  auto only_base_unit_nominator =
292  unit_nominator.size() == 1 && unit_nominator.front().size() == 3 &&
293  (unit_nominator.front().front() == '*' && unit_nominator.front().back() == '*');
294  if (only_base_unit_nominator) {
295  // base_units_names[i] = "*i-th base unit*" (ex. base_units_names[0] = "*a*")
296  base_units_names[unit_nominator.front()[1] - 'a'] = unit->get_name();
297  // if unit is found in table replace it
298  auto find_unit_name = table.find(unit->get_name());
299  if (find_unit_name == table.end()) {
300  table.insert({unit->get_name(), unit});
301  } else {
302  table.erase(unit->get_name());
303  table.insert({unit->get_name(), unit});
304  }
305  return;
306  }
307  // calculate unit's dimensions based on its nominator and denominator
308  for (const auto& it: unit->get_nominator_unit()) {
309  calc_nominator_dims(unit, it);
310  }
311  for (const auto& it: unit->get_denominator_unit()) {
312  calc_denominator_dims(unit, it);
313  }
314  // if unit is not in the table simply insert it, else replace with it with
315  // new definition
316  auto find_unit_name = table.find(unit->get_name());
317  if (find_unit_name == table.end()) {
318  table.insert({unit->get_name(), unit});
319  } else {
320  table.erase(unit->get_name());
321  table.insert({unit->get_name(), unit});
322  }
323 }
