99 /**
100  * Double numbers in the \c nrnunits.lib file are defined in the form [.0-9]+[-+][0-9]+
101  * To make sure they are parsed in the correct way and similarly to the NEURON parser
102  * we convert this string to a string compliant to scientific notation to be able to be parsed
103  * from std::stod(). To do that we have to add an `e` in case there is `-+` in the number
104  *string if it doesn't exist.
105  */
106 double Unit::parse_double(std::string double_string) {
107  auto pos = double_string.find_first_of("+-", 1); // start at 1 pos, because we don't care with
108  // the front sign
109  if (pos != std::string::npos && double_string.find_last_of("eE", pos) == std::string::npos) {
110  double_string.insert(pos, "e");
111  }
112  return std::stod(double_string);
113 }

115 // NOLINTNEXTLINE(readability-function-cognitive-complexity)
116 void UnitTable::calc_nominator_dims(const std::shared_ptr<Unit>& unit, std::string nominator_name) {
117  double nominator_prefix_factor = 1.0;
118  int nominator_power = 1;
119 
120  // if the nominator is DOUBLE, divide it from the unit factor
121  if (nominator_name.front() >= '1' && nominator_name.front() <= '9') {
122  unit->mul_factor(1 / std::stod(nominator_name));
123  return;
124  }
125 
126  std::string nom_name = nominator_name;
127  auto nominator = table.find(nominator_name);
128 
129  // if the nominator_name is not in the table, check if there are any prefixes or power
130  if (nominator == table.end()) {
131  int changed_nominator_name = 1;
132 
133  while (changed_nominator_name) {
134  changed_nominator_name = 0;
135  for (const auto& it: prefixes) {
136  auto res = std::mismatch(it.first.begin(), it.first.end(), nominator_name.begin());
137  if (res.first == it.first.end()) {
138  changed_nominator_name = 1;
139  nominator_prefix_factor *= it.second;
140  nominator_name.erase(nominator_name.begin(),
141  nominator_name.begin() +
142  static_cast<std::ptrdiff_t>(it.first.size()));
143  }
144  }
145  }
146  // if the nominator is only a prefix, just multiply the prefix factor with the unit factor
147  if (nominator_name.empty()) {
148  for (const auto& it: prefixes) {
149  auto res = std::mismatch(it.first.begin(), it.first.end(), nom_name.begin());
150  if (res.first == it.first.end()) {
151  unit->mul_factor(it.second);
152  return;
153  }
154  }
155  }
156 
157  // if the nominator_back is a UNIT_POWER save the power to be able
158  // to calculate the correct factor and dimensions later
159  char nominator_back = nominator_name.back();
160  if (nominator_back >= '2' && nominator_back <= '9') {
161  nominator_power = nominator_back - '0';
162  nominator_name.pop_back();
163  }
164 
165  nominator = table.find(nominator_name);
166 
167  // delete "s" char for plural of the nominator_name
168  if (nominator == table.end()) {
169  if (nominator_name.back() == 's') {
170  nominator_name.pop_back();
171  }
172  nominator = table.find(nominator_name);
173  }
174  }
175 
176  // if the nominator is still not found in the table then output error
177  // else multiply its factor to the unit factor and calculate unit's dimensions
178  if (nominator == table.end()) {
179  std::string ss = fmt::format("Unit {} not defined!", nominator_name);
180  throw std::runtime_error(ss);
181  } else {
182  for (int i = 0; i < nominator_power; i++) {
183  unit->mul_factor(nominator_prefix_factor * nominator->second->get_factor());
184  unit->add_nominator_dims(nominator->second->get_dimensions());
185  }
186  }
187 }

191  std::string denominator_name) {
192  double denominator_prefix_factor = 1.0;
193  int denominator_power = 1;
194 
195  // if the denominator is DOUBLE, divide it from the unit factor
196  if (denominator_name.front() >= '1' && denominator_name.front() <= '9') {
197  unit->mul_factor(std::stod(denominator_name));
198  return;
199  }
200 
201  std::string denom_name = denominator_name;
202  auto denominator = table.find(denominator_name);
203 
204  // if the denominator_name is not in the table, check if there are any prefixes or power
205  if (denominator == table.end()) {
206  bool changed_denominator_name = true;
207 
208  while (changed_denominator_name) {
209  changed_denominator_name = false;
210  for (const auto& it: prefixes) {
211  auto res =
212  std::mismatch(it.first.begin(), it.first.end(), denominator_name.begin());
213  if (res.first == it.first.end()) {
214  changed_denominator_name = true;
215  denominator_prefix_factor *= it.second;
216  denominator_name.erase(denominator_name.begin(),
217  denominator_name.begin() +
218  static_cast<std::ptrdiff_t>(it.first.size()));
219  }
220  }
221  }
222  // if the denominator is only a prefix, just multiply the prefix factor with the unit factor
223  if (denominator_name.empty()) {
224  for (const auto& it: prefixes) {
225  auto res = std::mismatch(it.first.begin(), it.first.end(), denom_name.begin());
226  if (res.first == it.first.end()) {
227  unit->mul_factor(it.second);
228  return;
229  }
230  }
231  }
232 
233  // if the denominator_back is a UNIT_POWER save the power to be able
234  // to calculate the correct factor and dimensions later
235  char denominator_back = denominator_name.back();
236  if (denominator_back >= '2' && denominator_back <= '9') {
237  denominator_power = denominator_back - '0';
238  denominator_name.pop_back();
239  }
240 
241  denominator = table.find(denominator_name);
242 
243  // delete "s" char for plural of the denominator_name
244  if (denominator == table.end()) {
245  if (denominator_name.back() == 's') {
246  denominator_name.pop_back();
247  }

251

254  throw std::runtime_error(ss);
255  } else {
256  for (int i = 0; i < denominator_power; i++) {
257  unit->mul_factor(1.0 / (denominator_prefix_factor * denominator->second->get_factor()));
258  unit->add_denominator_dims(denominator->second->get_dimensions());

263 void UnitTable::insert(const std::shared_ptr<Unit>& unit) {
264  // check if the unit is a base unit and
265  // then add it to the base units vector
266  auto unit_nominator = unit->get_nominator_unit();
267  auto only_base_unit_nominator =
268  unit_nominator.size() == 1 && unit_nominator.front().size() == 3 &&
269  (unit_nominator.front().front() == '*' && unit_nominator.front().back() == '*');
270  if (only_base_unit_nominator) {
271  // base_units_names[i] = "*i-th base unit*" (ex. base_units_names[0] = "*a*")
272  auto const index = unit_nominator.front()[1] - 'a';
273  assert(index >= 0 && index < base_units_names.size());
274  // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
275  base_units_names[index] = unit->get_name();
276  // if unit is found in table replace it
277  auto find_unit_name = table.find(unit->get_name());
278  if (find_unit_name == table.end()) {
279  table.insert({unit->get_name(), unit});
280  } else {
281  table.erase(unit->get_name());
282  table.insert({unit->get_name(), unit});
283  }
284  return;
285  }
286  // calculate unit's dimensions based on its nominator and denominator
287  for (const auto& it: unit->get_nominator_unit()) {
288  calc_nominator_dims(unit, it);
289  }
290  for (const auto& it: unit->get_denominator_unit()) {
291  calc_denominator_dims(unit, it);
292  }
293  // if unit is not in the table simply insert it, else replace with it with
294  // new definition
295  auto find_unit_name = table.find(unit->get_name());
296  if (find_unit_name == table.end()) {
297  table.insert({unit->get_name(), unit});
298  } else {
299  table.erase(unit->get_name());
300  table.insert({unit->get_name(), unit});
301  }
302 }
