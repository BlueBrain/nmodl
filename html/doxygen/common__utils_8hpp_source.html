133 /**
134  * \class SingletonRandomString
135  * \brief Singleton class for random strings
136  *
137  * Singleton class for random strings that are appended to the
138  * Eigen matrices names that are used in the solutions of
139  * nmodl::visitor::SympySolverVisitor and need to be the same to
140  * be printed by the nmodl::codegen::CodegenCVisitor
141  */
142 template <unsigned int SIZE = 4>
143 class SingletonRandomString {
144  public:
145  /// Delete public constructor needed for the singleton pattern to work
146  SingletonRandomString(SingletonRandomString const&) = delete;
147  /// Delete public "=" operator
148  SingletonRandomString& operator=(SingletonRandomString const&) = delete;
149 
150  /// Function to instantiate the SingletonRandomString class
151  static SingletonRandomString& instance() {
152  static SingletonRandomString srs;
153  return srs;
154  }
155 
156  /** Check if there is a random string assigned as suffix for the var_name variable
157  *
158  * @param var_name Variable name to check if exists in the std::map of random strings
159  * @return true if it exists, false if not
160  */
161  bool random_string_exists(const std::string& var_name) const {
162  return (random_strings.find(var_name) != random_strings.end());
163  }
164 
165  /** Get the random string of the var_name variable
166  *
167  * @param var_name Variable name for which to get the random string
168  * @return Random string assigned to var_name
169  */
170  const std::string& get_random_string(const std::string& var_name) const {
171  return random_strings.at(var_name);
172  }
173 
174  /** If var_name has already got a random string assigned remove it from map
175  * and assign a new one, else simply insert a new random string for var_name
176  *
177  * @param var_name Variable name for which to reset the random string
178  * @param use_numbers control whether random string can include numeric characters or not
179  * @return Random string assigned to var_name
180  */
181  const std::string& reset_random_string(const std::string& var_name,
182  UseNumbersInString use_numbers) {
183  const auto new_string = generate_random_string(SIZE, use_numbers);
184  auto status = random_strings.emplace(var_name, new_string);
185  if (!status.second) {
186  // insertion didn't take place since element was already there.
187  status.first->second = new_string;
188  }
189  return status.first->second;
190  }
191 
192  private:
193  /// \name Ctor & dtor
194  /// \{
195 
196  /// Constructor used by instance()
197  SingletonRandomString() = default;
198 
199  /// \}
200 
201  /// std::map that keeps the random strings assigned to variables as suffix
202  std::map<std::string, std::string> random_strings;
203 };
