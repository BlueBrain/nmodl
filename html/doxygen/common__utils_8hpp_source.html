107 /**
108  * \class SingletonRandomString
109  * \brief Singleton class for random strings
110  *
111  * Singleton class for random strings that are appended to the
112  * Eigen matrices names that are used in the solutions of
113  * nmodl::visitor::SympySolverVisitor and need to be the same to
114  * be printed by the nmodl::codegen::CodegenCVisitor
115  */
116 template <unsigned int SIZE = 4>
117 class SingletonRandomString {
118  public:
119  /// Delete public constructor needed for the singleton pattern to work
120  SingletonRandomString(SingletonRandomString const&) = delete;
121  /// Delete public "=" operator
122  SingletonRandomString& operator=(SingletonRandomString const&) = delete;
123 
124  /// Function to instantiate the SingletonRandomString class
125  static SingletonRandomString& instance() {
126  static SingletonRandomString srs;
127  return srs;
128  }
129 
130  /** Check if there is a random string assigned as suffix for the var_name variable
131  *
132  * @param var_name Variable name to check if exists in the std::map of random strings
133  * @return true if it exists, false if not
134  */
135  bool random_string_exists(const std::string& var_name) const {
136  return (random_strings.find(var_name) != random_strings.end());
137  }
138 
139  /** Get the random string of the var_name variable
140  *
141  * @param var_name Variable name for which to get the random string
142  * @return Random string assigned to var_name
143  */
144  const std::string& get_random_string(const std::string& var_name) const {
145  return random_strings.at(var_name);
146  }
147 
148  /** If var_name has already got a random string assigned remove it from map
149  * and assign a new one, else simply insert a new random string for var_name
150  *
151  * @param var_name Variable name for which to reset the random string
152  * @param use_numbers control whether random string can include numeric characters or not
153  * @return Random string assigned to var_name
154  */
155  const std::string& reset_random_string(const std::string& var_name,
156  UseNumbersInString use_numbers) {
157  const auto new_string = generate_random_string(SIZE, use_numbers);
158  auto status = random_strings.emplace(var_name, new_string);
159  if (!status.second) {
160  // insertion didn't take place since element was already there.
161  status.first->second = new_string;
162  }
163  return status.first->second;
164  }
165 
166  private:
167  /// \name Ctor & dtor
168  /// \{
169 
170  /// Constructor used by instance()
171  SingletonRandomString() = default;
172 
173  /// \}
174 
175  /// std::map that keeps the random strings assigned to variables as suffix
176  std::map<std::string, std::string> random_strings;
177 };
