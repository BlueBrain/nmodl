59 /**
60  * \class SingletonRandomString
61  * \brief Singleton class for random strings
62  *
63  * Singleton class for random strings that are appended to the
64  * Eigen matrices names that are used in the solutions of
65  * nmodl::visitor::SympySolverVisitor and need to be the same to
66  * be printed by the nmodl::codegen::CodegenCVisitor
67  */
68 template <unsigned int SIZE = 4>
69 class SingletonRandomString {
70  public:
71  /// Delete public constructor needed for the singleton pattern to work
72  SingletonRandomString(SingletonRandomString const&) = delete;
73  /// Delete public "=" operator
74  SingletonRandomString& operator=(SingletonRandomString const&) = delete;
75 
76  /// Function to instantiate the SingletonRandomString class
77  static SingletonRandomString& instance() {
78  static SingletonRandomString srs;
79  return srs;
80  }
81 
82  /** Check if there is a random string assigned as suffix for the var_name variable
83  *
84  * @param var_name Variable name to check if exists in the std::map of random strings
85  * @return true if it exists, false if not
86  */
87  bool random_string_exists(const std::string& var_name) const {
88  return (random_strings.find(var_name) != random_strings.end());
89  }
90 
91  /** Get the random string of the var_name variable
92  *
93  * @param var_name Variable name for which to get the random string
94  * @return Random string assigned to var_name
95  */
96  const std::string& get_random_string(const std::string& var_name) const {
97  return random_strings.at(var_name);
98  }
99 
100  /** If var_name has already got a random string assigned remove it from map
101  * and assign a new one, else simply insert a new random string for var_name
102  *
103  * @param var_name Variable name for which to reset the random string
104  * @param use_numbers control whether random string can include numeric characters or not
105  * @return Random string assigned to var_name
106  */
107  const std::string& reset_random_string(const std::string& var_name,
108  UseNumbersInString use_numbers) {
109  const auto new_string = generate_random_string(SIZE, use_numbers);
110  auto status = random_strings.emplace(var_name, new_string);
111  if (!status.second) {
112  // insertion didn't take place since element was already there.
113  status.first->second = new_string;
114  }
115  return status.first->second;
116  }
117 
118  private:
119  /// \name Ctor & dtor
120  /// \{
121 
122  /// Constructor used by instance()
123  SingletonRandomString() = default;
124 
125  /// \}
126 
127  /// std::map that keeps the random strings assigned to variables as suffix
128  std::map<std::string, std::string> random_strings;
129 };
