141 /**
142  * Check if current symbol table is in global scope
143  *
144  * We create program scope at the top level and it has global scope.
145  * It contains neuron variables like t, dt, celsius etc. Then each
146  * nmodl block defining variables are added under this program's symbol
147  * table. Hence there are multiple levels of global scopes. In this
148  * helper function we make sure current block as well as it's parent are
149  * under global scopes.
150  */
151 bool SymbolTable::under_global_scope() {
152  bool global_scope = global;
153  auto parent_table = parent;
154 
155  // traverse all parent blocks to make sure everyone is global
156  while (global_scope && (parent_table != nullptr)) {
157  parent_table = parent_table->parent;
158  if (parent_table != nullptr) {
159  global_scope = parent_table->global_scope();
160  }
161  }
162  return global_scope;
163 }

175 /// lookup in current sytab as well as all parent symbol tables
176 std::shared_ptr<Symbol> ModelSymbolTable::lookup(const std::string& name) {
177  if (current_symtab == nullptr) {
178  throw std::logic_error("Lookup with previous symtab = nullptr ");
179  }
180 
181  auto symbol = current_symtab->lookup(name);
182  if (symbol) {
183  return symbol;
184  }
185 
186  // check into all parent symbol tables
187  auto parent = current_symtab->get_parent_table();
188  while (parent != nullptr) {
189  symbol = parent->lookup(name);
190  if (symbol) {
191  break;
192  }
193  parent = parent->get_parent_table();
194  }
195  return symbol;
196 }

199 /**
200  * Emit warning message for shadowing definition or throw an exception
201  * if variable is being redefined in the same block.
202  */
203 void ModelSymbolTable::emit_message(const std::shared_ptr<Symbol>& first,
204  const std::shared_ptr<Symbol>& second,
205  bool redefinition) {
206  auto node = first->get_node();
207  std::string name = first->get_name();
208  auto properties = to_string(second->get_properties());
209  std::string type = "UNKNOWN";
210  if (node != nullptr) {
211  type = node->get_node_type_name();
212  }
213 
214  if (redefinition) {
215  std::string msg = "Re-declaration of " + name + " [" + type + "]";
216  msg += "<" + properties + "> in " + current_symtab->name();
217  msg += " with one in " + second->get_scope();
218  throw std::runtime_error(msg);
219  }
220  std::string msg = "SYMTAB :: " + name + " [" + type + "] in ";
221  msg += current_symtab->name() + " shadows <" + properties;
222  msg += "> definition in " + second->get_scope();
223  logger->warn(msg);
224 }

227 /**
228  * Insert symbol in the update mode i.e. symbol table is previously created
229  * and we are adding new symbol table.
230  *
231  * We set status as "created" because missing symbol means the variable is
232  * added by some intermediate passes.
233  *
234  * Consider inlining pass which creates a block like:
235  *
236  * DERIVATIVE states() {
237  * LOCAL xx
238  * {
239  * LOCAL xx
240  * }
241  * }
242  *
243  * Second xx is not added into symbol table (and hence not visible
244  * to other passes like local renamer). In this case we have to do
245  * local lookup in the current symtab and insert if doesn't exist.
246  */
247 std::shared_ptr<Symbol> ModelSymbolTable::update_mode_insert(
248  const std::shared_ptr<Symbol>& symbol) {
249  symbol->set_scope(current_symtab->name());
250  symbol->mark_created();
251 
252  std::string name = symbol->get_name();
253  auto search_symbol = lookup(name);
254 
255  /// if no symbol found then safe to insert
256  if (search_symbol == nullptr) {
257  current_symtab->insert(symbol);
258  return symbol;
259  }
260 
261  /// for global scope just combine properties
262  if (current_symtab->global_scope()) {
263  search_symbol->add_properties(symbol->get_properties());
264  return search_symbol;
265  }
266 
267  /// insert into current block's symbol table
268  if (current_symtab->lookup(name) == nullptr) {
269  current_symtab->insert(symbol);
270  }
271  return symbol;
272 }

274 void ModelSymbolTable::update_order(const std::shared_ptr<Symbol>& present_symbol,
275  const std::shared_ptr<Symbol>& new_symbol) {
276  auto symbol = (present_symbol != nullptr) ? present_symbol : new_symbol;
277 
278  bool is_parameter = new_symbol->has_any_property(NmodlType::param_assign);
279  bool is_assigned_definition = new_symbol->has_any_property(NmodlType::assigned_definition);
280 
281  if (symbol->get_definition_order() == -1) {
282  if (is_parameter || is_assigned_definition) {
283  symbol->set_definition_order(definition_order++);
284  }

288 std::shared_ptr<Symbol> ModelSymbolTable::insert(const std::shared_ptr<Symbol>& symbol) {
289  if (current_symtab == nullptr) {
290  throw std::logic_error("Can not insert symbol without entering scope");
291  }
292 
293  auto search_symbol = lookup(symbol->get_name());
294  update_order(search_symbol, symbol);
295 
296  /// handle update mode insertion
297  if (update_table) {
298  return update_mode_insert(symbol);
299  }
300 
301  symbol->set_scope(current_symtab->name());
302 
303  // if no symbol found then safe to insert
304  if (search_symbol == nullptr) {
305  current_symtab->insert(symbol);
306  return symbol;
307  }
308 
309  /**
310  * For global symbol tables, same variable can appear in multiple
311  * nmodl "global" blocks. It's an error if it appears multiple times
312  * in the same nmodl block. To check this we compare symbol properties
313  * which are bit flags. If they are not same that means, we have to
314  * add new properties to existing symbol. If the properties are same
315  * that means symbol are duplicate.
316  */
317  if (current_symtab->global_scope()) {
318  if (search_symbol->has_any_property(symbol->get_properties())) {
319  emit_message(symbol, search_symbol, true);
320  } else {
321  search_symbol->add_properties(symbol->get_properties());
322  }
323  return search_symbol;
324  }
325 
326  /**
327  * For non-global scopes, check if symbol that we found has same
328  * scope name as symbol table. This means variable is being re-declared
329  * within the same scope. Otherwise, there is variable with same name
330  * in parent scopes and it will shadow the definition. In this case just
331  * emit the warning and insert the symbol.
332  *
333  * Note:
334  * We suppress the warning for the voltage since in most cases it is extern
335  * as well as argument and it is fine like that.
336  */
337  if (search_symbol->get_scope() == current_symtab->name()) {
338  emit_message(symbol, search_symbol, true);
339  } else {
340  /**
341  * Suppress warning for voltage since it is often extern and argument.
342  */
343  if (symbol->get_name() != "v") {
344  emit_message(symbol, search_symbol, false);
345  }
346  current_symtab->insert(symbol);
347  }
348  return symbol;
349 }

357 std::string ModelSymbolTable::get_unique_name(const std::string& name, Ast* node, bool is_global) {
358  static int block_counter = 0;
359  std::string new_name(name);
360  if (is_global) {

370 /**
371  * Function callback at the entry of every block in nmodl file
372  * Every block starts a new scope and hence new symbol table is created.
373  * The same symbol table is returned so that visitor can store pointer to
374  * symbol table within a node.
375  */
376 SymbolTable* ModelSymbolTable::enter_scope(const std::string& name,
377  Ast* node,
378  bool global,
379  SymbolTable* node_symtab) {
380  if (node == nullptr) {
381  throw std::runtime_error("Can't enter with empty node");
382  }
383 
384  /**
385  * All global blocks in mod file have same global symbol table. If there
386  * is already symbol table setup in global scope, return the same.
387  */
388  if (symtab && global) {
389  return symtab.get();
390  }
391 
392  /// statement block within global scope is part of global block itself
393  if (symtab && node->is_statement_block() && current_symtab->under_global_scope()) {
394  return symtab.get();
395  }
396 
397  if (node_symtab == nullptr || !update_table) {
398  auto new_name = get_unique_name(name, node, global);
399  auto new_symtab = std::make_shared<SymbolTable>(new_name, node, global);
400  new_symtab->set_parent_table(current_symtab);
401  if (symtab == nullptr) {
402  symtab = new_symtab;
403  }
404  if (current_symtab != nullptr) {
405  current_symtab->insert_table(new_name, new_symtab);
406  }
407  node_symtab = new_symtab.get();
408  }
409  current_symtab = node_symtab;
410  return current_symtab;
411 }

454 void SymbolTable::Table::print(std::ostream& stream, std::string title, int indent) const {
455  using stringutils::text_alignment;
456  using utils::TableData;
457  if (!symbols.empty()) {
458  TableData table;
459  table.title = std::move(title);
460  table.headers = {
461  "NAME", "PROPERTIES", "STATUS", "LOCATION", "VALUE", "# READS", "# WRITES"};
462  table.alignments = {text_alignment::left,
463  text_alignment::left,
464  text_alignment::right,
465  text_alignment::right,
466  text_alignment::right};
467 
468  for (const auto& symbol: symbols) {
469  auto is_external = symbol->is_external_variable();
470  auto read_count = symbol->get_read_count();
471  auto write_count = symbol->get_write_count();
472 
473  // do not print external symbols which are not used in the current model
474  if (is_external && read_count == 0 && write_count == 0) {
475  continue;
476  }
477 
478  auto name = symbol->get_name();
479  if (symbol->is_array()) {
480  name += "[" + std::to_string(symbol->get_length()) + "]";
481  }
482  auto position = symbol->get_token().position();
483  auto properties = syminfo::to_string(symbol->get_properties());
484  auto status = syminfo::to_string(symbol->get_status());
485  auto reads = std::to_string(symbol->get_read_count());
486  std::string value;
487  auto sym_value = symbol->get_value();
488  if (sym_value) {
489  value = std::to_string(*sym_value);
490  }
491  auto writes = std::to_string(symbol->get_write_count());
492  table.rows.push_back({name, properties, status, position, value, reads, writes});
493  }
494  table.print(stream, indent);
495  }
496 }

509 void SymbolTable::print(std::ostream& ss, int level) const {
510  table.print(ss, title(), level);
511 
512  /// when current symbol table is empty, the children
513  /// can be printed from the same indentation level
514  /// (this is to avoid unnecessary empty indentations)
