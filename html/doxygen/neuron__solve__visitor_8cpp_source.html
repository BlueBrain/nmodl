44 void NeuronSolveVisitor::visit_binary_expression(ast::BinaryExpression& node) {
45  const auto& lhs = node.get_lhs();
46 
47  /// we have to only solve odes under derivative block where lhs is variable
48  if (!derivative_block || !differential_equation || !lhs->is_var_name()) {
49  return;
50  }
51 
52  const auto& solve_method = solve_blocks[derivative_block_name];
53  auto name = std::dynamic_pointer_cast<ast::VarName>(lhs)->get_name();
54 
55  if (name->is_prime_name()) {
56  auto equation = to_nmodl(node);
57  parser::DiffeqDriver diffeq_driver;
58 
59  if (solve_method == codegen::naming::CNEXP_METHOD) {
60  std::string solution;
61  /// check if ode can be solved with cnexp method
62  if (diffeq_driver.cnexp_possible(equation, solution)) {
63  auto statement = create_statement(solution);
64  auto expr_statement = std::dynamic_pointer_cast<ast::ExpressionStatement>(
65  statement);
66  const auto bin_expr = std::dynamic_pointer_cast<const ast::BinaryExpression>(
67  expr_statement->get_expression());
68  node.set_lhs(std::shared_ptr<ast::Expression>(bin_expr->get_lhs()->clone()));
69  node.set_rhs(std::shared_ptr<ast::Expression>(bin_expr->get_rhs()->clone()));
70  } else {
71  logger->warn("NeuronSolveVisitor :: cnexp solver not possible for {}",
72  to_nmodl(node));
73  }
74  } else if (solve_method == codegen::naming::EULER_METHOD) {
75  std::string solution = diffeq_driver.solve(equation, solve_method);
76  auto statement = create_statement(solution);
77  auto expr_statement = std::dynamic_pointer_cast<ast::ExpressionStatement>(statement);
78  const auto bin_expr = std::dynamic_pointer_cast<const ast::BinaryExpression>(
79  expr_statement->get_expression());
80  node.set_lhs(std::shared_ptr<ast::Expression>(bin_expr->get_lhs()->clone()));
81  node.set_rhs(std::shared_ptr<ast::Expression>(bin_expr->get_rhs()->clone()));
82  } else if (solve_method == codegen::naming::DERIVIMPLICIT_METHOD) {
83  auto varname = "D" + name->get_node_name();
84  node.set_lhs(std::make_shared<ast::Name>(new ast::String(varname)));
85  if (program_symtab->lookup(varname) == nullptr) {
86  auto symbol = std::make_shared<symtab::Symbol>(varname, ModToken());
87  symbol->set_original_name(name->get_node_name());
88  symbol->created_from_state();
89  program_symtab->insert(symbol);
90  }
91  } else {
92  logger->error("NeuronSolveVisitor :: solver method '{}' not supported", solve_method);
93  }
94  }
95 }
