24 /// Helper for creating C codegen visitor
25 std::shared_ptr<CodegenCVisitor> create_c_visitor(const std::shared_ptr<ast::Program>& ast,
26  const std::string& text,
27  std::stringstream& ss) {
28  /// construct symbol table
29  SymtabVisitor().visit_program(*ast);
30 
31  /// create C code generation visitor
32  auto cv = std::make_shared<CodegenCVisitor>("temp.mod", ss, "double", false);
33  cv->setup(*ast);
34  return cv;
35 }

226 SCENARIO("Check parameter constness with VERBATIM block",
227  "[codegen][verbatim_variable_constness]") {
228  GIVEN("A mod file containing parameter range variables that are updated in VERBATIM block") {
229  std::string const nmodl_text = R"(
230  NEURON {
231  SUFFIX IntervalFire
232  RANGE invl, burst_start
233  }
234  PARAMETER {
235  invl = 10 (ms) <1e-9,1e9>
236  burst_start = 0 (ms)
237  }
238  INITIAL {
239  LOCAL temp
240  : as invl is used in verbatim, it shouldn't be treated as const
241  VERBATIM
242  invl = 11
243  ENDVERBATIM
244  : burst_start time is read-only and hence can be const
245  temp = burst_start
246  }
247  )";
248 
249  THEN("Variable used in VERBATIM shouldn't be marked as const") {
250  std::stringstream ss;
251 
252  /// parse mod file & print mechanism structure
253  const auto& ast = NmodlDriver().parse_string(nmodl_text);
254  auto cvisitor = create_c_visitor(ast, nmodl_text, ss);
255  cvisitor->print_mechanism_range_var_structure();
256 
257  std::string expected_code = R"(
258  /** all mechanism instance variables */
259  struct IntervalFire_Instance {
260  double* __restrict__ invl;
261  const double* __restrict__ burst_start;
262  double* __restrict__ v_unused;
263  };
264  )";
265 
266  REQUIRE(reindent_text(ss.str()) == reindent_text(expected_code));
267  }
268  }
