32 /// Helper for creating C codegen visitor
33 std::shared_ptr<CodegenCVisitor> create_c_visitor(const std::shared_ptr<ast::Program>& ast,
34  const std::string& /* text */,
35  std::stringstream& ss) {
36  /// construct symbol table
37  SymtabVisitor().visit_program(*ast);
38 
39  /// run all necessary pass
40  InlineVisitor().visit_program(*ast);
41  NeuronSolveVisitor().visit_program(*ast);
42  SolveBlockVisitor().visit_program(*ast);
43 
44  /// create C code generation visitor
45  auto cv = std::make_shared<CodegenCVisitor>("temp.mod", ss, "double", false);
46  cv->setup(*ast);
47  return cv;
48 }

266 std::string get_instance_structure(std::string nmodl_text) {
267  // parse mod file & print mechanism structure
268  auto const ast = NmodlDriver{}.parse_string(nmodl_text);
269  // add implicit arguments
270  ImplicitArgumentVisitor{}.visit_program(*ast);
271  // update the symbol table for PerfVisitor
272  SymtabVisitor{}.visit_program(*ast);
273  // we need the read/write counts so the codegen knows whether or not
274  // global variables are used
275  PerfVisitor{}.visit_program(*ast);
276  // setup codegen
277  std::stringstream ss{};
278  CodegenCVisitor cv{"temp.mod", ss, "double", false};
279  cv.setup(*ast);
280  cv.print_mechanism_range_var_structure(true);
281  return ss.str();
282 }

285  "[codegen][verbatim_variable_constness]") {
286  GIVEN("A mod file containing parameter range variables that are updated in VERBATIM block") {
287  std::string const nmodl_text = R"(
288  NEURON {
289  SUFFIX IntervalFire

295  }
296  INITIAL {
297  LOCAL temp

417  REQUIRE_THAT(generated,
