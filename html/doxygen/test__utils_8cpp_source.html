35 /** Reindent nmodl text for text-to-text comparison
36  *
37  * Nmodl constructs defined in test database has extra leading whitespace.
38  * This is done for readability reason in nmodl_constructs.cpp. For example,
39  * we have following nmodl text with 8 leading whitespaces:
40 
41 
42  NEURON {
43  RANGE x
44  }
45 
46  * We convert above paragraph to:
47 
48 NEURON {
49  RANGE x
50 }
51 
52  * i.e. we get first non-empty line and count number of leading whitespaces (X).
53  * Then for every sub-sequent line, we remove first X characters (assuming those
54  * all are whitespaces). This is done because when ast is transformed back to
55  * nmodl, the nmodl output is without "extra" whitespaces in the provided input.
56  */
57 
58 std::string reindent_text(const std::string& text) {
59  std::string indented_text;
60  int num_whitespaces = 0;
61  bool flag = false;
62  std::string line;
63  std::stringstream stream(text);
64 
65  while (std::getline(stream, line)) {
66  if (!line.empty()) {
67  /// count whitespaces for first non-empty line only
68  if (!flag) {
69  flag = true;
70  num_whitespaces = count_leading_spaces(line);
71  }
72 
73  /// make sure we don't remove non whitespaces characters
74  if (!is_empty(line.substr(0, num_whitespaces))) {
75  throw std::runtime_error("Test nmodl input not correctly formatted");
76  }
77 
78  line.erase(0, num_whitespaces);
79  indented_text += line;
80  }
81  /// discard empty lines at very beginning
82  if (!stream.eof() && flag) {
83  indented_text += "\n";
84  }
85  }
86  return indented_text;
87 }
