26 /** Reindent nmodl text for text-to-text comparison
27  *
28  * Nmodl constructs defined in test database has extra leading whitespace.
29  * This is done for readability reason in nmodl_constructs.cpp. For example,
30  * we have following nmodl text with 8 leading whitespaces:
31 
32 
33  NEURON {
34  RANGE x
35  }
36 
37  * We convert above paragraph to:
38 
39 NEURON {
40  RANGE x
41 }
42 
43  * i.e. we get first non-empty line and count number of leading whitespaces (X).
44  * Then for every sub-sequent line, we remove first X characters (assuing those
45  * all are whitespaces). This is done because when ast is transformed back to
46  * nmodl, the nmodl output is without "extra" whitespaces in the provided input.
47  */
48 
49 std::string reindent_text(const std::string& text) {
50  std::string indented_text;
51  int num_whitespaces = 0;
52  bool flag = false;
53  std::string line;
54  std::stringstream stream(text);
55 
56  while (std::getline(stream, line)) {
57  if (!line.empty()) {
58  /// count whitespaces for first non-empty line only
59  if (!flag) {
60  flag = true;
61  num_whitespaces = count_leading_spaces(line);
62  }
63 
64  /// make sure we don't remove non whitespaces characters
65  if (!is_empty(line.substr(0, num_whitespaces))) {
66  throw std::runtime_error("Test nmodl input not correctly formatted");
67  }
68 
69  line.erase(0, num_whitespaces);
70  indented_text += line;
71  }
72  /// discard empty lines at very beginning
73  if (!stream.eof() && flag) {
74  indented_text += "\n";
75  }
76  }
77  return indented_text;
78 }
