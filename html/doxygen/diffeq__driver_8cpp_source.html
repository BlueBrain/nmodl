18 void DiffeqDriver::parse_equation(const std::string& equation,
19  std::string& state,
20  std::string& rhs,
21  int& order) {
22  auto parts = stringutils::split_string(equation, '=');
23  state = stringutils::trim(parts[0]);
24  rhs = stringutils::trim(parts[1]);
25 
26  /// expect prime on lhs, find order and remove quote
27  order = std::count(state.begin(), state.end(), '\'');
28  stringutils::remove_character(state, '\'');
29 
30  /// error if no prime in equation or not an assignment statement
31  if (order == 0 || state.empty()) {
32  throw std::runtime_error("Invalid equation, no prime on rhs? " + equation);
33  }
34 }
