30 /**
31  * Create solution expression node that will be used for solve block
32  * \param solve_block solve block used to describe node to solve and method
33  * \return solution expression that will be used to replace the solve block
34  *
35  * Depending on the solver used, solve block is converted to solve expression statement
36  * that will be used to replace solve block. Note that the blocks are clones instead of
37  * shared_ptr because DerivimplicitCallback is currently contain whole node
38  * instead of just pointer.
39  */
40 ast::SolutionExpression* SolveBlockVisitor::create_solution_expression(
41  ast::SolveBlock& solve_block) {
42  /// find out the block that is going to solved
43  const auto& block_name = solve_block.get_block_name()->get_node_name();
44  const auto& solve_node_symbol = symtab->lookup(block_name);
45  assert(solve_node_symbol != nullptr);
46  auto node_to_solve = solve_node_symbol->get_node();
47 
48  /// in case of derivimplicit method if neuron solver is used (i.e. not sympy) then
49  /// the solution is not in place but we have to create a callback to newton solver
50  const auto& method = solve_block.get_method();
51  std::string solve_method = method ? method->get_node_name() : "";
52  if (solve_method == codegen::naming::DERIVIMPLICIT_METHOD &&
53  !has_sympy_solution(*node_to_solve)) {
54  /// typically derivimplicit is used for derivative block only
55  assert(node_to_solve->get_node_type() == ast::AstNodeType::DERIVATIVE_BLOCK);
56  auto derivative_block = dynamic_cast<ast::DerivativeBlock*>(node_to_solve);
57  auto callback_expr = new ast::DerivimplicitCallback(derivative_block->clone());
58  return new ast::SolutionExpression(solve_block.clone(), callback_expr);
59  }
60 
61  auto block_to_solve = node_to_solve->get_statement_block();
62  return new ast::SolutionExpression(solve_block.clone(), block_to_solve->clone());
63 }

65 /**
66  * Replace solve blocks with solution expression
67  * @param node Ast node for SOLVE statement in the mod file
68  */
69 void SolveBlockVisitor::visit_expression_statement(ast::ExpressionStatement& node) {
70  node.visit_children(*this);
71  if (node.get_expression()->is_solve_block()) {
72  auto solve_block = dynamic_cast<ast::SolveBlock*>(node.get_expression().get());
73  auto sol_expr = create_solution_expression(*solve_block);
74  if (in_breakpoint_block) {
75  nrn_state_solve_statements.emplace_back(new ast::ExpressionStatement(sol_expr));
76  } else {
77  node.set_expression(std::shared_ptr<ast::SolutionExpression>(sol_expr));
78  }
79  }
80 }
