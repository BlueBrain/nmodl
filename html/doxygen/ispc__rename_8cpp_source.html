48 SCENARIO("Rename variables that ISPC parses as double constants", "[visitor][ispcrename]") {
49  GIVEN("mod file with variables with names that resemble for ISPC double constants") {
50  std::string input_nmodl = R"(
51  NEURON {
52  SUFFIX test_ispc_rename
53  RANGE d1, d2, var_d3, d4
54  }
55  ASSIGNED {
56  d1
57  d2
58  var_d3
59  d4
60  }
61  INITIAL {
62  d1 = 1
63  d2 = 2
64  var_d3 = 3
65  }
66  PROCEDURE func () {
67  VERBATIM d4 = 4; ENDVERBATIM
68  }
69  )";
70  auto ast = run_ispc_rename_visitor(input_nmodl);
71  auto verbatim_blocks = run_verbatim_visitor(ast);
72  auto symtab = ast->get_symbol_table();
73  THEN(
74  "Variables that match the constant double presentation in ISPC are renamed to "
75  "var_<original_name>") {
76  /// check if var_d1 and var_d2 exist and replaced d1 and d2
77  auto var_d1 = symtab->lookup("var_d1");
78  REQUIRE(var_d1 != nullptr);
79  auto d1 = symtab->lookup("d1");
80  REQUIRE(d1 == nullptr);
81  auto var_d2 = symtab->lookup("var_d2");
82  REQUIRE(var_d2 != nullptr);
83  auto d2 = symtab->lookup("d2");
84  REQUIRE(d2 == nullptr);
85  /// Check if VERBATIM block variable is renamed
86  REQUIRE(verbatim_blocks.size() == 1);
87  REQUIRE(verbatim_blocks.front() == " var_d4 = 4; ");
88  }
89  THEN("Variables that don't match the constant double presentation in ISPC stay the same") {
90  /// check if var_d3 exists
91  auto var_d3 = symtab->lookup("var_d3");
92  REQUIRE(var_d3 != nullptr);
93  }
94  }
