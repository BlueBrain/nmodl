31 void TableData::print(std::ostream& stream, int indent) const {
32  const int PADDING = 1;
33 
34  /// not necessary to print empty table
35  if (rows.empty() || headers.empty()) {
36  return;
37  }
38 
39  /// based on indentation level, spaces to prefix
40  auto gutter = std::string(indent * 4, ' ');
41 
42  auto ncolumns = headers.size();
43  std::vector<unsigned> col_width(ncolumns);
44 
45  /// alignment is optional, so fill remaining with right alignment
46  auto all_alignments = alignments;
47  all_alignments.reserve(ncolumns);
48  for (unsigned i = alignments.size(); i < ncolumns; i++) {
49  all_alignments.push_back(stringutils::text_alignment::center);
50  }
51 
52  /// calculate space required for each column
53  unsigned row_width = 0;
54  for (unsigned i = 0; i < headers.size(); i++) {
55  col_width[i] = headers[i].length() + PADDING;
56  row_width += col_width[i];
57  }
58 
59  /// if title is larger than headers then every column
60  /// width needs to be scaled
61  if (title.length() > row_width) {
62  int extra_size = title.length() - row_width;
63  int column_pad = extra_size / ncolumns;
64  if ((extra_size % ncolumns) != 0) {
65  column_pad++;
66  }
67  for (auto& column: col_width) {
68  column += column_pad;
69  }
70  }
71 
72  /// check length of columns in each row to find max length required
73  for (const auto& row: rows) {
74  for (unsigned i = 0; i < row.size(); i++) {
75  if (col_width[i] < (row[i].length()) + PADDING) {
76  col_width[i] = row[i].length() + PADDING;
77  }
78  }
79  }
80 
81  std::stringstream header;
82  header << "| ";
83  for (size_t i = 0; i < headers.size(); i++) {
84  auto text =
85  stringutils::align_text(headers[i], col_width[i], stringutils::text_alignment::center);
86  header << text << " | ";
87  }
88 
89  row_width = header.str().length();
90  auto separator_line = std::string(row_width - 1, '-');
91 
92  /// title row
93  if (!title.empty()) {
94  auto fmt_title =
95  stringutils::align_text(title, row_width - 3, stringutils::text_alignment::center);
96  stream << '\n' << gutter << separator_line;
97  stream << '\n' << gutter << '|' << fmt_title << '|';
98  }
99 
100  /// header row
101  stream << '\n' << gutter << separator_line;
102  stream << '\n' << gutter << header.str();
103  stream << '\n' << gutter << separator_line;
104 
105  /// data rows
106  for (const auto& row: rows) {
107  stream << '\n' << gutter << "| ";
108  for (unsigned i = 0; i < row.size(); i++) {
109  stream << stringutils::align_text(row[i], col_width[i], all_alignments[i]) << " | ";
110  }
111  }
112 
113  /// bottom separator line
114  stream << '\n' << gutter << separator_line << '\n';
115 }
