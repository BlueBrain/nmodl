87 static inline double egm1(double x, double px) {
88 
89  x -= px * 6.93145751953125d-1;
90  x -= px * 1.42860682030941723212d-6;
91 
92  const double xx = x * x;
93 
94  px = PX1exp;
95  px *= xx;
96  px += PX2exp;
97  px *= xx;
98  px += PX3exp;
99  px *= x;
100 
101  double qx = QX1exp;
102  qx *= xx;
103  qx += QX2exp;
104  qx *= xx;
105  qx += QX3exp;
106  qx *= xx;
107  qx += QX4exp;
108 
109  return 2.0d*px / (qx - px);
110 }

112 /// double precision exp function
113 static inline double vexp(double initial_x) {
114  double x = initial_x;
115  double px = dpfloor(LOG2E * x + 0.5d);
116  const int32 n = px;
117 
118  x = 1.0d + egm1(x, px);
119  x *= uint642dp((((unsigned int64) n) + 1023) << 52);
120

122  x = inf();
123  if (initial_x < -EXP_LIMIT)
124  x = 0.d;
125 
126  return x;
127 }

158 static inline float egm1(float x) {
159 
160  float z = x * PX1expf;
161  z += PX2expf;
162  z *= x;
163  z += PX3expf;
164  z *= x;
165  z += PX4expf;

167  z += PX5expf;
168  z *= x;
169  z += PX6expf;
170  z *= x*x;
171  z += x;
172 
173  return z;
