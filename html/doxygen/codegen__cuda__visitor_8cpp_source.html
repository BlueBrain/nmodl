27 /**
28  * As initial block is/can be executed on c/cpu backend, gpu/cuda
29  * backend can mark the parameter as constant even if they have
30  * write count > 0 (typically due to initial block).
31  */
32 bool CodegenCudaVisitor::is_constant_variable(const std::string& name) const {
33  auto symbol = program_symtab->lookup_in_scope(name);
34  bool is_constant = false;
35  if (symbol != nullptr) {
36  if (symbol->has_any_property(NmodlType::read_ion_var)) {
37  is_constant = true;
38  }
39  if (symbol->has_any_property(NmodlType::param_assign)) {
40  is_constant = true;
41  }
42  }
43  return is_constant;
44 }

61 void CodegenCudaVisitor::print_atomic_op(const std::string& lhs,
62  const std::string& op,
63  const std::string& rhs) const {
64  std::string function;
65  if (op == "+") {
66  function = "atomicAdd";
67  } else if (op == "-") {
68  function = "atomicSub";
69  } else {
70  throw std::runtime_error(fmt::format("CUDA backend error : {} not supported", op));
71  }
72  printer->add_line(fmt::format("{}(&{}, {});", function, lhs, rhs));
73 }

105 void CodegenCudaVisitor::print_fast_imem_calculation() {
106  if (!info.electrode_current) {
107  return;
108  }
109 
110  auto rhs_op = operator_for_rhs();
111  auto d_op = operator_for_d();
112  stringutils::remove_character(rhs_op, '=');
113  stringutils::remove_character(d_op, '=');
114  printer->start_block("if (nt->nrn_fast_imem)");
115  print_atomic_reduction_pragma();
116  print_atomic_op("nt->nrn_fast_imem->nrn_sav_rhs[node_id]", rhs_op, "rhs");
117  print_atomic_reduction_pragma();
118  print_atomic_op("nt->nrn_fast_imem->nrn_sav_d[node_id]", d_op, "g");
119  printer->end_block(1);
120 }

187 void CodegenCudaVisitor::print_wrapper_routine(std::string wrapper_function, BlockType type) {
188  static const auto args = "NrnThread* nt, Memb_list* ml, int type";
189  wrapper_function = method_name(wrapper_function);
190  auto compute_function = compute_method_name(type);
191 
192  printer->add_newline(2);
193  printer->start_block(fmt::format("void {}({})", wrapper_function, args));
194  printer->add_line("int nodecount = ml->nodecount;");
195  printer->add_line("int nthread = 256;");
196  printer->add_line("int nblock = (nodecount+nthread-1)/nthread;");
197  printer->add_line(fmt::format("{}<<<nblock, nthread>>>(nt, ml, type);", compute_function));
198  printer->add_line("cudaDeviceSynchronize();");
199  printer->end_block();
200  printer->add_newline();
201 }
