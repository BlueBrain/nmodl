23 /**
24  * As initial block is/can be executed on c/cpu backend, gpu/cuda
25  * backend can mark the parameter as constant even if they have
26  * write count > 0 (typically due to initial block).
27  */
28 bool CodegenCudaVisitor::is_constant_variable(const std::string& name) const {
29  auto symbol = program_symtab->lookup_in_scope(name);
30  bool is_constant = false;
31  if (symbol != nullptr) {
32  if (symbol->has_any_property(NmodlType::read_ion_var)) {
33  is_constant = true;
34  }
35  if (symbol->has_any_property(NmodlType::param_assign)) {
36  is_constant = true;
37  }
38  }
39  return is_constant;
40 }

57 void CodegenCudaVisitor::print_atomic_op(const std::string& lhs,
58  const std::string& op,
59  const std::string& rhs) const {
60  std::string function;

66  throw std::runtime_error("CUDA backend error : {} not supported"_format(op));
67  }
68  printer->add_line("{}(&{}, {});"_format(function, lhs, rhs));

183 void CodegenCudaVisitor::print_wrapper_routine(std::string wrapper_function, BlockType type) {
184  static const auto args = "NrnThread* nt, Memb_list* ml, int type";
185  wrapper_function = method_name(wrapper_function);
186  auto compute_function = compute_method_name(type);
187 
188  printer->add_newline(2);
189  printer->start_block("void {}({})"_format(wrapper_function, args));
190  printer->add_line("int nodecount = ml->nodecount;");
191  printer->add_line("int nthread = 256;");
192  printer->add_line("int nblock = (nodecount+nthread-1)/nthread;");
193  printer->add_line("{}<<<nblock, nthread>>>(nt, ml, type);"_format(compute_function));
194  printer->add_line("cudaDeviceSynchronize();");
195  printer->end_block();
196  printer->add_newline();
197 }
