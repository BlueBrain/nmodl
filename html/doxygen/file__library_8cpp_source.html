56 std::string FileLibrary::find_file(const fs::path& file) {
57  if (file.is_absolute() && fs::exists(file)) {
58  return "";
59  }
60  for (auto paths_it = paths_.rbegin(); paths_it != paths_.rend(); ++paths_it) {
61  auto file_abs = *paths_it / file;
62  if (fs::exists(file_abs)) {
63  return paths_it->string();
64  }
65  }
66  return "";
67 }
