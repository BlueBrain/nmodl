55 std::string FileLibrary::find_file(const std::string& file) {
56  if (utils::file_is_abs(file)) {
57  if (utils::file_exists(file)) {
58  return "";
59  }
60  }
61  for (auto paths_it = paths_.rbegin(); paths_it != paths_.rend(); ++paths_it) {
62  auto file_abs = *paths_it + utils::pathsep + file;
63  if (utils::file_exists(file_abs)) {
64  return *paths_it;
65  }
66  }
67  return "";
68 }
