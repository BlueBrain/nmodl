23 bool InlineVisitor::can_inline_block(const StatementBlock& block) {
24  bool to_inline = true;
25  const auto& statements = block.get_statements();
26  for (const auto& statement: statements) {
27  /// inlining is disabled if function/procedure contains table or lag statement
28  if (statement->is_table_statement() || statement->is_lag_statement()) {
29  to_inline = false;
30  break;
31  }
32  // verbatim blocks with return statement are not safe to inline
33  // especially for net_receive block
34  if (statement->is_verbatim()) {
35  const auto node = dynamic_cast<const Verbatim*>(statement.get());
36  assert(node);
37  auto text = node->get_statement()->eval();
38  parser::CDriver driver;
39  driver.scan_string(text);
40  if (driver.has_token("return")) {
41  to_inline = false;
42  break;
43  }
44  }
45  }
46  return to_inline;
47 }

57 /** We can replace statement if the entire statement itself is a function call.
58  * In this case we check if:
59  * - given statement is expression statement
60  * - if expression is wrapped expression
61  * - if wrapped expression is a function call
62  *
63  * \todo Add method to ast itself to simplify this implementation
64  */
65 bool InlineVisitor::can_replace_statement(const std::shared_ptr<Statement>& statement) {
66  if (!statement->is_expression_statement()) {
67  return false;
68  }
69 
70  bool to_replace = false;
71  auto es = dynamic_cast<ExpressionStatement*>(statement.get());
72  assert(es);
73  auto e = es->get_expression();
74  if (e->is_wrapped_expression()) {
75  auto wrapped_expression = dynamic_cast<WrappedExpression*>(e.get());
76  assert(wrapped_expression);
77  if (wrapped_expression->get_expression()->is_function_call()) {
78  // if caller is external function (i.e. neuron function) don't replace it
79  const auto& function_call = std::static_pointer_cast<FunctionCall>(
80  wrapped_expression->get_expression());
81  const auto& function_name = function_call->get_node_name();
82  const auto& symbol = program_symtab->lookup_in_scope(function_name);
83  to_replace = !symbol->is_external_variable();
84  }
85  }
86  return to_replace;
87 }

89 void InlineVisitor::inline_arguments(StatementBlock& inlined_block,
90  const ArgumentVector& callee_parameters,
91  const ExpressionVector& caller_expressions) {
92  /// nothing to inline if no arguments for function call
93  if (caller_expressions.empty()) {
94  return;
95  }
96 
97  size_t counter = 0;
98  const auto& statements = inlined_block.get_statements();
99 
100  for (const auto& argument: callee_parameters) {
101  auto name = argument->get_name()->clone();
102  auto old_name = name->get_node_name();
103  auto new_name = get_new_name(old_name, "in", inlined_variables);
104  name->set_name(new_name);
105 
106  /// for argument add new variable to local statement
107  add_local_variable(inlined_block, name);
108 
109  /// variables in cloned block needs to be renamed
110  RenameVisitor visitor(old_name, new_name);
111  inlined_block.visit_children(visitor);
112 
113  auto lhs = new VarName(name->clone(), nullptr, nullptr);
114  auto rhs = caller_expressions.at(counter)->clone();
115 
116  /// create assignment statement and insert after the local variables
117  auto expression = new BinaryExpression(lhs, BinaryOperator(ast::BOP_ASSIGN), rhs);
118  auto statement = std::make_shared<ExpressionStatement>(expression);
119  inlined_block.insert_statement(statements.begin() +
120  static_cast<std::ptrdiff_t>(counter + 1ul),
121  statement);
122  counter++;
123  }
124 }

127 bool InlineVisitor::inline_function_call(ast::Block& callee,
128  ast::FunctionCall& node,
129  ast::StatementBlock& caller) {
130  const auto& function_name = callee.get_node_name();
131 
132  /// do nothing if we can't inline given procedure/function
133  if (!can_inline_block(*callee.get_statement_block())) {
134  logger->warn("Can not inline function call to {}", function_name);
135  return false;
136  }
137 
138  /// make sure to rename conflicting local variable in caller block
139  /// because in case of procedure inlining they can conflict with
140  /// global variables used in procedure being inlined
141  LocalVarRenameVisitor v;
142  v.visit_statement_block(caller);
143 
144  const auto& caller_arguments = node.get_arguments();
145  std::string new_varname = get_new_name(function_name, "in", inlined_variables);
146 
147  /// check if caller statement could be replaced
148  bool to_replace = can_replace_statement(caller_statement);
149 
150  /// need to add local variable for function calls or for procedure call if it is part of
151  /// expression (standalone procedure calls don't need return statement)
152  if (callee.is_function_block() || !to_replace) {
153  /// create new variable which will be used for returning value from inlined block
154  auto name = new ast::Name(new ast::String(new_varname));
155  ModToken tok;
156  name->set_token(tok);
157 
158  auto local_list_statement = get_local_list_statement(caller);
159  /// each block should already have local statement
160  if (local_list_statement == nullptr) {
161  throw std::logic_error("got local statement as nullptr");
162  }
163  local_list_statement->emplace_back_local_var(std::make_shared<ast::LocalVar>(name));
164  }
165 
166  /// get a copy of function/procedure body
167  auto inlined_block = std::unique_ptr<ast::StatementBlock>(
168  callee.get_statement_block()->clone());
169 
170  /// function definition has function name as return value. we have to rename
171  /// it with new variable name
172  RenameVisitor visitor(function_name, new_varname);
173  inlined_block->visit_children(visitor);
174 
175  /// \todo Have to re-run symtab visitor pass to update symbol table
176  inlined_block->set_symbol_table(nullptr);
177 
178  /// each argument is added as new assignment statement
179  inline_arguments(*inlined_block, callee.get_parameters(), caller_arguments);
180 
181  /// to return value from procedure we have to add new variable
182  if (callee.is_procedure_block() && !to_replace) {
183  add_return_variable(*inlined_block, new_varname);
184  }
185 
186  auto statement = new ast::ExpressionStatement(std::move(inlined_block));
187 
188  if (to_replace) {
189  replaced_statements[caller_statement] = statement;
190  } else {
191  inlined_statements[caller_statement].push_back(
192  std::shared_ptr<ast::ExpressionStatement>(statement));
193  }
194 
195  /// variable name which will replace the function call that we just inlined
196  replaced_fun_calls[&node] = new_varname;
197  return true;
198 }

201 void InlineVisitor::visit_function_call(FunctionCall& node) {
202  /// argument can be function call itself
203  node.visit_children(*this);
204 
205  const auto& function_name = node.get_name()->get_node_name();
206  auto symbol = program_symtab->lookup_in_scope(function_name);
207 
208  /// nothing to do if called function is not defined or it's external
209  if (symbol == nullptr || symbol->is_external_variable()) {
210  return;
211  }
212 
213  auto function_definition = symbol->get_node();
214  if (function_definition == nullptr) {
215  throw std::runtime_error("symbol table doesn't have ast node for " + function_name);
216  }
217 
218  /// first inline called function
219  function_definition->visit_children(*this);
220 
221  bool inlined = false;
222 
223  if (function_definition->is_procedure_block()) {
224  auto proc = dynamic_cast<ProcedureBlock*>(function_definition);
225  assert(proc);
226  inlined = inline_function_call(*proc, node, *caller_block);
227  } else if (function_definition->is_function_block()) {
228  auto func = dynamic_cast<FunctionBlock*>(function_definition);
229  assert(func);
230  inlined = inline_function_call(*func, node, *caller_block);
231  }
232 
233  if (inlined) {
234  symbol->mark_inlined();
235  }
236 }

238 void InlineVisitor::visit_statement_block(StatementBlock& node) {
239  /** While inlining we have to add new statements before call site.
240  * In order to return result we also have to add new local variable
241  * to the caller block. Hence we have to keep track of caller block,
242  * caller block's symbol table and caller statement.
243  */
244  caller_block = &node;
245  statementblock_stack.push(&node);
246 
247  /** Add empty local statement at the begining of block if already doesn't exist.
248  * Why? When we iterate over statements and inline function call, we have to add
249  * local variable to return the result. As we can't modify vector while iterating,
250  * we pre-add local statement without any local variables. If inlining pass doesn't
251  * add any variable then this statement will be removed.
252  */
253  add_local_statement(node);
254 
255  const auto& statements = node.get_statements();
256 
257  for (const auto& statement: statements) {
258  caller_statement = statement;
259  statement_stack.push(statement);
260  caller_statement->visit_children(*this);
261  statement_stack.pop();
262  }
263 
264  /// each block should already have local statement
265  auto local_list_statement = get_local_list_statement(node);
266  if (local_list_statement->get_variables().empty()) {
267  node.erase_statement(statements.begin());
268  }
269 
270  /// check if any statement is candidate for replacement due to inlining
271  /// this is typicall case of procedure calls
272  for (auto it = statements.begin(); it < statements.end(); ++it) {
273  const auto& statement = *it;
274  if (replaced_statements.find(statement) != replaced_statements.end()) {
275  node.reset_statement(it, replaced_statements[statement]);
276  }
277  }
278 
279  /// all statements from inlining needs to be added before the caller statement
280  for (auto& element: inlined_statements) {
281  auto it = std::find(statements.begin(), statements.end(), element.first);
282  if (it != statements.end()) {
283  node.insert_statement(it, element.second, element.second.begin(), element.second.end());
284  element.second.clear();
285  }
286  }
287 
288  /** Restore the caller context : consider call chain A() -> B() -> none.
289  * When we finishes processing B's statements, even we pop the stack,
290  * caller_* variables still point to B's context. Hence first we have
291  * to pop elements and then use top() to get context of A(). We have to
292  * check for non-empty() stack because if there is only A() -> none then
293  * stack is already empty.
294  */
295  statementblock_stack.pop();
296 
297  if (!statement_stack.empty()) {
298  caller_statement = statement_stack.top();
299  }
300  if (!statementblock_stack.empty()) {
301  caller_block = statementblock_stack.top();
302  }
303 }

305 /** Visit all wrapped expressions which can contain function calls.
306  * If a function call is replaced then the wrapped expression is
307  * also replaced with new variable node from the inlining result.
308  */
309 void InlineVisitor::visit_wrapped_expression(WrappedExpression& node) {
310  node.visit_children(*this);
311  const auto& e = node.get_expression();
312  if (e->is_function_call()) {
313  auto expression = dynamic_cast<FunctionCall*>(e.get());
314  if (replaced_fun_calls.find(expression) != replaced_fun_calls.end()) {
315  auto var = replaced_fun_calls[expression];
316  node.set_expression(std::make_shared<Name>(new String(var)));
317  }
318  }
