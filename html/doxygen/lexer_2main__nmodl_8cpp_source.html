39 void tokenize(const std::string& mod_text) {
40  std::istringstream in(mod_text);
41 
42  /// lexer instance use driver object for error reporting
43  NmodlDriver driver;
44 
45  /// lexer instance with stream to read-in tokens
46  NmodlLexer scanner(driver, &in);
47 
48  auto get_token_type = [](TokenType token) {
49  return parser::NmodlParser::by_type(token).type_get();
50  };
51 
52  /// parse nmodl text and print token until EOF
53  while (true) {
54  SymbolType sym = scanner.next_token();
55  auto token_type = sym.type_get();
56 
57  if (token_type == get_token_type(Token::END)) {
58  break;
59  }
60 
61  /**
62  * Lexer returns different ast types base on token type. We
63  * retrieve token object from each instance and print it.
64  * Note that value is of ast type i.e. ast::Name* etc.
65  */
66  /// token with name ast class
67  if (token_type == get_token_type(Token::NAME) ||
68  token_type == get_token_type(Token::METHOD) ||
69  token_type == get_token_type(Token::SUFFIX) ||
70  token_type == get_token_type(Token::VALENCE) ||
71  token_type == get_token_type(Token::DEL) || token_type == get_token_type(Token::DEL2)) {
72  auto value = sym.value.as<ast::Name>();
73  std::cout << *(value.get_token()) << std::endl;
74  }
75  /// token with prime ast class
76  else if (token_type == get_token_type(Token::PRIME)) {
77  auto value = sym.value.as<ast::PrimeName>();
78  std::cout << *(value.get_token()) << std::endl;
79  }
80  /// token with integer ast class
81  else if (token_type == get_token_type(Token::INTEGER)) {
82  auto value = sym.value.as<ast::Integer>();
83  std::cout << *(value.get_token()) << std::endl;
84  }
85  /// token with double/float ast class
86  else if (token_type == get_token_type(Token::REAL)) {
87  auto value = sym.value.as<ast::Double>();
88  std::cout << *(value.get_token()) << std::endl;
89  }
90  /// token with string ast class
91  else if (token_type == get_token_type(Token::STRING)) {
92  auto value = sym.value.as<ast::String>();
93  std::cout << *(value.get_token()) << std::endl;
94  }
95  /// token with string data type
96  else if (token_type == get_token_type(Token::VERBATIM) ||
97  token_type == get_token_type(Token::BLOCK_COMMENT) ||
98  token_type == get_token_type(Token::LINE_PART)) {
99  auto str = sym.value.as<std::string>();
100  std::cout << str << std::endl;
101  }
102  /// all remaining tokens has ModToken* as a vaue
103  else {
104  auto token = sym.value.as<ModToken>();
105  std::cout << token << std::endl;
106  }
107  }
108 }

118  app.add_option("--text", mod_texts, "One or more NMODL constructs as text");
119 
120  CLI11_PARSE(app, argc, argv);
121 
122  for (const auto& file: mod_files) {

124  std::ifstream f(file);

126  tokenize(mod);
127  }
128 
129  for (const auto& text: mod_texts) {
