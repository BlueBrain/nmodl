31 std::string run_units_visitor(const std::string& text) {
32  NmodlDriver driver;
33  driver.parse_string(text);
34  const auto& ast = driver.get_ast();
35 
36  // Parse nrnunits.lib file and the UNITS block of the mod file
37  const std::string units_lib_path(NrnUnitsLib::get_path());
38  UnitsVisitor units_visitor = UnitsVisitor(units_lib_path);
39 
40  units_visitor.visit_program(*ast);
41 
42  // Keep the UnitTable created from parsing unit file and UNITS
43  // block of the mod file
44  parser::UnitDriver units_driver = units_visitor.get_unit_driver();
45  std::shared_ptr<units::UnitTable> unit_table = units_driver.table;
46 
47  std::stringstream ss;
48 
49  // Visit AST to find all the ast::UnitDef nodes to print their
50  // unit names, factors and dimensions as they are calculated in
51  // the units::UnitTable
52  const auto& unit_defs = collect_nodes(*ast, {ast::AstNodeType::UNIT_DEF});
53 
54  for (const auto& unit_def: unit_defs) {
55  auto unit_name = unit_def->get_node_name();
56  unit_name.erase(remove_if(unit_name.begin(), unit_name.end(), isspace), unit_name.end());
57  auto unit = units_driver.table->get_unit(unit_name);
58  ss << std::fixed << std::setprecision(8) << unit->get_name() << ' ' << unit->get_factor()
59  << ':';
60  // Dimensions of the unit are printed to check that the units are successfully
61  // parsed to the units::UnitTable
62  int dimension_id = 0;
63  auto constant = true;
64  for (const auto& dimension: unit->get_dimensions()) {
65  if (dimension != 0) {
66  constant = false;
67  ss << ' ' << units_driver.table->get_base_unit_name(dimension_id) << dimension;
68  }
69  dimension_id++;
70  }
71  if (constant) {
72  ss << " constant";
73  }
74  ss << '\n';
75  }
76 
77  // Visit AST to find all the ast::FactorDef nodes to print their
78  // unit names, factors and dimensions as they are calculated to
79  // be printed to the produced .cpp file
80  const auto& factor_defs = collect_nodes(*ast, {ast::AstNodeType::FACTOR_DEF});
81  for (const auto& factor_def: factor_defs) {
82  auto unit = units_driver.table->get_unit(factor_def->get_node_name());
83  ss << std::fixed << std::setprecision(8) << unit->get_name() << ' ';
84  auto factor_def_class = std::dynamic_pointer_cast<const nmodl::ast::FactorDef>(factor_def);
85  ss << factor_def_class->get_value()->eval() << ':';
86  // Dimensions of the unit are printed to check that the units are successfully
87  // parsed to the units::UnitTable
88  int dimension_id = 0;
89  auto constant = true;
90  for (const auto& dimension: unit->get_dimensions()) {
91  if (dimension != 0) {
92  constant = false;
93  ss << ' ' << units_driver.table->get_base_unit_name(dimension_id);
94  ss << dimension;
95  }
96  dimension_id++;
97  }
98  if (constant) {
99  ss << " constant";
100  }
101  ss << '\n';
102  }
103 
104  // check that, after visitor rearrangement, parents are still up-to-date
105  CheckParentVisitor().check_ast(*ast);
106 
107  return ss.str();
108 }
