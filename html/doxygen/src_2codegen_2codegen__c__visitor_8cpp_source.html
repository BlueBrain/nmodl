329 /**
330  * \details Certain statements like unit, comment, solve can/need to be skipped
331  * during code generation. Note that solve block is wrapped in expression
332  * statement and hence we have to check inner expression. It's also true
333  * for the initial block defined inside net receive block.
334  */
335 bool CodegenCVisitor::statement_to_skip(const Statement& node) const {
336  // clang-format off
337  if (node.is_unit_state()
338  || node.is_line_comment()
339  || node.is_block_comment()
340  || node.is_solve_block()
341  || node.is_conductance_hint()
342  || node.is_table_statement()) {
343  return true;
344  }
345  // clang-format on
346  if (node.is_expression_statement()) {
347  auto expression = dynamic_cast<const ExpressionStatement*>(&node)->get_expression();
348  if (expression->is_solve_block()) {
349  return true;
350  }
351  if (expression->is_initial_block()) {
352  return true;
353  }
354  }
355  return false;
356 }

466 /**
467  * \details Statements like if, else etc. don't need semicolon at the end.
468  * (Note that it's valid to have "extraneous" semicolon). Also, statement
469  * block can appear as statement using expression statement which need to
470  * be inspected.
471  */
472 bool CodegenCVisitor::need_semicolon(Statement* node) const {
473  // clang-format off
474  if (node->is_if_statement()
475  || node->is_else_if_statement()
476  || node->is_else_statement()
477  || node->is_from_statement()
478  || node->is_verbatim()
479  || node->is_for_all_statement()
480  || node->is_from_statement()
481  || node->is_conductance_hint()
482  || node->is_while_statement()) {
483  return false;
484  }
485  if (node->is_expression_statement()) {
486  auto expression = dynamic_cast<ExpressionStatement*>(node)->get_expression();
487  if (expression->is_statement_block()
488  || expression->is_eigen_newton_solver_block()
489  || expression->is_eigen_linear_solver_block()
490  || expression->is_solution_expression()
491  || expression->is_for_netcon()) {
492  return false;
493  }
494  }
495  // clang-format on
496  return true;
497 }

508 /**
509  * \details Current variable used in breakpoint block could be local variable.
510  * In this case, neuron has already renamed the variable name by prepending
511  * "_l". In our implementation, the variable could have been renamed by
512  * one of the pass. And hence, we search all local variables and check if
513  * the variable is renamed. Note that we have to look into the symbol table
514  * of statement block and not breakpoint.
515  */
516 std::string CodegenCVisitor::breakpoint_current(std::string current) const {
517  auto breakpoint = info.breakpoint_node;
518  if (breakpoint == nullptr) {
519  return current;
520  }
521  auto symtab = breakpoint->get_statement_block()->get_symbol_table();
522  auto variables = symtab->get_variables_with_properties(NmodlType::local_var);
523  for (const auto& var: variables) {
524  auto renamed_name = var->get_name();
525  auto original_name = var->get_original_name();
526  if (current == original_name) {
527  current = renamed_name;
528  break;
529  }
530  }
531  return current;
532 }

583 /**
584  * \details Depending upon the block type, we have to print read/write ion variables
585  * during code generation. Depending on block/procedure being printed, this
586  * method return statements as vector. As different code backends could have
587  * different variable names, we rely on backend-specific read_ion_variable_name
588  * and write_ion_variable_name method which will be overloaded.
589  *
590  * \todo After looking into mod2c and neuron implementation, it seems like
591  * Ode block type is not used (?). Need to look into implementation details.
592  */
593 std::vector<std::string> CodegenCVisitor::ion_read_statements(BlockType type) {
594  if (optimize_ion_variable_copies()) {
595  return ion_read_statements_optimized(type);
596  }
597  std::vector<std::string> statements;
598  for (const auto& ion: info.ions) {
599  auto name = ion.name;
600  for (const auto& var: ion.reads) {
601  if (type == BlockType::Ode && ion.is_ionic_conc(var) && state_variable(var)) {
602  continue;
603  }
604  auto variable_names = read_ion_variable_name(var);
605  auto first = get_variable_name(variable_names.first);
606  auto second = get_variable_name(variable_names.second);
607  statements.push_back("{} = {};"_format(first, second));
608  }
609  for (const auto& var: ion.writes) {
610  if (type == BlockType::Ode && ion.is_ionic_conc(var) && state_variable(var)) {
611  continue;
612  }
613  if (ion.is_ionic_conc(var)) {
614  auto variables = read_ion_variable_name(var);
615  auto first = get_variable_name(variables.first);
616  auto second = get_variable_name(variables.second);
617  statements.push_back("{} = {};"_format(first, second));
618  }
619  }
620  }
621  return statements;
622 }

644 std::vector<ShadowUseStatement> CodegenCVisitor::ion_write_statements(BlockType type) {
645  std::vector<ShadowUseStatement> statements;
646  for (const auto& ion: info.ions) {
647  std::string concentration;
648  auto name = ion.name;
649  for (const auto& var: ion.writes) {
650  auto variable_names = write_ion_variable_name(var);
651  if (ion.is_ionic_current(var)) {
652  if (type == BlockType::Equation) {
653  auto current = breakpoint_current(var);
654  auto lhs = variable_names.first;
655  auto op = "+=";
656  auto rhs = get_variable_name(current);
657  if (info.point_process) {
658  auto area = get_variable_name(naming::NODE_AREA_VARIABLE);
659  rhs += "*(1.e2/{})"_format(area);
660  }
661  statements.push_back(ShadowUseStatement{lhs, op, rhs});
662  }
663  } else {
664  if (!ion.is_rev_potential(var)) {
665  concentration = var;
666  }
667  auto lhs = variable_names.first;
668  auto op = "=";
669  auto rhs = get_variable_name(variable_names.second);
670  statements.push_back(ShadowUseStatement{lhs, op, rhs});
671  }
672  }
673 
674  if (type == BlockType::Initial && !concentration.empty()) {
675  int index = 0;
676  if (ion.is_intra_cell_conc(concentration)) {
677  index = 1;
678  } else if (ion.is_extra_cell_conc(concentration)) {
679  index = 2;
680  } else {
681  /// \todo Unhandled case in neuron implementation
682  throw std::logic_error("codegen error for {} ion"_format(ion.name));
683  }
684  auto ion_type_name = "{}_type"_format(ion.name);
685  auto lhs = "int {}"_format(ion_type_name);
686  auto op = "=";
687  auto rhs = get_variable_name(ion_type_name);
688  statements.push_back(ShadowUseStatement{lhs, op, rhs});
689  auto statement = conc_write_statement(ion.name, concentration, index);
690  statements.push_back(ShadowUseStatement{statement, "", ""});
691  }
692  }
693  return statements;
694 }

697 /**
698  * \details Often top level verbatim blocks use variables with old names.
699  * Here we process if we are processing verbatim block at global scope.
700  */
701 std::string CodegenCVisitor::process_verbatim_token(const std::string& token) {
702  const std::string& name = token;
703 
704  /*
705  * If given token is procedure name and if it's defined
706  * in the current mod file then it must be replaced
707  */
708  if (program_symtab->is_method_defined(token)) {
709  return method_name(token);
710  }
711 
712  /*
713  * Check if token is commongly used variable name in
714  * verbatim block like nt, \c \_threadargs etc. If so, replace
715  * it and return.
716  */

780  }
781  }

783  info.semantics.emplace_back(index++, "#{}_ion"_format(ion.name), 1);
784  }
785  }

787  if (info.first_pointer_var_index == -1) {

792  info.semantics.emplace_back(index, naming::POINTER_SEMANTIC, size);

795  }

797  }

1756 /**
1757  * @brief Checks whether the functor_block generated by sympy solver modifies any variable outside
1758  * its scope. If it does then return false, so that the operator() of the struct functor of the
1759  * Eigen Newton solver doesn't have const qualifier.
1760  *
1761  * @param variable_block Statement Block of the variables declarations used in the functor struct of
1762  * the solver
1763  * @param functor_block Actual code being printed in the operator() of the functor struct of the
1764  * solver
1765  * @return True if operator() is const else False
1766  */
1767 bool is_functor_const(const ast::StatementBlock& variable_block,
1768  const ast::StatementBlock& functor_block) {
1769  // Save DUChain for every variable in variable_block
1770  std::unordered_map<std::string, DUChain> chains;
1771 
1772  // Create complete_block with both variable declarations (done in variable_block) and solver
1773  // part (done in functor_block) to be able to run the SymtabVisitor and DefUseAnalyzeVisitor
1774  // then and get the proper DUChains for the variables defined in the variable_block
1775  ast::StatementBlock complete_block(functor_block);
1776  // Typically variable_block has only one statement, a statement containing the declaration
1777  // of the local variables
1778  for (const auto& statement: variable_block.get_statements()) {
1779  complete_block.insert_statement(complete_block.get_statements().begin(), statement);
1780  }
1781 
1782  // Create Symbol Table for complete_block
1783  auto model_symbol_table = std::make_shared<symtab::ModelSymbolTable>();
1784  SymtabVisitor(model_symbol_table.get()).visit_statement_block(complete_block);
1785  // Initialize DefUseAnalyzeVisitor to generate the DUChains for the variables defined in the
1786  // variable_block
1787  DefUseAnalyzeVisitor v(*complete_block.get_symbol_table());
1788 
1789  // Check the DUChains for all the variables in the variable_block
1790  // If variable is defined in complete_block don't add const quilifier in operator()
1791  auto is_functor_const = true;
1792  const auto& variables = collect_nodes(variable_block, {ast::AstNodeType::LOCAL_VAR});
1793  for (const auto& variable: variables) {
1794  const auto& chain = v.analyze(complete_block, variable->get_node_name());
1795  is_functor_const = !(chain.eval() == DUState::D || chain.eval() == DUState::LD ||
1796  chain.eval() == DUState::CD);
1797  if (!is_functor_const)
1798  break;
1799  }
1800 
1801  return is_functor_const;
1802 }
