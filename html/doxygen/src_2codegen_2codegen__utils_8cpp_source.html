16 /**
17  * \details We can directly print value but if user specify value as integer then
18  * then it gets printed as an integer. To avoid this, we use below wrapper.
19  * If user has provided integer then it gets printed as 1.0 (similar to mod2c
20  * and neuron where ".0" is appended). Otherwise we print double variables as
21  * they are represented in the mod file by user. If the value is in scientific
22  * representation (1e+20, 1E-15) then keep it as it is.
23  */
24 template <>
25 std::string format_double_string<CodegenCVisitor>(const std::string& s_value) {
26  double value = std::stod(s_value);
27  if (std::ceil(value) == value && s_value.find_first_of("eE") == std::string::npos) {
28  return "{:.1f}"_format(value);
29  }
30  return s_value;
31 }

34 /**
35  * \details In ISPC we have to explicitly append `d` to a floating point number
36  * otherwise it is treated as float. A value stored in the AST can be in
37  * scientific notation and hence we can't just append `d` to the string.
38  * Hence, we have to transform the value into the ISPC compliant format by
39  * replacing `e` and `E` with `d` to keep the same representation of the
40  * number as in the cpp backend.
41  */
42 template <>
43 std::string format_double_string<CodegenIspcVisitor>(const std::string& s_value) {
44  std::string return_string = s_value;
45  if (s_value.find_first_of("eE") != std::string::npos) {
46  std::replace(return_string.begin(), return_string.end(), 'E', 'd');
47  std::replace(return_string.begin(), return_string.end(), 'e', 'd');
48  } else if (s_value.find('.') == std::string::npos) {
49  return_string += ".0d";
50  } else if (s_value.front() == '.') {
51  return_string = '0' + return_string + 'd';
52  } else {
53  return_string += 'd';
54  }
55  return return_string;
56 }

69 /**
70  * \details For float variables we don't have to do the conversion with changing `e` and
71  * `E` with `f`, since the scientific notation numbers are already parsed as
72  * floats by ISPC. Instead we need to take care of only appending `f` to the
73  * end of floating point numbers, which is optional on ISPC.
74  */
75 template <>
76 std::string format_float_string<CodegenIspcVisitor>(const std::string& s_value) {
77  std::string return_string = s_value;
78  if (s_value.find_first_of("Ee.") == std::string::npos) {
79  return_string += ".0f";
80  } else if (s_value.front() == '.' && s_value.find_first_of("Ee") == std::string::npos) {
81  return_string = '0' + return_string + 'f';
82  } else if (s_value.find_first_of("Ee") == std::string::npos) {
83  return_string += 'f';
84  }
85  return return_string;
86 }
