28 /**
29  * \class IspcRenameVisitor
30  * \brief Rename variable names to match with ISPC backend requirement
31  *
32  * Wrapper class of RenameVisitor to rename the
33  * variables of the mod file that match the representation
34  * of double constants by the ISPC compiler. If those
35  * variables are not renamed to avoid matching the
36  * representation of the double constants of the ISPC
37  * compiler, then there are compilation issues.
38  * Here are some examples of double constants that
39  * ISPC compiler parses: 3.14d, 31.4d-1, 1.d, 1.0d,
40  * 1d-2. A regex which matches these variables is:
41  * ([0-9\.]*d[\-0-9]+)|([0-9\.]+d[\-0-9]*)
42  *
43  * Example mod file:
44  * \code{.mod}
45  * NEURON {
46  * SUFFIX test_ispc_rename
47  * RANGE d1, d2, d3, var_d3, d4
48  * }
49  * ASSIGNED {
50  * d1
51  * d2
52  * d3
53  * var_d3
54  * d4
55  * }
56  * INITIAL {
57  * d1 = 1
58  * d2 = 2
59  * d3 = 3
60  * var_d3 = 3
61  * }
62  * PROCEDURE func () {
63  * VERBATIM
64  * d4 = 4;
65  * ENDVERBATIM
66  * }
67  * \endcode
68  * Variables d1, d2 and d4 match the double constant
69  * presentation of variables by ISPC and should be
70  * renamed to var_d1, var_d2 and var_d4 to avoid
71  * compilation errors.
72  * var_d3 can stay the same because it doesn't match
73  * the regex.
74  * In case var_d1 already exists as variable then d1
75  * is renamed to var_d1_<random_sequence>.
76  */
77 class IspcRenameVisitor: public RenameVisitor {
78  public:
79  /// Default constructor
80  IspcRenameVisitor() = delete;
81 
82  /// Constructor that takes as parameter the AST and calls the RenameVisitor
83  explicit IspcRenameVisitor(std::shared_ptr<ast::Program> ast)
84  : RenameVisitor(ast, "([0-9\\.]*d[\\-0-9]+)|([0-9\\.]+d[\\-0-9]*)", "var_", true, true) {}
85 };
