36 /**
37  * \class Symbol
38  * \brief Represent symbol in symbol table
39  *
40  * Symbol table generator pass visit the AST and insert symbol for
41  * each node into symtab::SymbolTable. Symbol could appear multiple
42  * times in a block or different global blocks. NmodlType object has all
43  * nmodl properties information.
44  *
45  * \todo
46  * - Multiple tokens (i.e. location information) for symbol should
47  * be tracked
48  * - Scope information should be more than just string
49  * - Perf block should track information about all usage of the symbol
50  * (would be helpful for perf modeling)
51  * - Need to keep track of all renaming information, currently only we
52  * keep last state
53  */
54 class Symbol {
55  /// name of the symbol
56  std::string name;
57 
58  /// original name of the symbol if renamed
59  std::string renamed_from;
60 
61  /// unique id or index position when symbol is inserted into specific table
62  int id = 0;
63 
64  /// first AST node for which symbol is inserted
65  /// Variable can appear multiple times in the mod file. This node
66  /// represent the first occurance of the variable in the input. Currently
67  /// we don't track all AST nodes.
68  ast::Ast* node = nullptr;
69 
70  /// token associated with symbol (from node)
71  ModToken token;
72 
73  /// properties of symbol as a result of usage across whole mod file
74  syminfo::NmodlType properties{syminfo::NmodlType::empty};
75 
76  /// status of symbol after processing through various passes
77  syminfo::Status status{syminfo::Status::empty};
78 
79  /// scope of the symbol (nmodl block name where it appears)
80  std::string scope;
81 
82  /// order in case of state / prime variable
83  int order = 0;
84 
85  /// order in which symbol appears in the mod file
86  /// Different variables appear in different blocks (NEURON, PARAMETER, STATE)
87  /// and accordingly they appear in the data array (in NEURON). This order is
88  /// based on appearance in the mod file.
89  int definition_order = -1;
90 
91  /// associated value in case of parameters, constant variable
92  std::shared_ptr<double> value;
93 
94  /// true if symbol represent array variable
95  bool array = false;
96 
97  /// dimension/length in case of array variable
98  int length = 1;
99 
100  /// number of values that variable can take in case of table variable
101  int num_values = 0;
102 
103  /// number of times symbol is read
104  int read_count = 0;
105 
106  /// number of times symbol is written
107  int write_count = 0;
108 
109  public:
110  /// \name Ctor & dtor
111  /// \{
112 
113  Symbol() = delete;
114 
115  Symbol(std::string name, ast::Ast* node)
116  : name(std::move(name))
117  , node(node) {}
118 
119  Symbol(std::string name, ModToken token)
120  : name(std::move(name))
121  , token(std::move(token)) {}
122 
123  Symbol(std::string name, ast::Ast* node, ModToken token)
124  : name(std::move(name))
125  , node(node)
126  , token(std::move(token)) {}
127 
128  /// \}

131  void read() noexcept {

133  }
134

136  void write() noexcept {
137  write_count++;
138  }

144  scope = s;
145  }

147  void set_id(int i) noexcept {
148  id = i;
149  }

151  /**
152  * Set new name for the symbol
153  *
154  * If symbol is already renamed, do not rename it again
155  * as we want to keep track of original name and not intermediate
156  * renames
157  */
158  void set_name(const std::string& new_name) {
159  if (renamed_from.empty()) {
160  renamed_from = name;
161  }
162  name = new_name;
163  }
164 
165  /**
166  * Set order in case of prime/state variable
167  *
168  * Prime variable will appear in different block and could have
169  * multiple derivative orders. We have to store highest order.
170  */
171  void set_order(int new_order) noexcept {
172  if (new_order > order) {
173  order = new_order;
174  }
175  }
176 
177  void set_definition_order(int order) noexcept {
178  definition_order = order;
179  }
180

261  /**
262  * Check if symbol represent an external variable
263  *
264  * External variables are the variables that are defined in NEURON
265  * and available in mod file.
266  *
267  * \todo Need to check if we should check two properties using
268  * has_any_property instead of exact comparison
269  *
270  * \sa nmodl::details::NEURON_VARIABLES
271  */
272  bool is_external_variable() const noexcept {
273  return (properties == syminfo::NmodlType::extern_neuron_variable ||
274  properties == syminfo::NmodlType::extern_method);

277  /// check if symbol has any of the given property
278  bool has_any_property(syminfo::NmodlType new_properties) const noexcept {
279  return static_cast<bool>(properties & new_properties);
280  }
281 
282  /// check if symbol has all of the given properties
283  bool has_all_properties(syminfo::NmodlType new_properties) const noexcept {
284  return ((properties & new_properties) == new_properties);
285  }
286 
287  /// check if symbol has any of the status
288  bool has_any_status(syminfo::Status new_status) const noexcept {
289  return static_cast<bool>(status & new_status);
290  }
291 
292  /// check if symbol has all of the status
293  bool has_all_status(syminfo::Status new_status) const noexcept {
294  return ((status & new_status) == new_status);
295  }
296

299  properties |= new_properties;
300  }
301 
302  /// add new property to symbol
303  void add_property(syminfo::NmodlType property) noexcept {
304  properties |= property;
305  }
306 
307  /// remove property from symbol
308  void remove_property(syminfo::NmodlType property) {
309  properties &= ~property;
310  }
311 
312  /// mark symbol as inlined (in case of procedure/function)
313  void mark_inlined() noexcept {
314  status |= syminfo::Status::inlined;
315  }
316 
317  /// mark symbol as newly created (in case of new variable)
318  void mark_created() noexcept {
319  status |= syminfo::Status::created;
320  }
321 
322  void mark_renamed() noexcept {
323  status |= syminfo::Status::renamed;
324  }
325 
326  /// mark symbol as localized (e.g. from RANGE to LOCAL conversion)
327  void mark_localized() noexcept {
328  status |= syminfo::Status::localized;
329  }
330 
331  void mark_thread_safe() noexcept {

333  }
334 
335  /// mark symbol as newly created variable for the STATE variable
336  /// this is used with legacy euler/derivimplicit solver where DState
337  /// variables are created
338  void created_from_state() noexcept {
339  mark_created();
340  status |= syminfo::Status::from_state;
341  }
342 
343  bool is_array() const noexcept {
344  return array;
345  }
346 
347  /// check if symbol is a variable in nmodl
348  bool is_variable() const noexcept;
349 
350  std::string to_string() const;
351 };
