29 /// Helper for creating C codegen visitor
30 std::shared_ptr<CodegenNeuronCppVisitor> create_neuron_cpp_visitor(
31  const std::shared_ptr<ast::Program>& ast,
32  const std::string& /* text */,
33  std::stringstream& ss) {
34  /// construct symbol table
35  SymtabVisitor().visit_program(*ast);
36 
37  /// run all necessary pass
38  InlineVisitor().visit_program(*ast);
39  NeuronSolveVisitor().visit_program(*ast);
40  SolveBlockVisitor().visit_program(*ast);
41 
42  /// create C code generation visitor
43  auto cv = std::make_shared<CodegenNeuronCppVisitor>("_test", ss, "double", false);
44  cv->setup(*ast);
45  return cv;
46 }

49 /// print entire code
50 std::string get_neuron_cpp_code(const std::string& nmodl_text,
51  const bool generate_gpu_code = false) {
52  const auto& ast = NmodlDriver().parse_string(nmodl_text);
53  std::stringstream ss;
54  auto cvisitor = create_neuron_cpp_visitor(ast, nmodl_text, ss);
55  cvisitor->visit_program(*ast);
56  return ss.str();
