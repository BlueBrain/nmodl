286  /// write ast to nmodl
287  const auto ast_to_nmodl = [nmodl_ast](ast::Program& ast, const std::string& filepath) {
288  if (nmodl_ast) {
289  NmodlPrintVisitor(filepath).visit_program(ast);
290  logger->info("AST to NMODL transformation written to {}", filepath);

294  for (const auto& file: mod_files) {
295  logger->info("Processing {}", file.string());
296 
297  const auto modfile = file.stem().string();
298 
299  /// create file path for nmodl file
300  auto filepath = [scratch_dir, modfile](const std::string& suffix) {
301  static int count = 0;
302  return fmt::format(
303  "{}/{}.{}.{}.mod", scratch_dir, modfile, std::to_string(count++), suffix);
304  };
305 
306  /// driver object creates lexer and parser, just call parser method
307  NmodlDriver driver;
308 
309  /// parse mod file and construct ast
310  const auto& ast = driver.parse_file(file);
311 
312  /// whether to update existing symbol table or create new
313  /// one whenever we run symtab visitor.
314  bool update_symtab = false;
315 
316  /// just visit the ast
317  AstVisitor().visit_program(*ast);
318 
319  /// construct symbol table
320  {
321  logger->info("Running symtab visitor");
322  SymtabVisitor(update_symtab).visit_program(*ast);
323  }
324 
325  /// Check some rules that ast should follow
326  {
327  logger->info("Running semantic analysis visitor");
328  if (SemanticAnalysisVisitor(oacc_backend).check(*ast)) {
329  return 1;
330  }
331  }
332 
333  /// use cnexp instead of after_cvode solve method
334  {
335  logger->info("Running CVode to cnexp visitor");
336  AfterCVodeToCnexpVisitor().visit_program(*ast);
337  ast_to_nmodl(*ast, filepath("after_cvode_to_cnexp"));
338  }
339 
340  /// GLOBAL to RANGE rename visitor
341  if (nmodl_global_to_range) {
342  // make sure to run perf visitor because code generator
343  // looks for read/write counts const/non-const declaration
344  PerfVisitor().visit_program(*ast);
345  // make sure to run the GlobalToRange visitor after all the
346  // reinitializations of Symtab
347  logger->info("Running GlobalToRange visitor");
348  GlobalToRangeVisitor(*ast).visit_program(*ast);
349  SymtabVisitor(update_symtab).visit_program(*ast);
350  ast_to_nmodl(*ast, filepath("global_to_range"));
351  }
352 
353  /// LOCAL to ASSIGNED visitor
354  if (nmodl_local_to_range) {
355  logger->info("Running LOCAL to ASSIGNED visitor");
356  PerfVisitor().visit_program(*ast);
357  LocalToAssignedVisitor().visit_program(*ast);
358  SymtabVisitor(update_symtab).visit_program(*ast);
359  ast_to_nmodl(*ast, filepath("local_to_assigned"));
360  }
361 
362  {
363  // Compatibility Checking
364  logger->info("Running code compatibility checker");
365  // run perfvisitor to update read/write counts
366  PerfVisitor().visit_program(*ast);
367 
368  // If we want to just check compatibility we return the result
369  if (only_check_compatibility) {
370  return CodegenCompatibilityVisitor().find_unhandled_ast_nodes(*ast);
371  }
372 
373  // If there is an incompatible construct and code generation is not forced exit NMODL
374  if (CodegenCompatibilityVisitor().find_unhandled_ast_nodes(*ast) && !force_codegen) {
375  return 1;
376  }
377  }
378 
379  if (show_symtab) {
380  logger->info("Printing symbol table");
381  auto symtab = ast->get_model_symbol_table();
382  symtab->print(std::cout);
383  }
384 
385  ast_to_nmodl(*ast, filepath("ast"));
386 
387  if (json_ast) {
388  std::string file{scratch_dir};
389  file += "/";
390  file += modfile;
391  file += ".ast.json";
392  logger->info("Writing AST into {}", file);
393  JSONVisitor(file).write(*ast);
394  }
395 
396  if (verbatim_rename) {
397  logger->info("Running verbatim rename visitor");
398  VerbatimVarRenameVisitor().visit_program(*ast);
399  ast_to_nmodl(*ast, filepath("verbatim_rename"));
400  }
401 
402  if (nmodl_const_folding) {
403  logger->info("Running nmodl constant folding visitor");
404  ConstantFolderVisitor().visit_program(*ast);
405  ast_to_nmodl(*ast, filepath("constfold"));
406  }
407 
408  if (nmodl_unroll) {
409  logger->info("Running nmodl loop unroll visitor");
410  LoopUnrollVisitor().visit_program(*ast);
411  ConstantFolderVisitor().visit_program(*ast);
412  ast_to_nmodl(*ast, filepath("unroll"));
413  SymtabVisitor(update_symtab).visit_program(*ast);
414  }
415 
416  /// note that we can not symtab visitor in update mode as we
417  /// replace kinetic block with derivative block of same name
418  /// in global scope
419  {
420  logger->info("Running KINETIC block visitor");
421  auto kineticBlockVisitor = KineticBlockVisitor();
422  kineticBlockVisitor.visit_program(*ast);
423  SymtabVisitor(update_symtab).visit_program(*ast);
424  const auto filename = filepath("kinetic");
425  ast_to_nmodl(*ast, filename);
426  if (nmodl_ast && kineticBlockVisitor.get_conserve_statement_count()) {
427  logger->warn(
428  fmt::format("{} presents non-standard CONSERVE statements in DERIVATIVE "
429  "blocks. Use it only for debugging/developing",
430  filename));
431  }
432  }
433 
434  {
435  logger->info("Running STEADYSTATE visitor");
436  SteadystateVisitor().visit_program(*ast);
437  SymtabVisitor(update_symtab).visit_program(*ast);
438  ast_to_nmodl(*ast, filepath("steadystate"));
439  }
440 
441  /// Parsing units fron "nrnunits.lib" and mod files
442  {
443  logger->info("Parsing Units");
444  UnitsVisitor(units_dir).visit_program(*ast);
445  }
446 
447  /// once we start modifying (especially removing) older constructs
448  /// from ast then we should run symtab visitor in update mode so
449  /// that old symbols (e.g. prime variables) are not lost
450  update_symtab = true;
451 
452  if (nmodl_inline) {
453  logger->info("Running nmodl inline visitor");
454  InlineVisitor().visit_program(*ast);
455  SymtabVisitor(update_symtab).visit_program(*ast);
456  ast_to_nmodl(*ast, filepath("inline"));
457  }
458 
459  if (local_rename) {
460  logger->info("Running local variable rename visitor");

462  SymtabVisitor(update_symtab).visit_program(*ast);

465 
466  if (nmodl_localize) {

469  LocalizeVisitor(localize_verbatim).visit_program(*ast);

471  SymtabVisitor(update_symtab).visit_program(*ast);
472  ast_to_nmodl(*ast, filepath("localize"));
473  }
474 
475  if (sympy_conductance) {
476  logger->info("Running sympy conductance visitor");
477  SympyConductanceVisitor().visit_program(*ast);
478  SymtabVisitor(update_symtab).visit_program(*ast);
479  ast_to_nmodl(*ast, filepath("sympy_conductance"));
480  }
481 
482  if (sympy_analytic || sparse_solver_exists(*ast)) {
483  if (!sympy_analytic) {
484  logger->info(
485  "Automatically enable sympy_analytic because it exists solver of type sparse");
486  }
487  logger->info("Running sympy solve visitor");
488  SympySolverVisitor(sympy_pade, sympy_cse).visit_program(*ast);
489  SymtabVisitor(update_symtab).visit_program(*ast);
490  ast_to_nmodl(*ast, filepath("sympy_solve"));
491  }
492 
493  {
494  logger->info("Running cnexp visitor");
495  NeuronSolveVisitor().visit_program(*ast);
496  ast_to_nmodl(*ast, filepath("cnexp"));
497  }
498 
499  {
500  SolveBlockVisitor().visit_program(*ast);
501  SymtabVisitor(update_symtab).visit_program(*ast);
502  ast_to_nmodl(*ast, filepath("solveblock"));
503  }
504 
505  if (json_perfstat) {
506  std::string file{scratch_dir};
507  file.append("/");
508  file.append(modfile);
509  file.append(".perf.json");
510  logger->info("Writing performance statistics to {}", file);
511  PerfVisitor(file).visit_program(*ast);
512  }
513 
514  // Add implicit arguments (like celsius, nt) to NEURON functions (like
515  // nrn_ghk, at_time) whose signatures we have to massage.
516  ImplicitArgumentVisitor{}.visit_program(*ast);
517  SymtabVisitor(update_symtab).visit_program(*ast);
518 
519  {
520  // make sure to run perf visitor because code generator
521  // looks for read/write counts const/non-const declaration
522  PerfVisitor().visit_program(*ast);
523  }
524 
525  {
526  CodegenTransformVisitor{}.visit_program(*ast);
527  ast_to_nmodl(*ast, filepath("TransformVisitor"));
528  SymtabVisitor(update_symtab).visit_program(*ast);
529  }
530 
531  {
532  if (coreneuron_code && oacc_backend) {
533  logger->info("Running OpenACC backend code generator for CoreNEURON");
534  CodegenAccVisitor visitor(modfile,
535  output_dir,
536  data_type,
537  optimize_ionvar_copies_codegen);
538  visitor.visit_program(*ast);
539  }
540 
541  else if (coreneuron_code && !neuron_code && cpp_backend) {
542  logger->info("Running C++ backend code generator for CoreNEURON");
543  CodegenCoreneuronCppVisitor visitor(modfile,
544  output_dir,
545  data_type,
546  optimize_ionvar_copies_codegen);
547  visitor.visit_program(*ast);
548  }
549 
550  else if (neuron_code && cpp_backend) {
551  logger->info("Running C++ backend code generator for NEURON");
552  CodegenNeuronCppVisitor visitor(modfile,
553  output_dir,
554  data_type,
555  optimize_ionvar_copies_codegen);
556  visitor.visit_program(*ast);
557  }
558 
559  else {
560  throw std::runtime_error(
561  "Non valid code generation configuration. Code generation with NMODL is "
562  "supported for NEURON with C++ backend or CoreNEURON with C++/OpenACC "
563  "backends");
564  }
565  }
566  }
567 
568  if (sympy_opt) {
569  nmodl::pybind_wrappers::EmbeddedPythonLoader::get_instance().api()->finalize_interpreter();
570  }
