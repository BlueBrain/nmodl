25 std::string generate_mod_after_implicit_argument_visitor(std::string const& text) {
26  parser::NmodlDriver driver{};
27  auto const ast = driver.parse_string(text);
28  visitor::SymtabVisitor{}.visit_program(*ast);
29  visitor::ImplicitArgumentVisitor{}.visit_program(*ast);
30  return to_nmodl(*ast);
31 }

33 SCENARIO("Check insertion of implicit arguments", "[codegen][implicit_arguments]") {
34  GIVEN("A mod file that calls functions that may need implicit arguments") {
35  auto const nmodl_text = R"(
36  NEURON {
37  SUFFIX ImplicitTest
38  }
39  INITIAL {
40  at_time(foo)
41  at_time(a+b)
42  at_time(nt, flop)
43  nrn_ghk(-50(mV), .001(mM), 10(mM), 2)
44  nrn_ghk(-50(mV), .001(mM), 10(mM), 2, -273.15)
45  }
46  )";
47  auto const modified_nmodl = generate_mod_after_implicit_argument_visitor(nmodl_text);
48  THEN("at_time should have nt as its first argument") {
49  REQUIRE_THAT(modified_nmodl, Contains("at_time(nt, foo)"));
50  REQUIRE_THAT(modified_nmodl, Contains("at_time(nt, a+b)"));
51  REQUIRE_THAT(modified_nmodl, Contains("at_time(nt, flop)"));
52  }
53  THEN("nrn_ghk should have a temperature as its last argument") {
54  REQUIRE_THAT(modified_nmodl,
55  Contains("nrn_ghk(-50(mV), .001(mM), 10(mM), 2, celsius)"));
56  REQUIRE_THAT(modified_nmodl,
57  Contains("nrn_ghk(-50(mV), .001(mM), 10(mM), 2, -273.15)"));
58  }
59  }
60 }
