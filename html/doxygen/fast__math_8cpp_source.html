17 template <class T, class = typename std::enable_if<std::is_floating_point<T>::value>::type>
18 bool check_over_span(T f_ref(T),
19  T f_test(T),
20  const T low_limit,
21  const T high_limit,
22  const size_t npoints) {
23  constexpr uint nULP = 4;
24  constexpr T eps = std::numeric_limits<T>::epsilon();
25  constexpr T one_o_eps = 1.0 / std::numeric_limits<T>::epsilon();
26  T low = std::numeric_limits<T>::min() * one_o_eps * 1e2;
27 
28  T range = high_limit - low_limit;
29 
30  bool ret = true;
31  for (size_t i = 0; i < npoints; ++i) {
32  T x = low_limit + range * i / npoints;
33  T ref = f_ref(x);
34  T test = f_test(x);
35  T diff = std::abs(ref - test);
36  T max = std::max(std::abs(ref), std::abs(test));
37  T tol = max * nULP;
38  // normalize based on range
39  if (tol > low) {
40  tol *= eps;
41  } else {
42  diff *= one_o_eps;
43  }
44  if (diff > tol && diff != 0.0) {
45  ret = false;
46  }
47  }
48  return ret;
49 }

51 template <class T, class = typename std::enable_if<std::is_floating_point<T>::value>::type>
52 T exprelr_ref(const T x) {
53  return (1.0 + x == 1.0) ? 1.0 : x / (std::exp(x) - 1.0);
54 };
