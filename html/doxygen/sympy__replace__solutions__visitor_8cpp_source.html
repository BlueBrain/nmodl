20 /**
21  * \details SympyReplaceSolutionsVisitor tells us that a new equation appear and, depending where
22  * it is located, it can determine if it is part of the main system of equations or is something
23  * else. Every time we are out of the system and we print a new equation that is in the system
24  * we update the counter. \ref in_system follows, with lag, \param is_in_system and every time
25  * they are false and true respectively we detect a switch.
26  *
27  * \param is_in_system is a bool provided from outside that tells us if a new equation is indeed
28  * part of the main system of equations
29  */
30 void SympyReplaceSolutionsVisitor::InterleavesCounter::new_equation(const bool is_in_system) {
31  n_interleaves += (!in_system && is_in_system); // count an interleave only if in_system ==
32  // false and is_in_system == true
33  in_system = is_in_system; // update in_system
34 }

36 SympyReplaceSolutionsVisitor::SympyReplaceSolutionsVisitor(
37  const std::vector<std::string>& pre_solve_statements,
38  const std::vector<std::string>& solutions,
39  const std::unordered_set<ast::Statement*>& to_be_removed,
40  const ReplacePolicy policy,
41  const size_t n_next_equations,
42  const std::string& tmp_unique_prefix)
43  : pre_solve_statements(pre_solve_statements.begin(), pre_solve_statements.end(), 2)
44  , to_be_removed(&to_be_removed)
45  , policy(policy)
46  , n_next_equations(n_next_equations)
47  , replaced_statements_range(-1, -1) {
48  // if tmp_unique_prefix we do not expect tmp_statements
49  const auto ss_tmp_delimeter =
50  tmp_unique_prefix.empty()
51  ? solutions.begin()
52  : std::find_if(solutions.begin(),
53  solutions.end(),
54  [&tmp_unique_prefix](const std::string& statement) {
55  return statement.substr(0, tmp_unique_prefix.size()) !=
56  tmp_unique_prefix;
57  });
58  tmp_statements = StatementDispenser(solutions.begin(), ss_tmp_delimeter, -1);
59  solution_statements = StatementDispenser(ss_tmp_delimeter, solutions.end(), -1);
60 
61  replacements.clear();
62  is_top_level_statement_block = true;
63 }

66 void SympyReplaceSolutionsVisitor::visit_statement_block(ast::StatementBlock& node) {
67  const bool current_is_top_level_statement_block =
68  is_top_level_statement_block; // we mark it down since we are going to change it for
69  // visiting the children
70  is_top_level_statement_block = false;
71 
72  if (current_is_top_level_statement_block) {
73  logger->debug("SympyReplaceSolutionsVisitor :: visit statements. Matching policy: {}",
74  (policy == ReplacePolicy::VALUE ? "VALUE" : "GREEDY"));
75  interleaves_counter = InterleavesCounter();
76  node.visit_children(*this);
77 
78  if (!solution_statements.tags.empty() && policy == ReplacePolicy::VALUE) {
79  logger->debug(
80  "SympyReplaceSolutionsVisitor :: not all solutions were replaced. Policy: GREEDY");
81  interleaves_counter = InterleavesCounter();
82  policy = ReplacePolicy::GREEDY;
83  node.visit_children(*this);
84 
85  if (interleaves_counter.n() > 0) {
86  logger->warn(
87  "SympyReplaceSolutionsVisitor :: Found ambiguous system of equations "
88  "interleaved with {} assignment statements. I do not know what equations go "
89  "before and what "
90  "equations go after the assignment statements. Either put all the equations "
91  "that need to be solved "
92  "in the form: x = f(...) and with distinct variable assignments or do not "
93  "interleave the system with assignments.",
94  interleaves_counter.n());
95  }
96  }
97  } else {
98  node.visit_children(*this);
99  }
100 
101  const auto& old_statements = node.get_statements();
102 
103  ast::StatementVector new_statements;
104  new_statements.reserve(2 * old_statements.size());
105  for (const auto& old_statement: old_statements) {
106  const auto& replacement_ptr = replacements.find(old_statement);
107  if (replacement_ptr != replacements.end()) {
108  if (replaced_statements_range.first == -1) {
109  replaced_statements_range.first = new_statements.size();
110  }
111 
112  new_statements.insert(new_statements.end(),
113  replacement_ptr->second.begin(),
114  replacement_ptr->second.end());
115 
116  replaced_statements_range.second = new_statements.size();
117 
118  logger->debug("SympyReplaceSolutionsVisitor :: erasing {}", to_nmodl(old_statement));
119  for (const auto& replacement: replacement_ptr->second) {
120  logger->debug("SympyReplaceSolutionsVisitor :: adding {}", to_nmodl(replacement));
121  }
122  } else if (to_be_removed == nullptr ||
123  to_be_removed->find(&(*old_statement)) == to_be_removed->end()) {
124  logger->debug("SympyReplaceSolutionsVisitor :: found {}, nothing to do",
125  to_nmodl(old_statement));
126  new_statements.emplace_back(std::move(old_statement));
127  } else {
128  logger->debug("SympyReplaceSolutionsVisitor :: erasing {}", to_nmodl(old_statement));
129  }
130  }
131 
132  if (current_is_top_level_statement_block) {
133  if (!solution_statements.tags.empty()) {
134  std::ostringstream ss;
135  for (const auto ii: solution_statements.tags) {
136  ss << to_nmodl(solution_statements.statements[ii]) << '\n';
137  }
138  throw std::runtime_error(
139  "Not all solutions were replaced! Sympy returned {} equations but I could not find "
140  "a place "
141  "for all of them. In particular, the following equations remain to be replaced "
142  "somewhere:\n{}This is "
143  "probably a bug and I invite you to report it to a developer. Possible causes:\n"
144  " - I did not do a GREEDY pass and some solutions could not be replaced by VALUE\n "
145  "sympy "
146  "returned more equations than what we expected\n - There is a bug in the GREEDY "
147  "pass\n - some "
148  "solutions were replaced but not untagged"_format(
149  solution_statements.statements.size(), ss.str()));
150  }
151 
152  if (replaced_statements_range.first == -1) {
153  replaced_statements_range.first = new_statements.size();
154  }
155  if (replaced_statements_range.second == -1) {
156  replaced_statements_range.second = new_statements.size();
157  }
158  }
159 
160  node.set_statements(std::move(new_statements));
161 }

163 void SympyReplaceSolutionsVisitor::try_replace_tagged_statement(
164  const ast::Node& node,
165  const std::shared_ptr<ast::Expression>& get_lhs(const ast::Node& node),
166  const std::shared_ptr<ast::Expression>& get_rhs(const ast::Node& node)) {
167  interleaves_counter.new_equation(true);
168 
169  const auto& statement = std::static_pointer_cast<ast::Statement>(
170  node.get_parent()->get_shared_ptr());
171 
172  // do not visit if already marked
173  if (replacements.find(statement) != replacements.end()) {
174  return;
175  }
176 
177 
178  switch (policy) {
179  case ReplacePolicy::VALUE: {
180  const auto dependencies = statement_dependencies(get_lhs(node), get_rhs(node));
181  const auto& key = dependencies.first;
182 
183  if (solution_statements.is_var_assigned_here(key)) {
184  logger->debug("SympyReplaceSolutionsVisitor :: marking for replacement {}",
185  to_nmodl(statement));
186 
187  ast::StatementVector new_statements;
188 
189  pre_solve_statements.emplace_back_all_tagged_statements(new_statements);
190  tmp_statements.emplace_back_all_tagged_statements(new_statements);
191  solution_statements.try_emplace_back_tagged_statement(new_statements, key);
192 
193  replacements.emplace(statement, new_statements);
194  }
195  break;
196  }
197  case ReplacePolicy::GREEDY: {
198  logger->debug("SympyReplaceSolutionsVisitor :: marking for replacement {}",
199  to_nmodl(statement));
200 
201  ast::StatementVector new_statements;
202 
203  pre_solve_statements.emplace_back_all_tagged_statements(new_statements);
204  tmp_statements.emplace_back_all_tagged_statements(new_statements);
205  solution_statements.emplace_back_next_tagged_statements(new_statements, n_next_equations);
206 
207  replacements.emplace(statement, new_statements);
208  break;
209  }
210  }

279 /**
280  * \details CHere we construct a map variable -> affected equations. In other words this map tells
281  * me what equations need to be updated when I change a particular variable. To do that we build a a
282  * graph of dependencies var -> vars and in the mean time we reduce it to the root variables. This
283  * is ensured by the fact that the tmp variables are sorted so that the next tmp variable may depend
284  * on the previous one. Since it is a relation of equivalence (if an equation depends on a variable,
285  * it needs to be updated if the variable changes), we build the two maps at the same time.
286  *
287  * An example:
288  *
289  * \code{.mod}
290  * tmp0 = x + a
291  * tmp1 = tmp0 + b
292  * tmp2 = y
293  * \endcode
294  *
295  * dependency_map should be (the order of the equation is unimportant since we are building
296  * a map):
297  *
298  * - tmp0 : x, a
299  * - tmp1 : x, a, b
300  * - tmp2 : y
301  *
302  * and the var2statement map should be (the order of the following equations is unimportant
303  * since we are building a map. The number represents the index of the original equations):
304  *
305  * - x : 0, 1
306  * - y : 2
307  * - a : 0, 1
308  * - b : 1
309  *
310  */
311 void SympyReplaceSolutionsVisitor::StatementDispenser::build_maps() {
312  for (size_t ii = 0; ii < statements.size(); ++ii) {
313  const auto& statement = statements[ii];
314 
315  if (statement->is_expression_statement()) {
316  const auto& e_statement =
317  std::static_pointer_cast<ast::ExpressionStatement>(statement)->get_expression();
318  if (e_statement->is_binary_expression()) {
319  const auto& bin_exp = std::static_pointer_cast<ast::BinaryExpression>(e_statement);
320  const auto& dependencies = statement_dependencies(bin_exp->get_lhs(),
321  bin_exp->get_rhs());
322 
323  const auto& key = dependencies.first;
324  const auto& vars = dependencies.second;
325  if (!key.empty()) {
326  var2statement.emplace(key, ii);
327  for (const auto& var: vars) {
328  const auto& var_already_inserted = dependency_map.find(var);
329  if (var_already_inserted != dependency_map.end()) {
330  dependency_map[key].insert(var_already_inserted->second.begin(),
331  var_already_inserted->second.end());
332  for (const auto& root_var: var_already_inserted->second) {
333  var2dependants[root_var].insert(ii);
334  }
335  } else {
336  dependency_map[key].insert(var);
337  var2dependants[var].insert(ii);
338  }
339  }
340  }
341  }
342  }
343  }
344 
345  logger->debug("SympyReplaceSolutionsVisitor::StatementDispenser :: var2dependants map");
346  for (const auto& entry: var2dependants) {
347  logger->debug("SympyReplaceSolutionsVisitor::StatementDispenser :: var `{}` used in:",
348  entry.first);
349  for (const auto ii: entry.second) {
350  logger->debug("SympyReplaceSolutionsVisitor::StatementDispenser :: -> {}",
351  to_nmodl(statements[ii]));
352  }
353  }
354  logger->debug("SympyReplaceSolutionsVisitor::StatementDispenser :: var2statement map");
355  for (const auto& entry: var2statement) {
356  logger->debug("SympyReplaceSolutionsVisitor::StatementDispenser :: var `{}` defined in:",
357  entry.first);
358  logger->debug("SympyReplaceSolutionsVisitor::StatementDispenser :: -> {}",
359  to_nmodl(statements[entry.second]));
360  }
361 }

363 bool SympyReplaceSolutionsVisitor::StatementDispenser::try_emplace_back_tagged_statement(
364  ast::StatementVector& new_statements,
365  const std::string& var) {
366  auto ptr = var2statement.find(var);
367  bool emplaced = false;
368  if (ptr != var2statement.end()) {
369  const auto ii = ptr->second;
370  const auto tag_ptr = tags.find(ii);
371  if (tag_ptr != tags.end()) {
372  new_statements.emplace_back(statements[ii]->clone());
373  tags.erase(tag_ptr);
374  emplaced = true;
375 
376  logger->debug(
377  "SympyReplaceSolutionsVisitor::StatementDispenser :: adding to replacement rule {}",
378  to_nmodl(statements[ii]));
379  } else {
380  logger->error(
381  "SympyReplaceSolutionsVisitor::StatementDispenser :: tried adding to replacement "
382  "rule {} but statement is not "
383  "tagged",
384  to_nmodl(statements[ii]));
385  }
386  }
387  return emplaced;
388 }

391  ast::StatementVector& new_statements,
392  const size_t n_next_statements) {
393  size_t counter = 0;
394  for (size_t next_statement_ii = 0;
395  next_statement_ii < statements.size() && counter < n_next_statements;
396  ++next_statement_ii) {
397  const auto tag_ptr = tags.find(next_statement_ii);
398  if (tag_ptr != tags.end()) {
399  logger->debug(
400  "SympyReplaceSolutionsVisitor::StatementDispenser :: adding to replacement rule {}",
401  to_nmodl(statements[next_statement_ii]));
402  new_statements.emplace_back(statements[next_statement_ii]->clone());
403  tags.erase(tag_ptr);
404  ++counter;
405  }
406  }
