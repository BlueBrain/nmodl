35 std::vector<std::string> run_sympy_solver_visitor(
36  const std::string& text,
37  bool pade = false,
38  bool cse = false,
39  AstNodeType ret_nodetype = AstNodeType::DIFF_EQ_EXPRESSION,
40  bool kinetic = false) {
41  std::vector<std::string> results;
42 
43  // construct AST from text
44  NmodlDriver driver;
45  const auto& ast = driver.parse_string(text);
46 
47  // construct symbol table from AST
48  SymtabVisitor().visit_program(*ast);
49 
50  // unroll loops and fold constants
51  ConstantFolderVisitor().visit_program(*ast);
52  LoopUnrollVisitor().visit_program(*ast);
53  ConstantFolderVisitor().visit_program(*ast);
54  SymtabVisitor().visit_program(*ast);
55 
56  if (kinetic) {
57  KineticBlockVisitor().visit_program(*ast);
58  }
59 
60  // run SympySolver on AST
61  SympySolverVisitor(pade, cse).visit_program(*ast);
62 
63  // check that, after visitor rearrangement, parents are still up-to-date
64  CheckParentVisitor().check_ast(*ast);
65 
66  // run lookup visitor to extract results from AST
67  for (const auto& eq: collect_nodes(*ast, {ret_nodetype})) {
68  results.push_back(to_nmodl(eq));
69  }
70 
71  return results;
72 }

74 // check if in a list of vars (like LOCAL) there are duplicates
75 bool is_unique_vars(std::string result) {
76  result.erase(std::remove(result.begin(), result.end(), ','), result.end());
77  std::stringstream ss(result);
78  std::string token;
79 
80  std::unordered_set<std::string> old_vars;
81 
82  while (getline(ss, token, ' ')) {
83  if (!old_vars.insert(token).second) {
84  return false;
85  }
86  }
87  return true;
88 }

91 /**
92  * \brief Compare nmodl blocks that contain systems of equations (i.e. derivative, linear, etc.)
93  *
94  * This is basically and advanced string == string comparison where we detect the (various) systems
95  * of equations and check if they are equivalent. Implemented mostly in python since we need a call
96  * to sympy to simplify the equations.
97  *
98  * - compare_systems_of_eq The core of the code. \p result_dict and \p expected_dict are
99  * dictionaries that represent the systems of equations in this way:
100  *
101  * a = b*x + c -> result_dict['a'] = 'b*x + c'
102  *
103  * where the variable \p a become a key \p k of the dictionary.
104  *
105  * In there we go over all the equations in \p result_dict and \p expected_dict and check that
106  * result_dict[k] - expected_dict[k] simplifies to 0.
107  *
108  * - sanitize is to transform the equations in something treatable by sympy (i.e. pow(dt, 3) ->
109  * dt**3
110  * - reduce back-substitution of the temporary variables
111  *
112  * \p require_fail requires that the equations are different. Used only for unit-test this function
113  *
114  * \warning do not use this method when there are tmp variables not in the form: tmp_<number>
115  */
116 void compare_blocks(const std::string& result,
117  const std::string& expected,
118  const bool require_fail = false) {
119  using namespace pybind11::literals;
120 
121  auto locals =
122  pybind11::dict("result"_a = result, "expected"_a = expected, "is_equal"_a = false);
123  pybind11::exec(R"(
124  # Comments are in the doxygen for better highlighting
125  def compare_blocks(result, expected):
126 
127  def sanitize(s):
128  import re
129  d = {'\[(\d+)\]':'_\\1', 'pow\((\w+), ?(\d+)\)':'\\1**\\2', 'beta': 'beta_var', 'gamma': 'gamma_var'}
130  out = s
131  for key, val in d.items():
132  out = re.sub(key, val, out)
133  return out
134 
135  def compare_systems_of_eq(result_dict, expected_dict):
136  from sympy.parsing.sympy_parser import parse_expr
137  try:
138  for k, v in result_dict.items():
139  if parse_expr(f'simplify(({v})-({expected_dict[k]}))'):
140  return False
141  except KeyError:
142  return False
143 
144  result_dict.clear()
145  expected_dict.clear()
146  return True
147 
148  def reduce(s):
149  max_tmp = -1
150  d = {}
151 
152  sout = ""
153  # split of sout and a dict with the tmp variables
154  for line in s.split('\n'):
155  line_split = line.lstrip().split('=')
156 
157  if len(line_split) == 2 and line_split[0].startswith('tmp_'):
158  # back-substitution of tmp variables in tmp variables
159  tmp_var = line_split[0].strip()
160  if tmp_var in d:
161  continue
162 
163  max_tmp = max(max_tmp, int(tmp_var[4:]))
164  for k, v in d.items():
165  line_split[1] = line_split[1].replace(k, f'({v})')
166  d[tmp_var] = line_split[1]
167  elif 'LOCAL' in line:
168  sout += line.split('tmp_0')[0] + '\n'
169  else:
170  sout += line + '\n'
171 
172  # Back-substitution of the tmps
173  # so that we do not replace tmp_11 with (tmp_1)1
174  for j in range(max_tmp, -1, -1):
175  k = f'tmp_{j}'
176  sout = sout.replace(k, f'({d[k]})')
177 
178  return sout
179 
180  result = reduce(sanitize(result)).split('\n')
181  expected = reduce(sanitize(expected)).split('\n')
182 
183  if len(result) != len(expected):
184  return False
185 
186  result_dict = {}
187  expected_dict = {}
188  for token1, token2 in zip(result, expected):
189  if token1 == token2:
190  if not compare_systems_of_eq(result_dict, expected_dict):
191  return False
192  continue
193 
194  eq1 = token1.split('=')
195  eq2 = token2.split('=')
196  if len(eq1) == 2 and len(eq2) == 2:
197  result_dict[eq1[0]] = eq1[1]
198  expected_dict[eq2[0]] = eq2[1]
199  continue
200 
201  return False
202  return compare_systems_of_eq(result_dict, expected_dict)
203 
204  is_equal = compare_blocks(result, expected))",
205  pybind11::globals(),
206  locals);
207 
208  // Error log
209  if (require_fail == locals["is_equal"].cast<bool>()) {
210  if (require_fail) {
211  REQUIRE(result != expected);
212  } else {
213  REQUIRE(result == expected);
214  }
215  } else { // so that we signal to ctest that an assert was performed
216  REQUIRE(true);
217  }
218 }
