246 void parse_neuron_block_string(const std::string& name, nmodl::ModToken& value) {
247  nmodl::parser::NmodlDriver driver;
248  driver.parse_string(name);
249 
250  const auto& ast_program = driver.get_ast();
251  const auto& neuron_blocks = nmodl::collect_nodes(*ast_program->get_shared_ptr(),
252  {nmodl::ast::AstNodeType::NEURON_BLOCK});
253  value = *(neuron_blocks.front()->get_token());
254 }

257  GIVEN("A single NEURON block") {

265  RANGE gcabar, m_inf, tau_m, alph1, alph2, KK, shift
