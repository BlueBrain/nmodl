181 /**
182  * Definition type similar to old implementation
183  *
184  * In the original implementation of NMODL (mod2c, nocmodl) different vectors were
185  * created for \c extdef, \c extdef2, \c extdef3, \c extdef4 etc. We are changing
186  * those vectors with <c><name, type></c> map. This will help us to search
187  * in single map and find it's type. The types are defined as follows:
188  *
189  * - DefinitionType::EXT_DOUBLE : external names that can be used as doubles
190  * without giving an error message
191  * - DefinitionType::EXT_2 : external function names that can be used with
192  * array and function name arguments
193  * - DefinitionType::EXT_3 : function names that get two reset arguments
194  * - DefinitionType::EXT_4 : functions that need a first arg of \c NrnThread*
195  * - DefinitionType::EXT_5 : external definition names that are not \c threadsafe
196  *
197  */
198 enum class DefinitionType { EXT_DOUBLE, EXT_2, EXT_3, EXT_4, EXT_5 };

305 /**
306  * Checks if \c token is one of the functions coming from NEURON/CoreNEURON and needs
307  * passing NrnThread* as first argument (typical name of variable \c nt)
308  *
309  * @param token Name of function
310  * @return True or false depending if the function needs NrnThread* argument
311  */
312 bool needs_neuron_thread_first_arg(const std::string& token) {
313  auto extern_def = extern_definitions.find(token);
314  return extern_def != extern_definitions.end() && extern_def->second == DefinitionType::EXT_4;
315 }

356 /**
357  * Return token type for given token name
358  * @param name Token name from lexer
359  * @return type of NMODL token
360  */
361 TokenType token_type(const std::string& name) {
362  if (is_keyword(name)) {
363  return details::keyword_type(name);
364  }
365  if (is_method(name)) {
366  return Token::METHOD;
367  }
368  throw std::runtime_error("token_type called for non-existent token " + name);
369 }

383 /**
384  * Return functions that can be used in the NMODL
385  * @return vector of function names used in NMODL
386  */
387 std::vector<std::string> get_external_functions() {
388  std::vector<std::string> result;
389  result.reserve(details::methods.size());
390  for (auto& method: details::methods) {
391  result.push_back(method.first);
392  }
393  for (auto& definition: details::extern_definitions) {
394  result.push_back(definition.first);
395  }
396  return result;
