227 /**

229  * passing NrnThread* as first argument (typical name of variable \c nt)

258 /**
259  * Check if given name is a keyword in NMODL
260  * @param name token name
261  * @return true if name is a keyword
262  */
263 bool is_keyword(const std::string& name) {
264  return details::keywords.find(name) != details::keywords.end();
265 }

268 /**
269  * Check if given name is an integration method in NMODL
270  * @param name Name of the integration method
271  * @return true if name is an integration method in NMODL
272  */
273 bool is_method(const std::string& name) {
274  return (details::methods.find(name) != details::methods.end());
275 }

278 /**
279  * Return token type for given token name
280  * @param name Token name from lexer
281  * @return type of NMODL token
282  */
283 TokenType token_type(const std::string& name) {
284  if (is_keyword(name)) {
285  return details::keyword_type(name);
286  }
287  if (is_method(name)) {
288  return Token::METHOD;
289  }
290  throw std::runtime_error("token_type called for non-existent token " + name);
291 }

303 /**
304  * Return functions that can be used in the NMODL
305  * @return vector of function names used in NMODL
306  */
307 std::vector<std::string> get_external_functions() {
308  std::vector<std::string> result;
309  result.reserve(details::methods.size() + details::extern_definitions.size());
310  for (auto& method: details::methods) {
311  result.push_back(method.first);
312  }
313  result.insert(result.cend(),
314  details::extern_definitions.begin(),
315  details::extern_definitions.end());
316  return result;
317 }
