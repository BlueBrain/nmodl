131  * wrapper library and provides access to the API wrapper struct that can be used to access the
132  * pybind11 embedded python functionality.

141  static EmbeddedPythonLoader& get_instance() {
142  static EmbeddedPythonLoader instance;
143 
144  return instance;
145  }
146

148  void operator=(const EmbeddedPythonLoader&) = delete;

157  const pybind_wrap_api* api();
