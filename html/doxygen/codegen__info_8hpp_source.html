44 /**
45  * \class Ion
46  * \brief Represent ions used in mod file
47  */
48 struct Ion {
49  /// name of the ion
50  std::string name;
51 
52  /// ion variables that are being read
53  std::vector<std::string> reads;
54 
55  /// ion variables that are being implicitly read
56  std::vector<std::string> implicit_reads;
57 
58  /// ion variables that are being written
59  std::vector<std::string> writes;
60 
61  /// if style semantic needed
62  bool need_style = false;
63 
64  Ion() = delete;
65 
66  explicit Ion(std::string name)
67  : name(std::move(name)) {}
68 
69  /**
70  * Check if variable name is a ionic current
71  *
72  * This is equivalent of IONCUR flag in mod2c.
73  * If it is read variable then also get NRNCURIN flag.
74  * If it is write variables then also get NRNCUROUT flag.
75  */
76  bool is_ionic_current(const std::string& text) const {
77  return text == ("i" + name);
78  }
79 
80  /**
81  * Check if variable name is internal cell concentration
82  *
83  * This is equivalent of IONIN flag in mod2c.
84  */
85  bool is_intra_cell_conc(const std::string& text) const {
86  return text == (name + "i");
87  }
88 
89  /**
90  * Check if variable name is external cell concentration
91  *
92  * This is equivalent of IONOUT flag in mod2c.
93  */
94  bool is_extra_cell_conc(const std::string& text) const {
95  return text == (name + "o");
96  }
97 
98  /**
99  * Check if variable name is reveral potential
100  *
101  * This is equivalent of IONEREV flag in mod2c.
102  */
103  bool is_rev_potential(const std::string& text) const {
104  return text == ("e" + name);
105  }
106 
107  /// check if it is either internal or external concentration
108  bool is_ionic_conc(const std::string& text) const {
109  return is_intra_cell_conc(text) || is_extra_cell_conc(text);
110  }
111 
112  /// for a given ion, return different variable names/properties
113  /// like internal/external concentration, reversial potential,
114  /// ionic current etc.
115  static std::vector<std::string> get_possible_variables(const std::string& ion_name) {
116  return {"i" + ion_name, ion_name + "i", ion_name + "o", "e" + ion_name};
117  }
118 };
