41 /**
42  * \class Ion
43  * \brief Represent ions used in mod file
44  */
45 struct Ion {
46  /// name of the ion
47  std::string name;
48 
49  /// ion variables that are being read
50  std::vector<std::string> reads;
51 
52  /// ion variables that are being written
53  std::vector<std::string> writes;
54 
55  /// if style semantic needed
56  bool need_style = false;
57 
58  Ion() = delete;
59 
60  Ion(std::string name)
61  : name(std::move(name)) {}
62 
63  /**
64  * Check if variable name is a ionic current
65  *
66  * This is equivalent of IONCUR flag in mod2c.
67  * If it is read variable then also get NRNCURIN flag.
68  * If it is write variables then also get NRNCUROUT flag.
69  */
70  bool is_ionic_current(const std::string& text) const {
71  return text == ("i" + name);
72  }
73 
74  /**
75  * Check if variable name is internal cell concentration
76  *
77  * This is equivalent of IONIN flag in mod2c.
78  */
79  bool is_intra_cell_conc(const std::string& text) const {
80  return text == (name + "i");
81  }
82 
83  /**
84  * Check if variable name is external cell concentration
85  *
86  * This is equivalent of IONOUT flag in mod2c.
87  */
88  bool is_extra_cell_conc(const std::string& text) const {
89  return text == (name + "o");
90  }
91 
92  /**
93  * Check if variable name is reveral potential
94  *
95  * This is equivalent of IONEREV flag in mod2c.
96  */
97  bool is_rev_potential(const std::string& text) const {
98  return text == ("e" + name);
99  }
100 
101  /// check if it is either internal or external concentration
102  bool is_ionic_conc(const std::string& text) const {
103  return is_intra_cell_conc(text) || is_extra_cell_conc(text);
104  }
105 };
