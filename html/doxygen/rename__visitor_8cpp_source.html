20 std::string RenameVisitor::new_name_generator(const std::string& old_name) {
21  std::string new_name;
22  if (add_random_suffix) {
23  if (renamed_variables.find(old_name) != renamed_variables.end()) {
24  new_name = renamed_variables[old_name];
25  } else {
26  const auto& vars = get_global_vars(*ast);
27  if (add_prefix) {
28  new_name = suffix_random_string(vars,
29  new_var_name_prefix + old_name,
30  UseNumbersInString::WithoutNumbers);
31  } else {
32  new_name =
33  suffix_random_string(vars, new_var_name, UseNumbersInString::WithoutNumbers);
34  }
35  renamed_variables[old_name] = new_name;
36  }
37  } else {
38  if (add_prefix) {
39  new_name = new_var_name_prefix + old_name;
40  } else {
41  new_name = new_var_name;
42  }
43  }
44  return new_name;
45 }

48 void RenameVisitor::visit_name(const ast::Name& node) {
49  const auto& name = node.get_node_name();
50  if (std::regex_match(name, var_name_regex)) {
51  std::string new_name = new_name_generator(name);

54  ? " at " + node.get_token()->position()

56  logger->warn("RenameVisitor :: Renaming variable {}{} to {}", name, token_string, new_name);
57  }

69 /**
70  * Parse verbatim blocks and rename variable if it is used.
71  */
72 void RenameVisitor::visit_verbatim(const ast::Verbatim& node) {
73  if (!rename_verbatim) {
74  return;
75  }
76 
77  const auto& statement = node.get_statement();
78  const auto& text = statement->eval();
79  parser::CDriver driver;
80 
81  driver.scan_string(text);
82  auto tokens = driver.all_tokens();
83 
84  std::ostringstream result;
85  for (auto& token: tokens) {
86  if (std::regex_match(token, var_name_regex)) {
87  /// Check if variable is already renamed and use the same naming otherwise add the
88  /// new_name to the renamed_variables map
89  const std::string& new_name = new_name_generator(token);
90  result << new_name;
91  logger->warn("RenameVisitor :: Renaming variable {} in VERBATIM block to {}",
92  token,
93  new_name);
94  } else {
95  result << token;
96  }
97  }
98  statement->set(result.str());
99 }
