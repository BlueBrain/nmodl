27 /**
28  * Analyse breakpoint block to check if it is safe to insert CONDUCTANCE statements
29  *
30  * Most of the mod files have simple breakpoint blocks without any control flow
31  * statements. SympyConductanceVisitor just collects all the statements in the
32  * breakpoint block and insert conductance statements. If there are control flow
33  * statements like `IF { a } ELSE { b }` block with conflicting current statements
34  * inside IF and ELSE blocks or VERBATIM block then the resulting CONDUCTANCE
35  * statements may be incorrect. For now the simple approach is to not generate
36  * CONDUCTANCE statements if if-else statements exist in the block.
37  *
38  * @param node Ast node for breakpoint block
39  * @return true if it is safe to insert conductance statements otherwise false
40  */
41 static bool conductance_statement_possible(const ast::BreakpointBlock& node) {
42  return collect_nodes(node, {AstNodeType::IF_STATEMENT, AstNodeType::VERBATIM}).empty();
43 }

46 // Generate statement strings to be added to BREAKPOINT section
47 std::vector<std::string> SympyConductanceVisitor::generate_statement_strings(
48  ast::BreakpointBlock& node) {
49  std::vector<std::string> statements;
50 
51  // instead of passing all variables in the symbol table find out variables
52  // that are used in the current block and then pass to sympy
53  // name could be parameter or unit so check if it exist in symbol table
54  const auto& names_in_block = collect_nodes(node, {AstNodeType::NAME});
55  string_set used_names_in_block;
56  for (const auto& name: names_in_block) {
57  auto varname = name->get_node_name();
58  if (all_vars.find(varname) != all_vars.end()) {
59  used_names_in_block.insert(varname);
60  }
61  }
62 
63  // iterate over binary expression lhs's from breakpoint
64  for (const auto& lhs_str: ordered_binary_exprs_lhs) {
65  // look for a current name that matches lhs of expr (current write name)
66  auto it = i_name.find(lhs_str);
67  if (it != i_name.end()) {
68  const auto& equation_string = ordered_binary_exprs[binary_expr_index[lhs_str]];
69  std::string i_name_str = it->second;
70  // SymPy needs the current expression & all previous expressions
71  std::vector<std::string> expressions(ordered_binary_exprs.begin(),
72  ordered_binary_exprs.begin() +
73  binary_expr_index[lhs_str] + 1);
74  // differentiate dI/dV
75  auto analytic_diff =
76  pywrap::EmbeddedPythonLoader::get_instance().api()->create_ads_executor();
77  analytic_diff->expressions = expressions;
78  analytic_diff->used_names_in_block = used_names_in_block;
79  (*analytic_diff)();
80  auto dIdV = analytic_diff->solution;
81  auto exception_message = analytic_diff->exception_message;
82  pywrap::EmbeddedPythonLoader::get_instance().api()->destroy_ads_executor(analytic_diff);
83  if (!exception_message.empty()) {
84  logger->warn("SympyConductance :: python exception: {}", exception_message);
85  }
86  if (dIdV.empty()) {
87  logger->warn(
88  "SympyConductance :: analytic differentiation of ionic current "
89  "not possible");
90  } else {
91  std::string g_var = dIdV;
92  // if conductance g_var is not an existing variable, need to generate
93  // a new variable name, declare it, and assign it
94  if (all_vars.find(g_var) == all_vars.end()) {
95  // generate new variable name
96  std::map<std::string, int> var_map;
97  for (auto const& v: all_vars) {
98  var_map[v] = 0;
99  }
100  g_var = get_new_name("g", i_name_str, var_map);
101  // declare it
102  add_local_variable(*node.get_statement_block(), g_var);
103  // asign dIdV to it
104  std::string statement_str = g_var;
105  statement_str.append(" = ").append(dIdV);
106  statements.insert(statements.begin(), statement_str);
107  logger->debug("SympyConductance :: Adding BREAKPOINT statement: {}",
108  statement_str);
109  }
110  std::string statement_str = "CONDUCTANCE " + g_var;
111  if (!i_name_str.empty()) {
112  statement_str += " USEION " + i_name_str;
113  }
114  statements.push_back(statement_str);
115  logger->debug("SympyConductance :: Adding BREAKPOINT statement: {}", statement_str);
116  }
117  }
118  }
119  return statements;
120 }

122 void SympyConductanceVisitor::visit_binary_expression(ast::BinaryExpression& node) {
123  // only want binary expressions from breakpoint block
124  if (!under_breakpoint_block) {
125  return;
126  }
127  // only want binary expressions of form x = ...
128  if (node.get_lhs()->is_var_name() && (node.get_op().get_value() == BinaryOp::BOP_ASSIGN)) {
129  auto lhs_str =
130  std::dynamic_pointer_cast<ast::VarName>(node.get_lhs())->get_name()->get_node_name();
131  binary_expr_index[lhs_str] = ordered_binary_exprs.size();
132  ordered_binary_exprs.push_back(to_nmodl_for_sympy(node));
133  ordered_binary_exprs_lhs.push_back(lhs_str);
134  }
135 }

137 void SympyConductanceVisitor::lookup_nonspecific_statements() {
138  // add NONSPECIFIC_CURRENT statements to i_name map between write vars and names
139  // note that they don't have an ion name, so we set it to ""
140  if (!NONSPECIFIC_CONDUCTANCE_ALREADY_EXISTS) {
141  for (const auto& ns_curr_ast: nonspecific_nodes) {
142  logger->debug("SympyConductance :: Found NONSPECIFIC_CURRENT statement");
143  for (const auto& write_name:
144  std::dynamic_pointer_cast<const ast::Nonspecific>(ns_curr_ast)
145  .get()
146  ->get_currents()) {
147  const std::string& curr_write = write_name->get_node_name();
148  logger->debug("SympyConductance :: -> Adding non-specific current write name: {}",
149  curr_write);
150  i_name[curr_write] = "";
151  }
152  }
153  }
154 }

161 void SympyConductanceVisitor::lookup_useion_statements() {
162  // add USEION statements to i_name map between write vars and names
163  for (const auto& useion_ast: use_ion_nodes) {
164  const auto& ion = std::dynamic_pointer_cast<const ast::Useion>(useion_ast);
165  const std::string& ion_name = ion->get_node_name();
166  logger->debug("SympyConductance :: Found USEION statement {}", to_nmodl_for_sympy(*ion));
167  if (i_ignore.find(ion_name) != i_ignore.end()) {
168  logger->debug("SympyConductance :: -> Ignoring ion current name: {}", ion_name);
169  } else {
170  for (const auto& w: ion->get_writelist()) {
171  std::string ion_write = w->get_node_name();
172  logger->debug(
173  "SympyConductance :: -> Adding ion write name: {} for ion current name: {}",
174  ion_write,
175  ion_name);
176  i_name[ion_write] = ion_name;
177  }
178  }
179  }

182 void SympyConductanceVisitor::visit_conductance_hint(ast::ConductanceHint& node) {
183  // find existing CONDUCTANCE statements - do not want to alter them
184  // so keep a set of ion names i_ignore that we should ignore later
185  logger->debug("SympyConductance :: Found existing CONDUCTANCE statement: {}",
186  to_nmodl_for_sympy(node));
187  if (const auto& ion = node.get_ion()) {
188  logger->debug("SympyConductance :: -> Ignoring ion current name: {}", ion->get_node_name());
189  i_ignore.insert(ion->get_node_name());
190  } else {
191  logger->debug("SympyConductance :: -> Ignoring all non-specific currents");
192  NONSPECIFIC_CONDUCTANCE_ALREADY_EXISTS = true;
193  }
194 };

196 void SympyConductanceVisitor::visit_breakpoint_block(ast::BreakpointBlock& node) {
197  // return if it's not safe to insert conductance statements
198  if (!conductance_statement_possible(node)) {
199  logger->warn("SympyConductance :: Unsafe to insert CONDUCTANCE statement");
200  return;
201  }
202 
203  // add any breakpoint local variables to vars
204  if (auto* symtab = node.get_statement_block()->get_symbol_table()) {
205  for (const auto& localvar: symtab->get_variables_with_properties(NmodlType::local_var)) {
206  all_vars.insert(localvar->get_name());
207  }
208  }
209  // visit BREAKPOINT block statements
210  under_breakpoint_block = true;
211  node.visit_children(*this);
212  under_breakpoint_block = false;
213 
214  // lookup USEION and NONSPECIFIC statements from NEURON block
215  lookup_useion_statements();
216  lookup_nonspecific_statements();
217 
218  // add new CONDUCTANCE statements to BREAKPOINT
219  auto new_statements = generate_statement_strings(node);
220  if (!new_statements.empty()) {
221  // get a copy of existing BREAKPOINT statements
222  auto brkpnt_statements = node.get_statement_block()->get_statements();
223  // insert new CONDUCTANCE statements at top of BREAKPOINT
224  // or just below LOCAL statement if it exists
225  auto insertion_point = brkpnt_statements.begin();
226  while ((*insertion_point)->is_local_list_statement()) {
227  ++insertion_point;
228  }
229  for (const auto& statement_str: new_statements) {
230  insertion_point = brkpnt_statements.insert(insertion_point,
231  create_statement(statement_str));
232  }
233  // replace old set of BREAKPOINT statements in AST with new one
234  node.get_statement_block()->set_statements(std::move(brkpnt_statements));
235  }
236 }

245 }
